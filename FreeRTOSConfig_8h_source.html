134/*------------- CMSIS-RTOS V2 specific defines -----------*/
135/* When using CMSIS-RTOSv2 set configSUPPORT_STATIC_ALLOCATION to 1
136 * is mandatory to avoid compile errors.
137 * CMSIS-RTOS V2 implmentation requires the following defines
138 *
139#define configSUPPORT_STATIC_ALLOCATION 1 <-- cmsis_os threads are
140 created using xTaskCreateStatic() API
141#define configMAX_PRIORITIES (56) <-- Priority range in CMSIS-RTOS V2 is [0 ..
142 56]
143#define configUSE_PORT_OPTIMISED_TASK_SELECTION 0 <-- when set to 1,
144 configMAX_PRIORITIES can't be more than 32 which is not suitable for the new
145 CMSIS-RTOS v2 priority range
146*/

163#ifdef __NVIC_PRIO_BITS
164/* __BVIC_PRIO_BITS will be specified when CMSIS is being used. */
165#define configPRIO_BITS __NVIC_PRIO_BITS
166#else
167#define configPRIO_BITS 4 /* 15 priority levels */
168#endif
169
170/* The lowest interrupt priority that can be used in a call to a "set priority"
171function. */
172#define configLIBRARY_LOWEST_INTERRUPT_PRIORITY 0xf
173
174/* The highest interrupt priority that can be used by any interrupt service
175routine that makes calls to interrupt safe FreeRTOS API functions. DO NOT CALL
176INTERRUPT SAFE FREERTOS API FUNCTIONS FROM ANY INTERRUPT THAT HAS A HIGHER
177PRIORITY THAN THIS! (higher priorities are lower numeric values. */
178#define configLIBRARY_MAX_SYSCALL_INTERRUPT_PRIORITY 5
179
180/* Interrupt priorities used by the kernel port layer itself. These are generic
181to all Cortex-M ports, and do not rely on any particular library functions. */
182#define configKERNEL_INTERRUPT_PRIORITY \
183 (configLIBRARY_LOWEST_INTERRUPT_PRIORITY << (8 - configPRIO_BITS))
184/* !!!! configMAX_SYSCALL_INTERRUPT_PRIORITY must not be set to zero !!!!
185See http://www.FreeRTOS.org/RTOS-Cortex-M3-M4.html. */
186#define configMAX_SYSCALL_INTERRUPT_PRIORITY \
187 (configLIBRARY_MAX_SYSCALL_INTERRUPT_PRIORITY << (8 - configPRIO_BITS))

189/* Normal assert() semantics without relying on the provision of an assert.h
190header file. */
191#define configASSERT(x) \
192 if ((x) == 0) \
193 { \
194 taskDISABLE_INTERRUPTS(); \
195 for (;;) \
196 ; \
197 }
