167// DHAB S/134 current sensor conversion
168static int16_t isense_to_current(uint16_t isense_raw) {
169 static constexpr float ADC_VREF = 3.3f;
170 static constexpr float ADC_MAX = 4095.0f;
171
172 static constexpr float DIV_R1 = 2400.0f;
173 static constexpr float DIV_R2 = 4700.0f;
174
175 static constexpr float V_OFFSET = 2.5f;
176 static constexpr float G = 10.0e-3f;
177
178 float divider_gain = (DIV_R1 + DIV_R2) / DIV_R2;
179 float v_adc = isense_raw * ADC_VREF / ADC_MAX;
180 float v_sensor = v_adc * divider_gain;
181 float current = (v_sensor - V_OFFSET) / G;
182 float scaled_current = current * PACK_COEFF_PACK_STATS_PACK_CURRENT;
183
184 return (int16_t)scaled_current;
185}
