97bool adbms_read_and_check_rega(adbms_bms_t *bms) {
98 strbuf_clear(&bms->tx_strbuf);
99 adbms6380_prepare_command(&bms->tx_strbuf, RDCFGA);
100 adbms6380_read_result_t rdcfga_result =
101 adbms6380_read_data_with_retries(bms->spi,
102 ADBMS_PEC_FAIL_MAX_RETRIES,
103 ADBMS_MODULE_COUNT,
104 bms->tx_strbuf.data,
105 bms->rx_buf);
106 switch (rdcfga_result) {
107 case ADBMS6380_READ_SUCCESS:
108 // continue to check for mismatch
109 bms->err_rega_pec = false;
110 break;
111 case ADBMS6380_READ_PEC_FAILURE:
112 bms->err_rega_pec = true;
113 return false;
114 case ADBMS6380_READ_SPI_FAILURE:
115 bms->err_spi = true;
116 return false;
117 }
118 bms->err_rega_mismatch = false;
119 for (size_t i = 0; i < ADBMS_MODULE_COUNT; i++) {
120 uint8_t *module_data = &bms->rx_buf[i * ADBMS6380_SINGLE_DATA_PKT_SIZE];
121 if (memcmp(&module_data[0], bms->modules[i].rega, ADBMS6380_SINGLE_DATA_RAW_SIZE) != 0) {
122 bms->err_rega_mismatch = true;
123 bms->modules[i].err_rega_mismatch = true;
124 } else {
125 bms->modules[i].err_rega_mismatch = false;

211void adbms_calculate_balance_cells(adbms_bms_t *bms, float min_voltage, float min_delta) {
212 if (!bms->is_discharge_enabled) {
213 // Disable all discharging
214 for (size_t i = 0; i < ADBMS_MODULE_COUNT; i++) {
215 for (size_t j = 0; j < ADBMS6380_CELL_COUNT; j++) {
216 bms->modules[i].is_discharging[j] = false;
217 }
218 }
219 return;
220 }
221
222 if (bms->min_voltage < min_voltage) {
223 for (size_t i = 0; i < ADBMS_MODULE_COUNT; i++) {
224 for (size_t j = 0; j < ADBMS6380_CELL_COUNT; j++) {
225 bms->modules[i].is_discharging[j] = false;
226 }
227 }
228
229 return;
230 } else {
231 float balance_threshold = bms->min_voltage + min_delta;
232 for (size_t i = 0; i < ADBMS_MODULE_COUNT; i++) {
233 for (size_t j = 0; j < ADBMS6380_CELL_COUNT; j++) {
234 if (bms->modules[i].cell_voltages[j] > balance_threshold) {
235 bms->modules[i].is_discharging[j] = true;
236 } else {
237 bms->modules[i].is_discharging[j] = false;
238 }
239 }
240 }

257void adbms_read_cells(adbms_bms_t *bms) {
258 float *cell_voltage_ptrs[ADBMS_MODULE_COUNT] = {0};
259 int16_t *cell_voltages_raw_ptrs[ADBMS_MODULE_COUNT] = {0};
260 for (size_t i = 0; i < ADBMS_MODULE_COUNT; i++) {
261 cell_voltage_ptrs[i] = bms->modules[i].cell_voltages;
262 cell_voltages_raw_ptrs[i] = bms->modules[i].cell_voltages_raw;
263 }
264
265 if (!adbms6380_read_cell_voltages(bms->spi,
266 &bms->tx_strbuf,
267 bms->rx_buf,
268 cell_voltage_ptrs,
269 cell_voltages_raw_ptrs,
270 bms->err_cell_voltage_pecs,
271 ADBMS_MODULE_COUNT,
272 ADBMS_PEC_FAIL_MAX_RETRIES)) {
273 bms->state = ADBMS_STATE_IDLE;
274 bms->err_spi = true;
275 return;
276 }
277
278 // Calculate min/avg/max for each module and overall pack
279 bms->max_voltage = bms->modules[0].cell_voltages[0];
280 bms->min_voltage = bms->modules[0].cell_voltages[0];
281 bms->sum_voltage = 0.0f;
282 for (size_t i = 0; i < ADBMS_MODULE_COUNT; i++) {
283 bms->modules[i].max_voltage = bms->modules[i].cell_voltages[0];
284 bms->modules[i].min_voltage = bms->modules[i].cell_voltages[0];
285 bms->modules[i].sum_voltage = 0.0f;
286 for (size_t j = 0; j < ADBMS6380_CELL_COUNT; j++) {
287 float cell_v = cell_voltage_ptrs[i][j];
288 if (cell_v > bms->modules[i].max_voltage) {
289 bms->modules[i].max_voltage = cell_v;
290 }
291 if (cell_v < bms->modules[i].min_voltage) {
292 bms->modules[i].min_voltage = cell_v;
293 }
294 bms->modules[i].sum_voltage += cell_v;
295 }
296 bms->modules[i].avg_voltage = bms->modules[i].sum_voltage / ADBMS6380_CELL_COUNT;
297
298 if (bms->modules[i].max_voltage > bms->max_voltage) {
299 bms->max_voltage = bms->modules[i].max_voltage;
300 }
301 if (bms->modules[i].min_voltage < bms->min_voltage) {
302 bms->min_voltage = bms->modules[i].min_voltage;
303 }
304 bms->sum_voltage += bms->modules[i].sum_voltage;
305 }
306 bms->avg_voltage = bms->sum_voltage / (ADBMS_MODULE_COUNT * ADBMS6380_CELL_COUNT);
307}

309void adbms_read_therms(adbms_bms_t *bms) {
310 float *gpio_voltage_ptrs[ADBMS_MODULE_COUNT] = {0};
311 for (size_t i = 0; i < ADBMS_MODULE_COUNT; i++) {
312 gpio_voltage_ptrs[i] = bms->modules[i].therms_voltages;
313 }
314
315 // Start ADAX
316 strbuf_clear(&bms->tx_strbuf);
317 uint8_t adax_cmd[2] = {0};
318 adbms6380_adax(adax_cmd, ADBMS_ADAX_OW, ADBMS_ADAX_PUP, ADBMS_ADAX_CH);
319 adbms6380_prepare_command(&bms->tx_strbuf, adax_cmd);
320 adbms6380_set_cs_low(bms->spi);
321 if (!PHAL_SPI_transfer_noDMA(bms->spi, bms->tx_strbuf.data, bms->tx_strbuf.length, 0, NULL)) {
322 adbms6380_set_cs_high(bms->spi);
323 bms->err_spi = true;
324 return;
325 }
326 adbms6380_set_cs_high(bms->spi);
327
328 if (!adbms6380_read_gpio_voltages(bms->spi,
329 &bms->tx_strbuf,
330 bms->rx_buf,
331 gpio_voltage_ptrs,
332 bms->err_gpio_voltage_pecs,
333 ADBMS_MODULE_COUNT,
334 ADBMS_PEC_FAIL_MAX_RETRIES)) {
335 bms->state = ADBMS_STATE_IDLE;
336 bms->err_spi = true;
337 return;
338 }
339
340 // Convert all GPIO voltages to temperatures
341 // GPIO voltage -> R2 in the thermistor divider -> temperature via thermistor curve
342 for (size_t i = 0; i < ADBMS_MODULE_COUNT; i++) {
343 for (size_t j = 0; j < ADBMS6380_GPIO_COUNT; j++) {
344 float v_out = bms->modules[i].therms_voltages[j];
345 // Avoid divide by 0:
346 if (v_out == ADBMS_GPIO_VIN) {
347 v_out = ADBMS_GPIO_VIN - 0.00001f;
348 }
349 float r2 = (v_out * ADBMS_GPIO_R1) / (ADBMS_GPIO_VIN - v_out);
350 bms->modules[i].therms_temps[j] = thermistor_R_to_T(r2);
351 }
352 }
353
354 // Calculate min/avg/max thermistor temps for each module and overall
355 bms->max_therm_temp = bms->modules[0].therms_temps[0];
356 bms->min_therm_temp = bms->modules[0].therms_temps[0];
357 float bms_sum_therm_temp = 0.0f;
358 for (size_t i = 0; i < ADBMS_MODULE_COUNT; i++) {
359 bms->modules[i].max_therm_temp = bms->modules[i].therms_temps[0];
360 bms->modules[i].min_therm_temp = bms->modules[i].therms_temps[0];
361 float module_sum_therm_temp = 0.0f;
362 for (size_t j = 0; j < ADBMS6380_GPIO_COUNT; j++) {
363 float t = bms->modules[i].therms_temps[j];
364 if (t > bms->modules[i].max_therm_temp) {
365 bms->modules[i].max_therm_temp = t;
366 }
367 if (t < bms->modules[i].min_therm_temp) {
368 bms->modules[i].min_therm_temp = t;
369 }
370 module_sum_therm_temp += t;
371 }
372 bms->modules[i].avg_therm_temp = module_sum_therm_temp / ADBMS6380_GPIO_COUNT;
373 if (bms->modules[i].max_therm_temp > bms->max_therm_temp) {
374 bms->max_therm_temp = bms->modules[i].max_therm_temp;
375 }
376 if (bms->modules[i].min_therm_temp < bms->min_therm_temp) {
377 bms->min_therm_temp = bms->modules[i].min_therm_temp;
378 }
379 bms_sum_therm_temp += module_sum_therm_temp;
380 }
381 bms->avg_therm_temp = bms_sum_therm_temp / (ADBMS_MODULE_COUNT * ADBMS6380_GPIO_COUNT);
382}
