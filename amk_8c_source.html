79void AMK_periodic(AMK_t *amk) {
80 amk->state = amk->next_state;
81 amk->next_state = amk->state; // default: stay in current state
82
83 bool is_ready = *(amk->precharge_ptr) && amk->info->AMK_Status_bSystemReady;
84 bool is_bError = amk->info->AMK_Status_bError;
85 bool is_simple_error = (amk->err1->AMK_DiagnosticNumber == AMK_CAN_ERR_ID
86 || amk->err1->AMK_DiagnosticNumber == AMK_DC_BUS_ID);
87
88 switch (amk->state) {
89 case AMK_STATE_OFF:
90 if (is_bError && is_simple_error) {
91 AMK_reset(amk);
92 } else if (!is_bError && is_ready) {
93 amk->next_state = AMK_STATE_INIT;
94 }
95 break;
96
97 case AMK_STATE_INIT:
98 if (!is_ready) {
99 AMK_stop(amk);
100 amk->next_state = AMK_STATE_OFF;
101 break;
102 }
103
104 amk->set->AMK_TorqueSetpoint = 0;
105 amk->set->AMK_PositiveTorqueLimit = 0;
106 amk->set->AMK_NegativeTorqueLimit = 0;
107 amk->set->AMK_Control_bDcOn = true;
108 amk->set->AMK_Control_bInverterOn = true;
109 amk->set->AMK_Control_bEnable = true;
110 amk->next_state = AMK_STATE_RUNNING;
111 break;
112
113 case AMK_STATE_RUNNING:
114 if (!is_ready || is_bError) {
115 AMK_stop(amk);
116 amk->next_state = AMK_STATE_OFF;
117 break;
118 }
119
120 amk->set->AMK_PositiveTorqueLimit = AMK_DEFAULT_POS_LIMIT;
121 amk->set->AMK_NegativeTorqueLimit = AMK_DEFAULT_NEG_LIMIT;
122 break;
123 }
124
125 // flush the internal state to the CAN library
126 amk->set_function();
127 amk->log_function();
128
129 // clear error reset
130 amk->set->AMK_Control_bErrorReset = false;
131}
