29// Called periodically, Calculates current through each switch in mA
30void updateCurrent() {
31 // High power switches
32 auto_switches.current[SW_PUMP_1] = calcCurrent_HP(adc_readings.pump_1_imon);
33 auto_switches.current[SW_PUMP_2] = calcCurrent_HP(adc_readings.pump_2_imon);

35 auto_switches.current[SW_AUX] = calcCurrent_HP(adc_readings.aux_hp_imon);

38 auto_switches.current[SW_FAN_1] = calcCurrent_LP(adc_readings.fan_1_cs);
39 auto_switches.current[SW_FAN_2] = calcCurrent_LP(adc_readings.fan_2_cs);
40 auto_switches.current[SW_DASH] = calcCurrent_LP(adc_readings.dash_cs);
41 auto_switches.current[SW_ABOX] = calcCurrent_LP(adc_readings.abox_cs);
42 auto_switches.current[SW_MAIN] = calcCurrent_LP(adc_readings.main_cs);
43
44 // Upstream CS
45 calcCurrent_Total();

56// Current helper functions
57uint16_t calcCurrent_HP(uint16_t current) {
58 current = current * ADC_REF_mV / ADC_MAX; // Convert to mV
59 current = current * (HP_CS_R1 + HP_CS_R2) / HP_CS_R2;
60 current = current * HP_CS_R3 / (HP_CS_R1 + HP_CS_R2);
61 return current;
62}

69// CS signals for upstream 24V and 5V (total current through each)
70void calcCurrent_Total() {
71 // 24V current
72 uint16_t current = adc_readings.lv_24_i_sense;
73 current = current * ADC_REF_mV / ADC_MAX; // Convert to mV
74 current = current / HP_CS_R_SENSE / CS_GAIN;
75 auto_switches.current[CS_24V] = current;
76
77 // 5V current
78 current = adc_readings.lv_5_i_sense;
79 current = current * ADC_REF_mV / ADC_MAX; // Convert to mA
80 auto_switches.current[CS_5V] = current;
81}

83// Converts ADC voltage reading to mV
84uint16_t calcVoltage(uint16_t voltage, int r1, int r2) {
85 // Compensate for voltage divider
86 voltage = voltage * (r1 + r2) / r2;
87 // Scale for 12-bit adc at 3.3v
88 voltage = voltage * ADC_REF_mV / ADC_MAX;
89 return voltage;
90}
