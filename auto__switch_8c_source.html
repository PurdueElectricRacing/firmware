34// Called periodically, Calculates current through each switch in mA
35void updateCurrent() {
36 // High power switches (direct ADC)
37 auto_switches.current[SW_PUMP_1] = calcCurrent_HP(adc_readings.pump_1_imon);
38 auto_switches.current[SW_PUMP_2] = calcCurrent_HP(adc_readings.pump_2_imon);
39 auto_switches.current[SW_SDC] = calcCurrent_HP(adc_readings.sdc_imon);
40 auto_switches.current[SW_HXFAN] = calcCurrent_HP(adc_readings.hxfan_imon);
41
42 // High power switches (mux-sensed)
43 auto_switches.current[SW_FAN_1] = calcCurrent_HP(mux_readings[0]);
44 auto_switches.current[SW_FAN_2] = calcCurrent_HP(mux_readings[1]);
45 auto_switches.current[SW_FAN_3] = calcCurrent_HP(mux_readings[2]);
46 auto_switches.current[SW_FAN_4] = calcCurrent_HP(mux_readings[3]);
47 auto_switches.current[SW_AMK1] = calcCurrent_HP(mux_readings[4]);
48 auto_switches.current[SW_AMK2] = calcCurrent_HP(mux_readings[5]);
49
50 // Low power switches
51 auto_switches.current[SW_DASH] = calcCurrent_LP(adc_readings.dash_cs);
52 auto_switches.current[SW_ABOX] = calcCurrent_LP(adc_readings.abox_cs);
53 auto_switches.current[SW_MAIN] = calcCurrent_LP(adc_readings.main_cs);
54 auto_switches.current[SW_DLFR] = calcCurrent_LP(adc_readings.dlfr_cs);
55 auto_switches.current[SW_DLBK] = calcCurrent_LP(adc_readings.dlbk_cs);
56
57 // Upstream CS
58 calcCurrent_Total();
59}

82static void setMuxChannel(uint8_t channel) {
83 PHAL_writeGPIO(MUX_CTRL_A_GPIO_Port, MUX_CTRL_A_Pin, (channel & 0x01U) != 0U);

88// Current helper functions
89uint16_t calcCurrent_HP(uint16_t current) {
90 current = current * ADC_REF_mV / ADC_MAX; // Convert to mV
91 current = current * (HP_CS_R1 + HP_CS_R2) / HP_CS_R2;
92 current = current * HP_CS_R3 / (HP_CS_R1 + HP_CS_R2);
93 return current;
94}
95
96uint16_t calcCurrent_LP(uint16_t current) {
97 current = current * ADC_REF_mV / ADC_MAX; // Convert to mA
98 return current;
99}

101// CS signals for upstream 24V and 5V (total current through each)
102void calcCurrent_Total() {
103 // 24V current
104 uint16_t current = adc_readings.v24_cs;
105 current = current * ADC_REF_mV / ADC_MAX; // Convert to mV
106 current = current / HP_CS_R_SENSE / CS_GAIN;
107 auto_switches.current[CS_24V] = current;
108
109 // 5V current
110 current = adc_readings.v5_cs;
111 current = current * ADC_REF_mV / ADC_MAX; // Convert to mA
112 auto_switches.current[CS_5V] = current;
113}

115// Converts ADC voltage reading to mV
116uint16_t calcVoltage(uint16_t voltage, int r1, int r2) {
117 // Compensate for voltage divider
118 voltage = voltage * (r1 + r2) / r2;
119 // Scale for 12-bit adc at 3.3v
120 voltage = voltage * ADC_REF_mV / ADC_MAX;
121 return voltage;
122}
