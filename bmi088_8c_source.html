114static inline float BMI088_getGyroScale(BMI088_Handle_t *bmi, bool range_up, bool range_down) {
115 float scale = 0.0;
116 switch (bmi->gyro_range) {
117 case (GYRO_RANGE_2000):
118 scale = (16.384 / DEG_TO_RAD);
119 break;
120 case (GYRO_RANGE_1000):
121 scale = (32.768 / DEG_TO_RAD);
122 break;
123 case (GYRO_RANGE_500):
124 scale = (65.536 / DEG_TO_RAD);
125 if (range_down) {
126 bmi->gyro_range = GYRO_RANGE_250;
127 PHAL_SPI_writeByte(bmi->spi, BMI088_GYRO_RANGE_ADDR, bmi->gyro_range);
128 }
129 break;
130 case (GYRO_RANGE_250):
131 scale = (131.072 / DEG_TO_RAD);
132
133 if (range_up) {
134 bmi->gyro_range = GYRO_RANGE_500;
135 PHAL_SPI_writeByte(bmi->spi, BMI088_GYRO_RANGE_ADDR, bmi->gyro_range);
136 } else if (range_down) {
137 bmi->gyro_range = GYRO_RANGE_125;
138 PHAL_SPI_writeByte(bmi->spi, BMI088_GYRO_RANGE_ADDR, bmi->gyro_range);
139 }
140 break;
141 case (GYRO_RANGE_125):
142 scale = (262.144 / DEG_TO_RAD);
143
144 if (range_up) {
145 bmi->gyro_range = GYRO_RANGE_250;
146 PHAL_SPI_writeByte(bmi->spi, BMI088_GYRO_RANGE_ADDR, bmi->gyro_range);
147 }
148 break;
149 default:
150 scale = 1.0; // prevent div by zero
151 }
152 return scale;
153}

155bool BMI088_readGyro(BMI088_Handle_t *bmi) {
156 uint8_t *tx_buffer = bmi->bmi_tx_buffer;
157 uint8_t *rx_buffer = bmi->bmi_rx_buffer;
158
159 BMI088_selectGyro(bmi);
160 while (PHAL_SPI_busy(bmi->spi)) {
161 __asm__("nop");
162 }
163
164 tx_buffer[0] = (1 << 7) | BMI088_GYRO_RATE_X_LSB_ADDR;
165 PHAL_SPI_transfer_noDMA(bmi->spi, tx_buffer, 1, 7, rx_buffer);
166 // while (PHAL_SPI_busy(bmi->spi))
167 // ;
168 int16_t raw_x, raw_y, raw_z;
169 raw_x = (((int16_t)rx_buffer[2]) << 8) | rx_buffer[1];
170 raw_y = (((int16_t)rx_buffer[4]) << 8) | rx_buffer[3];
171 raw_z = (((int16_t)rx_buffer[6]) << 8) | rx_buffer[5];
172
173 int16_t max_raw = MAX(MAX(raw_x, raw_y), raw_z);
174 bool range_up = bmi->enableDynamicRange && (ABS(max_raw) >= 32000); // int16_t range is -32,768 to +32,767
175 bool range_down = bmi->enableDynamicRange && (ABS(max_raw) <= 1000);
176
177 // Convert raw values into physical values based on range
178 // Decimal is fixed in the first place
179 float scale = BMI088_getGyroScale(bmi, range_up, range_down);
180
181 bmi->data.gyro_x = raw_x / scale;
182 bmi->data.gyro_y = raw_y / scale;
183 bmi->data.gyro_z = raw_z / scale;
184
185 // Write the CSB state to high to deselect the gyro
186 PHAL_writeGPIO(SPI_CS_GYRO_GPIO_Port, SPI_CS_GYRO_Pin, 1);
187 return true;
188}

190bool BMI088_readAccel(BMI088_Handle_t *bmi) {
191 uint8_t *tx_buffer = bmi->bmi_tx_buffer;
192 uint8_t *rx_buffer = bmi->bmi_rx_buffer;
193
194 BMI088_selectAccel(bmi);
195 while (PHAL_SPI_busy(bmi->spi)) {

198
199 tx_buffer[0] = (1 << 7) | BMI088_ACC_RATE_X_LSB_ADDR;
200 PHAL_SPI_transfer_noDMA(bmi->spi, tx_buffer, 1, 7, rx_buffer);
201 // while (PHAL_SPI_busy(bmi->spi))
202 // ;
203 int16_t raw_ax, raw_ay, raw_az;
204 raw_ax = (((int16_t)rx_buffer[2 + 1]) << 8) | rx_buffer[1 + 1];
205 raw_ay = (((int16_t)rx_buffer[4 + 1]) << 8) | rx_buffer[3 + 1];
206 raw_az = (((int16_t)rx_buffer[6 + 1]) << 8) | rx_buffer[5 + 1];
207
208 // Conversion taken from datasheet pg 22.
209
210 // NOTE - changed x and y to correspond to data sheet

212 bmi->data.accel_y = (float)(raw_ay << (bmi->accel_range + 1)) / 32768.0f * G_TO_M_S * 1.5f;
213 bmi->data.accel_z = (float)(raw_az << (bmi->accel_range + 1)) / 32768.0f * G_TO_M_S * 1.5f;

216 PHAL_writeGPIO(SPI_CS_ACEL_GPIO_Port, SPI_CS_ACEL_Pin, 1);
