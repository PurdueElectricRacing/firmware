98static void BL_JumptoApplication(void) {
99 uint32_t app_reset_handler_address = *(uint32_t*)(((void*)&_eboot_flash + 4));
100 uint32_t msp = (uint32_t)*((uint32_t*)(((void*)&_eboot_flash)));
101 uint32_t estack = (uint32_t)((uint32_t*)(((void*)&_estack)));
102
103 // Confirm app exists
104 if (app_reset_handler_address == 0xFFFFFFFF || app_reset_handler_address <= BL_ADDRESS_CRC || msp != estack)
105 return;
106
107 // Reset all of our used peripherals
108 RCC->AHB1RSTR |= RCC_AHB1RSTR_CRCRST;
109 RCC->AHB1RSTR &= ~(RCC_AHB1RSTR_CRCRST);
110#if defined(STM32F407xx) || defined(STM32F732xx)
111 RCC->APB1RSTR |= RCC_APB1RSTR_CAN1RST;
112 RCC->APB1RSTR &= ~(RCC_APB1RSTR_CAN1RST);
113#else
114 RCC->APB1RSTR1 |= RCC_APB1RSTR1_CAN1RST;
115 RCC->APB1RSTR1 &= ~(RCC_APB1RSTR1_CAN1RST);
116#endif
117 SysTick->CTRL = 0;
118 SysTick->LOAD = 0;
119 SysTick->VAL = 0;
120
121 // Make sure the interrupts are disabled before we start attempting to jump to the app
122 // Getting an interrupt after we set VTOR would be bad.
123 // Actually jump to application
124 __disable_irq();
125 __set_MSP(msp);
126 SCB->VTOR = (uint32_t)(uint32_t*)(((void*)&_eboot_flash));
127 __enable_irq();
128 ((void (*)(void))app_reset_handler_address)(); // jump
129}

131void BL_checkAndBoot(void) {
132 // Check 16K metadata region to decide what to boot
133 uint32_t crc_stored = PHAL_flashReadU32(BL_ADDRESS_CRC_CRC);
134 uint32_t addr = PHAL_flashReadU32(BL_ADDRESS_CRC_ADDR);
135 uint32_t size = PHAL_flashReadU32(BL_ADDRESS_CRC_SIZE);
136 if (crc_stored && size && (size < MAX_FIRMWARE_SIZE) && (addr >= FLASH_BASE) && (addr <= FLASH_END)) {
137 if (PHAL_CRC32_Calculate((uint32_t*)addr, size / 4) == crc_stored) {
138 BL_JumptoApplication();
139 }
140 }
141}

143static int BL_processCommand_Start(uint32_t size) {
144 firmware_size_total = 0;
145
146 // TODO store CRC + size at the end of firmware
147 if (!size || size >= MAX_FIRMWARE_SIZE || size & 3) {
148 BL_sendStatusMessage(BLSTAT_INVALID, BLERROR_SIZE);
149 return -1;
150 }
151
152 if (PHAL_flashErase((uint32_t*)BL_ADDRESS_BUFFER, size / 4) != FLASH_OK) {
153 BL_sendStatusMessage(BLSTAT_INVALID, BLERROR_FLASH);
154 return -1;
155 }
156
157 buffer_index = 0;
158 firmware_size_total = size;
159 BL_sendStatusMessage(BLSTAT_VALID, firmware_size_total);
160 bl_unlock = true;
161 return 0;
162}

164static int BL_memcpy_buffer(uint32_t addr_dst, uint32_t addr_src, uint32_t size) {
165 if (PHAL_flashErase((uint32_t*)addr_dst, size / 4) != FLASH_OK)
166 return -1;
167
168 for (uint32_t i = 0; i < size / 4; i++) {
169 uint32_t offset = i * sizeof(uint32_t);
170 if (PHAL_flashWriteU32(addr_dst + offset, *(__IO uint32_t*)(addr_src + offset)) != FLASH_OK)
171 return -1;
172 }
173
174 return 0;
175}

177static int BL_setCRCConfig(uint32_t crc, uint32_t addr, uint32_t size) {
178 int ret;
179
180 PHAL_flashErase((uint32_t*)BL_ADDRESS_CRC, 3);
181
182 if (PHAL_flashWriteU32(BL_ADDRESS_CRC_CRC, crc) != FLASH_OK) {
183 ret = -1;
184 goto erase_crc;
185 }
186 if (PHAL_flashWriteU32(BL_ADDRESS_CRC_ADDR, addr) != FLASH_OK) {
187 ret = -1;
188 goto erase_crc;
189 }
190 if (PHAL_flashWriteU32(BL_ADDRESS_CRC_SIZE, size) != FLASH_OK) {
191 ret = -1;
192 goto erase_crc;
193 }
194
195 ret = 0;
196 goto exit;
197
198erase_crc:
199 // erases whole sector so size doesnt matter
200 PHAL_flashErase((uint32_t*)BL_ADDRESS_CRC, 3); // nothing we can do on failure
201exit:
202 bl_unlock = false;
203 return ret;
204}

234static int BL_processCommand_CRC(uint32_t crc_app, uint32_t dst_addr) {
235 uint32_t size = firmware_size_total;
236 if (!size) {
237 BL_sendStatusMessage(BLSTAT_INVALID, BLERROR_SIZE);
238 return -1;
239 }
240
241 uint32_t crc_flash;
242 crc_flash = PHAL_CRC32_Calculate((uint32_t*)BL_ADDRESS_BUFFER, size / 4);
243 if (crc_flash != crc_app) {
244 BL_sendStatusMessage(BLSTAT_INVALID_CRC, crc_flash);
245 return -1;
246 }
247
248#ifdef BL_ENABLE_DOUBLE_BANK
249 BL_memcpy_buffer(dst_addr, BL_ADDRESS_BUFFER, size);
250 crc_flash = PHAL_CRC32_Calculate((uint32_t*)dst_addr, size / 4);
251 if (crc_flash != crc_app) {
252 BL_sendStatusMessage(BLSTAT_INVALID_CRC, crc_flash);
253 return -1;
254 }
255#endif // BL_ENABLE_DOUBLE_BANK
256
257 if (BL_setCRCConfig(crc_app, dst_addr, size)) {
258 BL_sendStatusMessage(BLSTAT_INVALID, BLERROR_FLASH);
259 return -1;
260 }
261
262 BL_sendStatusMessage(BLSTAT_VALID, crc_flash);
263 return 0;
264}

280void BL_processCommand(BLCmd_t cmd, uint32_t data) {
281 switch (cmd) {
282 case BLCMD_START:
283 BL_processCommand_Start(data);
284 break;
285 case BLCMD_CRC:
286 BL_processCommand_CRC(data, BL_ADDRESS_APP);
287 break;
288 case BLCMD_CRC_BACKUP:
289 BL_processCommand_CRC(data, BL_ADDRESS_BACKUP);
290 break;
291 case BLCMD_JUMP:
292 BL_processCommand_Jump(data);
293 break;
294 case BLCMD_RST:
295 BL_processCommand_Reset(data);
296 break;
297 default: {
298 BL_sendStatusMessage(BLSTAT_UNKNOWN_CMD, cmd);
299 break;
300 }
301 }
