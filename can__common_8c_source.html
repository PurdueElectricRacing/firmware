26void CAN_enqueue_tx(CanMsgTypeDef_t *msg) {
27 uint8_t mailbox;
28 uint8_t periph_idx = GET_PERIPH_IDX(msg->Bus);
29
30 if (msg->IDE != 0) {
31 // Extended ID: Use HLP bits (26-28) to determine priority mailbox
32 switch ((msg->ExtId >> 26) & 0b111) {
33 case 0:
34 case 1:
35 mailbox = CAN_MAILBOX_HIGH_PRIO;
36 break;
37 case 2:
38 case 3:
39 mailbox = CAN_MAILBOX_MED_PRIO;
40 break;
41 default:
42 mailbox = CAN_MAILBOX_LOW_PRIO;
43 break;
44 }
45 } else {
46 // Standard ID: Default to high priority
47 mailbox = CAN_MAILBOX_HIGH_PRIO;
48 }
49
50 if (qSendToBack(&q_tx_can[periph_idx][mailbox], msg) != SUCCESS_G) {
51 can_stats.can_peripheral_stats[periph_idx].tx_of++;
52 }
53}

128[[gnu::always_inline]]
129static inline bool CAN_prepare_filter_config(CAN_TypeDef *can) {
130 can->MCR |= CAN_MCR_INRQ; // Enter back into INIT state (required for changing scale)
131 uint32_t timeout = 0;
132 while (!(can->MSR & CAN_MSR_INAK) && ++timeout < PHAL_CAN_INIT_TIMEOUT)
133 ;
134 if (timeout == PHAL_CAN_INIT_TIMEOUT)
135 return false;
136
137 can->FMR |= CAN_FMR_FINIT; // Enter init mode for filter banks
138
139 return true;
140}
141
142[[gnu::always_inline]]
143static inline bool CAN_exit_filter_config(CAN_TypeDef *can) {
144 can->FMR &= ~CAN_FMR_FINIT; // Enable Filters (exit filter init mode)
145
146 // Enter back into NORMAL mode
147 can->MCR &= ~CAN_MCR_INRQ;
148 uint32_t timeout = 0;
149 while ((can->MSR & CAN_MSR_INAK) && ++timeout < PHAL_CAN_INIT_TIMEOUT)
150 ;
151
152 return timeout != PHAL_CAN_INIT_TIMEOUT;
153}

155bool CAN_library_init() {
156 for (uint8_t can_periph = 0; can_periph < NUM_CAN_PERIPHERALS; can_periph++) {
157 for (uint8_t mbx = 0; mbx < CAN_TX_MAILBOX_CNT; mbx++) {
158 qConstruct(&q_tx_can[can_periph][mbx], sizeof(CanMsgTypeDef_t));
159 can_mbx_last_send_time[can_periph][mbx] = 0;
160 }
161 }
162 qConstruct(&q_rx_can, sizeof(CanMsgTypeDef_t));
163 can_stats = (can_stats_t) {0};
164 CAN_data_init();
165
166#if defined(USE_CAN1) || defined(USE_CAN2)
167 // Ensure CAN1 clock is enabled (CAN1 owns all bxCAN filters)
168 RCC->APB1ENR |= RCC_APB1ENR_CAN1EN;
169 if (!CAN_prepare_filter_config(CAN1)) {
170 return false;
171 }
172
173 bxcan_set_filters(); // from NODE.h
174
175 if (!CAN_exit_filter_config(CAN1)) {
176 return false;
177 }
178#endif
179
180 return true;
181}
