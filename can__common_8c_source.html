53void CAN_enqueue_tx(CanMsgTypeDef_t *msg) {
54 uint8_t mailbox;
55 uint8_t periph_idx = GET_PERIPH_IDX(msg->Bus);
56
57 if (msg->IDE != 0) {
58 // Extended ID: Use HLP bits (26-28) to determine priority mailbox
59 switch ((msg->ExtId >> 26) & 0b111) {
60 case 0:
61 case 1:
62 mailbox = CAN_MAILBOX_HIGH_PRIO;
63 break;
64 case 2:
65 case 3:
66 mailbox = CAN_MAILBOX_MED_PRIO;
67 break;
68 default:
69 mailbox = CAN_MAILBOX_LOW_PRIO;
70 break;
71 }
72 } else {
73 // Standard ID: Default to high priority
74 mailbox = CAN_MAILBOX_HIGH_PRIO;
75 }
76
77 if (xQueueSendToBack(q_tx_can[periph_idx][mailbox], msg, pdMS_TO_TICKS(CAN_TX_BACKPRESSURE_MS)) != pdPASS) {
78 can_stats.can_peripheral_stats[periph_idx].tx_of++;
79 }
80}

82void CAN_tx_update() {
83 CanMsgTypeDef_t tx_msg;
84 for (uint8_t i = 0; i < CAN_TX_MAILBOX_CNT; ++i) {
85#ifdef USE_CAN1
86 // Handle CAN1
87 if (PHAL_txMailboxFree(CAN1, i)) {
88 if (xQueueReceive(q_tx_can[CAN1_IDX][i], &tx_msg, 0) == pdPASS) {
89 PHAL_txCANMessage(&tx_msg, i);
90 can_mbx_last_send_time[CAN1_IDX][i] = OS_TICKS;
91 }
92 } else if (OS_TICKS - can_mbx_last_send_time[CAN1_IDX][i] > CAN_TX_TIMEOUT_MS) {
93 PHAL_txCANAbort(CAN1, i);
94 can_stats.can_peripheral_stats[CAN1_IDX].tx_fail++;
95 }
96#endif
97
98#ifdef USE_CAN2
99 // Handle CAN2
100 if (PHAL_txMailboxFree(CAN2, i)) {
101 if (xQueueReceive(q_tx_can[CAN2_IDX][i], &tx_msg, 0) == pdPASS) {
102 PHAL_txCANMessage(&tx_msg, i);
103 can_mbx_last_send_time[CAN2_IDX][i] = OS_TICKS;
104 }
105 } else if (OS_TICKS - can_mbx_last_send_time[CAN2_IDX][i] > CAN_TX_TIMEOUT_MS) {
106 PHAL_txCANAbort(CAN2, i);
107 can_stats.can_peripheral_stats[CAN2_IDX].tx_fail++;

114inline void CAN_handle_irq(CAN_TypeDef *bus, uint8_t fifo) {
115 CanMsgTypeDef_t rx_msg;
116 while (PHAL_rxCANMessage(bus, fifo, &rx_msg)) {
117 BaseType_t xHigherPriorityTaskWoken = pdFALSE;
118 if (xQueueSendFromISR(q_rx_can, &rx_msg, &xHigherPriorityTaskWoken) != pdPASS) {
119 can_stats.rx_of++;
120 }
121 portYIELD_FROM_ISR(xHigherPriorityTaskWoken);
122 }
123}

146[[gnu::always_inline]]
147static inline bool CAN_prepare_filter_config(CAN_TypeDef *can) {
148 can->MCR |= CAN_MCR_INRQ; // Enter back into INIT state (required for changing scale)
149 uint32_t timeout = 0;
150 while (!(can->MSR & CAN_MSR_INAK) && ++timeout < PHAL_CAN_INIT_TIMEOUT)
151 ;
152 if (timeout == PHAL_CAN_INIT_TIMEOUT)
153 return false;
154
155 can->FMR |= CAN_FMR_FINIT; // Enter init mode for filter banks
156
157 return true;
158}
159
160[[gnu::always_inline]]
161static inline bool CAN_exit_filter_config(CAN_TypeDef *can) {
162 can->FMR &= ~CAN_FMR_FINIT; // Enable Filters (exit filter init mode)
163
164 // Enter back into NORMAL mode
165 can->MCR &= ~CAN_MCR_INRQ;
166 uint32_t timeout = 0;
167 while ((can->MSR & CAN_MSR_INAK) && ++timeout < PHAL_CAN_INIT_TIMEOUT)
168 ;
169
170 return timeout != PHAL_CAN_INIT_TIMEOUT;
171}
