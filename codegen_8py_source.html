52def generate_node_header(env, node: Node, context: SystemContext):
53 filename = GENERATED_DIR / f"{node.name}.h"
54 mapping = context.mappings.get(node.name)
55
56 # Collect all messages this node sees (TX and RX)
57 rx_msgs = []
58 tx_msgs = []
59
60 peripherals = sorted(list(set(bus.peripheral for bus in node.busses.values())))
61 node_busses = sorted(node.busses.keys())
62
63 for bus_name in node_busses:
64 bus = node.busses[bus_name]
65 for rx_msg in bus.rx_messages:
66 if rx_msg.resolved_message:
67 rx_msgs.append((rx_msg, bus.peripheral, bus_name))
68 for msg in bus.tx_messages:
69 tx_msgs.append((msg, bus.peripheral, bus_name))
70
71 render_template(env, 'node_header.h.jinja',
72 filename,
73 node=node,
74 context=context,
75 mapping=mapping,
76 rx_msgs=rx_msgs,
77 tx_msgs=tx_msgs,
78 peripherals=peripherals,
79 node_busses=node_busses)
80 print_as_ok(f"Generated {filename.name}")
