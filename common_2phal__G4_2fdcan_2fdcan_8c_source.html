27// -1 unsupported baud
28// -2 bad args or BRP out of range (1..512)
29// -3 ker_hz not an EXACT multiple of (baud * TQ)
30static int PHAL_FDCAN_makeNBTP(uint32_t ker_hz, uint32_t baud_bps, uint32_t *nbtp_out) {
31 if (!nbtp_out || ker_hz == 0u)
32 return -2;
33
34 uint8_t tq, seg1, seg2, sjw;
35
36 switch (baud_bps) {
37 case 125000u: // 16 TQ, SP=87.5%
38 case 250000u:
39 case 500000u:
40 case 1000000u:
41 tq = 16;
42 seg1 = 13;
43 seg2 = 2;
44 sjw = 2;
45 break;
46
47 case 2000000u: // 8 TQ, SP≈62.5% (1 + 4)/8
48 tq = 8;
49 seg1 = 4;
50 seg2 = 3;
51 sjw = 2;
52 break;
53
54 default:
55 return -1; // unsupported
56 }
57
58 if (sjw > seg2)
59 sjw = seg2; // hardware requires SJW ≤ TSEG2
60
61 // BRP = ker_hz / (baud * tq) (must be integer)
62 uint64_t denom = (uint64_t)baud_bps * (uint64_t)tq;
63 if (denom == 0u)
64 return -2;
65
66 uint32_t brp = (uint32_t)(ker_hz / denom);
67 if (brp < 1u || brp > 512u)
68 return -2;
69
70 // Enforce exact match so the nominal bit-rate is precise
71 if ((uint64_t)brp * denom != (uint64_t)ker_hz) {
72 return -3; // pick a different ker_hz (or use CKDIV) or another TQ shape
73 }
74
75 *nbtp_out = ((uint32_t)(brp - 1u) << FDCAN_NBTP_NBRP_Pos)
76 | ((uint32_t)(seg1 - 1u) << FDCAN_NBTP_NTSEG1_Pos)
77 | ((uint32_t)(seg2 - 1u) << FDCAN_NBTP_NTSEG2_Pos)
78 | ((uint32_t)(sjw - 1u) << FDCAN_NBTP_NSJW_Pos);
79
80 return 0;
81}

83static void
84PHAL_FDCAN_setExtendedFilter(FDCAN_GlobalTypeDef *fdcan, uint32_t num_xid, uint32_t *xid_list) {
85 if (!num_xid || num_xid > MAX_NUM_XID_FILTER) {
86 return;
87 }
88
89 volatile uint32_t *ram = (volatile uint32_t *)(PHAL_FDCAN_get_ram_base(fdcan) + SRAMCAN_FLESA);
90 // Program num_xid filter elements, 2 words each
91 for (uint32_t i = 0; i < num_xid; i++) {
92 uint32_t xid = xid_list[i] & 0x1FFFFFFFu;
93 // Word 0: EFEC=001 (store to FIFO0), EFID1 = id
94 ram[i * 2 + 0] = (1u << 29) | xid;
95 // Word 1: EFT=10 (classic mask), mask=all ones -> exact match
96 ram[i * 2 + 1] = (2u << 30) | 0x1FFFFFFFu;
97 }
98
99 // Set number of extended filters
100 fdcan->RXGFC &= ~FDCAN_RXGFC_LSE_Msk;
101 fdcan->RXGFC |= ((num_xid & 0x7u) << FDCAN_RXGFC_LSE_Pos);
102
103 fdcan->XIDAM = 0x1FFFFFFFu;
104}

106static void
107PHAL_FDCAN_setStandardFilter(FDCAN_GlobalTypeDef *fdcan, uint32_t num_sid, uint32_t *sid_list) {
108 if (!num_sid || num_sid > MAX_NUM_SID_FILTER) {
109 return;
110 }
111
112 volatile uint32_t *ram = (volatile uint32_t *)(PHAL_FDCAN_get_ram_base(fdcan) + SRAMCAN_FLSSA);
113
114 // Program standard filters (1 word each)
115 for (uint32_t i = 0; i < num_sid; i++) {
116 uint32_t sid = sid_list[i] & 0x7FFu; // 11-bit
117 // exact match (mask = 0x7FF), store to FIFO0
118 // SFT=10 (classic mask), SFEC=001 (FIFO0), SFID1 = sid, SFID2 = 0x7FF
119 ram[i] = (2u << 30) | (1u << 27) | (sid << 16) | 0x7FFu;
120 }
121 // Set number of standard filters
122 fdcan->RXGFC &= ~(FDCAN_RXGFC_LSS_Msk);
123 fdcan->RXGFC |= ((num_sid & 0xFFu) << FDCAN_RXGFC_LSS_Pos);
124}

126void PHAL_FDCAN_setFilters(FDCAN_GlobalTypeDef *fdcan,
127 uint32_t *sid_list,
128 uint32_t num_sid,
129 uint32_t *xid_list,
130 uint32_t num_xid) {
131 fdcan->CCCR |= FDCAN_CCCR_INIT;
132 while ((fdcan->CCCR & FDCAN_CCCR_INIT) == 0) {}
133 fdcan->CCCR |= FDCAN_CCCR_CCE;
134
135 fdcan->RXGFC = (FDCAN_REJECT << FDCAN_RXGFC_ANFS_Pos) | (FDCAN_REJECT << FDCAN_RXGFC_ANFE_Pos)
136 | (FDCAN_ACCEPT_IN_RX_FIFO0 << FDCAN_RXGFC_RRFS_Pos)
137 | (FDCAN_ACCEPT_IN_RX_FIFO0 << FDCAN_RXGFC_RRFE_Pos);
138
139 PHAL_FDCAN_setStandardFilter(fdcan, num_sid, sid_list);
140 PHAL_FDCAN_setExtendedFilter(fdcan, num_xid, xid_list);
141
142 fdcan->CCCR &= ~FDCAN_CCCR_INIT;
143 while ((fdcan->CCCR & FDCAN_CCCR_INIT) != 0) {}
144}

146bool PHAL_FDCAN_init(FDCAN_GlobalTypeDef *fdcan, bool test_mode, uint32_t bit_rate) {
147 // Enable clocks
148 RCC->CCIPR &= ~RCC_CCIPR_FDCANSEL_Msk;
149 RCC->CCIPR |= RCC_FDCANCLKSOURCE_PCLK1;
150 RCC->APB1ENR1 |= RCC_APB1ENR1_FDCANEN; // Always use PCLK1
151
152 // Enable port B clock
153 RCC->AHB2ENR |= RCC_AHB2ENR_GPIOBEN;
154
155 // Ensure peripheral is out of sleep
156 fdcan->CCCR &= ~FDCAN_CCCR_CSR;
157 while (fdcan->CCCR & FDCAN_CCCR_CSA) {}
158 // Enter init
159 fdcan->CCCR |= FDCAN_CCCR_INIT;
160 while ((fdcan->CCCR & FDCAN_CCCR_INIT) == 0) {}
161 // Allow configuration
162 fdcan->CCCR |= FDCAN_CCCR_CCE;
163
164 /* Classic CAN (no FD) */
165 fdcan->CCCR &= ~(FDCAN_CCCR_FDOE | FDCAN_CCCR_BRSE | FDCAN_CCCR_TXP);
166 fdcan->CCCR |= (FDCAN_CCCR_DAR /* Auto retransmission: disable */
167 | FDCAN_CCCR_PXHD); /* Protocol exception: disable */
168 fdcan->CCCR &= ~(FDCAN_CCCR_MON | FDCAN_CCCR_ASM | FDCAN_CCCR_TEST);
169
170 fdcan->CCCR &= ~(FDCAN_CCCR_DAR); // Auto retransmission
171 fdcan->CCCR |= FDCAN_CCCR_TXP; // Transmit pause
172
173 uint32_t nbtp;
174 // TODO: Get clock frequency
175 if (PHAL_FDCAN_makeNBTP(/*ker_hz*/ 16000000u, bit_rate, &nbtp) == 0) {
176 fdcan->NBTP = nbtp; // do this while INIT=1 and CCE=1
177 } else {
178 return false;
179 }
180 // No FD/DBTP yet
181
182 // FIFOs
183 fdcan->TXBC &= ~FDCAN_TXBC_TFQM; // Tx FIFO mode
184
185 // Clear interrupt flags
186 fdcan->IR |= 0xFFFFFFFF;
187 // Enable RX FIFO0 interrupts
188 fdcan->IE |= FDCAN_IE_RF0NE // New message
189 | FDCAN_IE_RF0FE // FIFO full (optional)
190 | FDCAN_IE_RF0LE; // Message lost (optional)
191 fdcan->ILS = 0; // Route to interrupt line 0
192 fdcan->ILE = FDCAN_ILE_EINT0; // Enable line 0
193
194 // Mode
195 fdcan->CCCR &= ~(FDCAN_CCCR_MON | FDCAN_CCCR_TEST | FDCAN_CCCR_ASM);
196 fdcan->TEST &= ~FDCAN_TEST_LBCK;
197 if (test_mode) {
198 fdcan->CCCR |= FDCAN_CCCR_TEST | FDCAN_CCCR_MON;
199 fdcan->TEST |= FDCAN_TEST_LBCK;
200 }
201
202 // Filters (accept all by default)
203 fdcan->RXGFC = (FDCAN_ACCEPT_IN_RX_FIFO0 << FDCAN_RXGFC_ANFS_Pos)
204 | (FDCAN_ACCEPT_IN_RX_FIFO0 << FDCAN_RXGFC_ANFE_Pos)
205 | (FDCAN_ACCEPT_IN_RX_FIFO0 << FDCAN_RXGFC_RRFS_Pos)
206 | (FDCAN_ACCEPT_IN_RX_FIFO0 << FDCAN_RXGFC_RRFE_Pos);
207
208 fdcan->CCCR &= ~FDCAN_CCCR_INIT;
209 while (fdcan->CCCR & FDCAN_CCCR_INIT) {}
210
211 return true;
212}

219void PHAL_FDCAN_send(CanMsgTypeDef_t *msg) {
220 FDCAN_GlobalTypeDef *fdcan = msg->Bus;
221 // TX FIFO/Queue full?
222 if (fdcan->TXFQS & FDCAN_TXFQS_TFQF) {
223 // TODO: count drops
224 return;
225 }
226
227 uint32_t ram_base = PHAL_FDCAN_get_ram_base(fdcan);
228 uint32_t put = (fdcan->TXFQS & FDCAN_TXFQS_TFQPI_Msk) >> FDCAN_TXFQS_TFQPI_Pos;
229
230 volatile uint32_t *tx = (uint32_t *)((ram_base + SRAMCAN_TFQSA) + (put * SRAMCAN_TFQ_SIZE));
231 uint32_t w0;
232 if (msg->IDE == 0) {
233 // Standard ID in [28:18], IDE=0
234 w0 = ((uint32_t)(msg->StdId & 0x7FFu) << 18);
235 } else {
236 // Extended ID in [28:0], set IDE/XTD bit (bit 30)
237 w0 = (msg->ExtId & 0x1FFFFFFFu) | (1u << 30);
238 }
239 tx[0] = w0;
240
241 // 4) Header word 1: DLC at [19:16], classic CAN: BRS=0, FDF=0, EFC=0, MM=0, RTR=0
242 uint32_t dlc = (msg->DLC & 0xFu);
243 if (dlc > 8u)
244 dlc = 8u; // assumes TXESC TBDS = 8 bytes
245 tx[1] = (dlc << 16);
246
247 uint32_t d0 = 0;
248 uint32_t d1 = 0;
249 if (dlc > 0)
250 d0 |= (uint32_t)msg->Data[0];
251 if (dlc > 1)
252 d0 |= (uint32_t)msg->Data[1] << 8;
253 if (dlc > 2)
254 d0 |= (uint32_t)msg->Data[2] << 16;
255 if (dlc > 3)
256 d0 |= (uint32_t)msg->Data[3] << 24;
257 if (dlc > 4)
258 d1 |= (uint32_t)msg->Data[4];
259 if (dlc > 5)
260 d1 |= (uint32_t)msg->Data[5] << 8;
261 if (dlc > 6)
262 d1 |= (uint32_t)msg->Data[6] << 16;
263 if (dlc > 7)
264 d1 |= (uint32_t)msg->Data[7] << 24;
265
266 tx[2] = d0;
267 tx[3] = d1;
268 tx[4] = 0; // pad (8-byte elements)
269
270 fdcan->TXBAR = (1u << put);
271}

278// Return 0 on success; -2 if FIFO empty; -1 on bad args
279static int PHAL_FDCAN_getRxMessage(FDCAN_GlobalTypeDef *fdcan, CanMsgTypeDef_t *m) {
280 if (!fdcan || !m)
281 return -1;
282
283 // Empty?
284 uint32_t f0s = fdcan->RXF0S;
285 if ((f0s & FDCAN_RXF0S_F0FL_Msk) == 0u)
286 return -2;
287
288 // If FIFO0 is full AND overwrite mode is enabled, drop the oldest first
289 if ((f0s & FDCAN_RXF0S_F0F_Msk) && (fdcan->RXGFC & FDCAN_RXGFC_F0OM)) {
290 uint32_t gi_drop = (f0s & FDCAN_RXF0S_F0GI_Msk) >> FDCAN_RXF0S_F0GI_Pos;
291 fdcan->RXF0A = gi_drop; // discard oldest
292 // Re-check after dropping
293 f0s = fdcan->RXF0S;
294 if ((f0s & FDCAN_RXF0S_F0FL_Msk) == 0u)
295 return -2;
296 }
297
298 // Element index to read now
299 uint32_t get = (f0s & FDCAN_RXF0S_F0GI_Msk) >> FDCAN_RXF0S_F0GI_Pos;
300
301 // Message RAM address (your fixed layout macros)
302 uintptr_t ram_base = PHAL_FDCAN_get_ram_base(fdcan);
303 volatile uint32_t *rx = (uint32_t *)((ram_base + SRAMCAN_RF0SA) + (get * SRAMCAN_RF0_SIZE));
304
305 uint32_t w0 = rx[0]; // ID + XTD/RTR/ESI
306 uint32_t w1 = rx[1]; // TS + DLC/BRS/FDF/FIDX/ANMF
307
308 // Fill your CanMsgTypeDef_t
309 *m = (CanMsgTypeDef_t) {0};
310 m->Bus = fdcan;
311
312 if (w0 & (1u << 30)) { // XTD (IDE) = 1 → extended
313 m->IDE = 1;
314 m->ExtId = (w0 & 0x1FFFFFFFu);
315 } else { // standard
316 m->IDE = 0;
317 m->StdId = (uint16_t)((w0 >> 18) & 0x7FFu);
318 m->ExtId = m->StdId; // your parser expects ExtId too
319 }
320
321 // DLC code → bytes; cap to our 8-byte buffer
322 uint8_t dlc_code = (uint8_t)((w1 >> 16) & 0xF);
323 uint8_t len_bytes = dlc_to_len(dlc_code);
324 if (len_bytes > sizeof m->Data)
325 len_bytes = sizeof m->Data;
326 m->DLC = len_bytes;
327
328 // Copy payload (rx[2].. as bytes)
329 const uint8_t *p = (const uint8_t *)&rx[2];
330 for (uint8_t i = 0; i < len_bytes; ++i)
331 m->Data[i] = p[i];
332
333 // Pop this FIFO0 element
334 fdcan->RXF0A = get;
335
336 return 0;
337}

339void PHAL_FDCAN_RX_IRQHandler(FDCAN_GlobalTypeDef *fdcan) {
340 uint32_t ir = fdcan->IR;
341 if (ir & FDCAN_IR_RF0L) {
342 fdcan->IR = FDCAN_IR_RF0L;
343 // TODO message lost
344 }
345
346 if (ir & FDCAN_IR_RF0F) {
347 fdcan->IR = FDCAN_IR_RF0F;
348 // TODO FIFO full
349 }
350
351 if (ir & FDCAN_IR_RF0N) {
352 CanMsgTypeDef_t m;
353 while (PHAL_FDCAN_getRxMessage(fdcan, &m) == 0) {
354 PHAL_FDCAN_rxCallback(&m); // or enqueue
355 }
356 fdcan->IR = FDCAN_IR_RF0N; // clear after draining
357 }
358}
