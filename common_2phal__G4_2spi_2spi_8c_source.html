22bool PHAL_SPI_init(SPI_InitConfig_t *cfg) {
23 zero = 0;
24
25 // Enable RCC Clock for selected SPI on G4
26 switch ((uint32_t)cfg->periph) {
27 case SPI1_BASE:
28 RCC->APB2ENR |= RCC_APB2ENR_SPI1EN;
29 break;
30 case SPI2_BASE:
31 RCC->APB1ENR1 |= RCC_APB1ENR1_SPI2EN;
32 break;
33 case SPI3_BASE:
34 RCC->APB1ENR1 |= RCC_APB1ENR1_SPI3EN;
35 break;
36 default:
37 return false;
38 }
39
40 // Mode configuration
41 if (cfg->mode == SPI_MODE_MASTER) {
42 // Master mode, software NSS
43 cfg->periph->CR1 |= SPI_CR1_MSTR | SPI_CR1_SSM | SPI_CR1_SSI;
44 // Baud rate prescaler (BR) in CR1, source depends on bus
45 uint32_t f_div;
46 if ((uint32_t)cfg->periph == SPI1_BASE)
47 f_div = LOG2_DOWN(APB2ClockRateHz / cfg->data_rate) - 1;
48 else
49 f_div = LOG2_DOWN(APB1ClockRateHz / cfg->data_rate) - 1;
50 f_div = CLAMP(f_div, 0, 0b111);
51 cfg->periph->CR1 &= ~SPI_CR1_BR_Msk;
52 cfg->periph->CR1 |= f_div << SPI_CR1_BR_Pos;
53 } else {
54 // Slave mode: clear MSTR.
55 cfg->periph->CR1 &= ~SPI_CR1_MSTR;
56 if (cfg->nss_sw) {
57 // Software NSS: internally select the slave (SSM=1, SSI=0)
58 cfg->periph->CR1 |= SPI_CR1_SSM;
59 cfg->periph->CR1 &= ~SPI_CR1_SSI;
60 } else {
61 // Hardware NSS: NSS managed by external pin (SSM=0). SSI ignored.
62 cfg->periph->CR1 &= ~SPI_CR1_SSM;
63 }
64 // BR ignored in slave
65 }
66 cfg->periph->CR1 &= ~(SPI_CR1_CPOL | SPI_CR1_CPHA);
67 if (cfg->cpol)
68 cfg->periph->CR1 |= SPI_CR1_CPOL;
69 if (cfg->cpha)
70 cfg->periph->CR1 |= SPI_CR1_CPHA;
71
72 // Frame size via CR2 DS[3:0] on G4
73 cfg->periph->CR2 &= ~(SPI_CR2_DS_Msk);
74 uint8_t ds = (CLAMP(cfg->data_len, 4, 16) - 1) & 0xF; // DS = bits-1
75 cfg->periph->CR2 |= (ds << SPI_CR2_DS_Pos);
76 // RX FIFO threshold: set FRXTH for 8-bit threshold
77 cfg->periph->CR2 |= SPI_CR2_FRXTH;
78
79 // DMA setup if provided
80 if (cfg->rx_dma_cfg && !PHAL_initDMA(cfg->rx_dma_cfg))
81 return false;
82 if (cfg->tx_dma_cfg && !PHAL_initDMA(cfg->tx_dma_cfg))
83 return false;
84
85 // Deassert CS in master when using software NSS
86 if (cfg->mode == SPI_MODE_MASTER && cfg->nss_sw)
87 PHAL_writeGPIO(cfg->nss_gpio_port, cfg->nss_gpio_pin, 1);
88
89 cfg->_busy = false;
90 cfg->_error = false;
91 cfg->_direct_mode_error = false;
92 cfg->_fifo_overrun = false;
93
94 return true;
95}

97bool PHAL_SPI_transfer_noDMA(SPI_InitConfig_t *spi,
98 const uint8_t *out_data,
99 uint32_t txlen,
100 uint32_t rxlen,
101 uint8_t *in_data) {
102 // Prepare RX pointer to skip echoed TX bytes
103 uint8_t *rx_ptr = in_data ? (in_data + txlen) : NULL;
104
105 if (PHAL_SPI_busy(spi))
106 return false;
107
108 spi->_busy = true;
109
110 // Assert CS for master only
111 if (spi->mode == SPI_MODE_MASTER && spi->nss_sw)
112 PHAL_writeGPIO(spi->nss_gpio_port, spi->nss_gpio_pin, 0);
113
114 // Enable SPI
115 spi->periph->CR1 |= SPI_CR1_SPE;
116
117 // Transmit txlen bytes and capture echoed RX into in_data if provided
118 for (uint32_t i = 0; i < txlen; i++) {
119 // Wait for TXE
120 while (!(spi->periph->SR & SPI_SR_TXE))
121 ;
122 // Write byte
123 uint8_t b = out_data ? out_data[i] : 0;
124 *(volatile uint8_t *)&spi->periph->DR = b;
125 // Wait for RXNE and read echo
126 while (!(spi->periph->SR & SPI_SR_RXNE))
127 ;
128 uint8_t rxb = *(volatile uint8_t *)&spi->periph->DR;
129 if (in_data)
130 in_data[i] = rxb;
131 }
132
133 // If additional rxlen bytes requested beyond txlen, clock out dummy
134 for (uint32_t i = 0; i < rxlen; i++) {
135 while (!(spi->periph->SR & SPI_SR_TXE))
136 ;
137 *(volatile uint8_t *)&spi->periph->DR = 0;
138 while (!(spi->periph->SR & SPI_SR_RXNE))
139 ;
140 uint8_t rb = *(volatile uint8_t *)&spi->periph->DR;
141 if (rx_ptr)
142 rx_ptr[i] = rb;
143 }
144
145 // Wait until not busy and TXE set
146 while ((spi->periph->SR & SPI_SR_BSY))
147 ;
148 while (!(spi->periph->SR & SPI_SR_TXE))
149 ;
150
151 // Deassert CS for master only
152 if (spi->mode == SPI_MODE_MASTER && spi->nss_sw)
153 PHAL_writeGPIO(spi->nss_gpio_port, spi->nss_gpio_pin, 1);
154
155 // Disable SPI
156 spi->periph->CR1 &= ~SPI_CR1_SPE;
157
158 spi->_busy = false;
159 return true;
160}

162bool PHAL_SPI_transfer(SPI_InitConfig_t *spi,
163 const uint8_t *out_data,
164 const uint32_t data_len,
165 uint8_t *in_data) {
166 if (spi->tx_dma_cfg == 0)
167 return false;
168 if (PHAL_SPI_busy(spi))
169 return false;
170
171 // Assert CS for master only
172 if (spi->mode == SPI_MODE_MASTER && spi->nss_sw)
173 PHAL_writeGPIO(spi->nss_gpio_port, spi->nss_gpio_pin, 0);
174
175 spi->_busy = true;
176
177 // TX DMA enable
178 spi->periph->CR2 |= SPI_CR2_TXDMAEN;
179 if (!out_data) {
180 spi->tx_dma_cfg->mem_inc = false;
181 PHAL_DMA_setMemAddress(spi->tx_dma_cfg, (uint32_t)&zero);
182 } else {
183 spi->tx_dma_cfg->mem_inc = true;
184 PHAL_DMA_setMemAddress(spi->tx_dma_cfg, (uint32_t)out_data);
185 }
186 PHAL_DMA_setTxferLength(spi->tx_dma_cfg, data_len);
187
188 // RX DMA optional
189 if (spi->rx_dma_cfg) {
190 spi->periph->CR2 |= SPI_CR2_RXDMAEN;
191 if (!in_data) {
192 spi->rx_dma_cfg->mem_inc = false;
193 PHAL_DMA_setMemAddress(spi->rx_dma_cfg, (uint32_t)&trash_can);
194 } else {
195 spi->rx_dma_cfg->mem_inc = true;
196 PHAL_DMA_setMemAddress(spi->rx_dma_cfg, (uint32_t)in_data);
197 }
198 PHAL_DMA_setTxferLength(spi->rx_dma_cfg, data_len);
199 PHAL_reEnable(spi->rx_dma_cfg);
200 }
201
202 // Enable DMA IRQ for selected channel and track active transfer per-channel
203 volatile SPI_InitConfig_t **active_table;
204 if (spi->tx_dma_cfg->periph == DMA1) {
205 NVIC_EnableIRQ(DMA1_Channel1_IRQn + (spi->tx_dma_cfg->channel_idx - 1));
206 active_table = dma1_active_tx;
207 } else if (spi->tx_dma_cfg->periph == DMA2) {
208 NVIC_EnableIRQ(DMA2_Channel1_IRQn + (spi->tx_dma_cfg->channel_idx - 1));
209 active_table = dma2_active_tx;
210 } else {
211 return false;
212 }
213 active_table[spi->tx_dma_cfg->channel_idx] = spi;
214
215 // Start SPI and kick TX DMA
216 spi->periph->CR1 |= SPI_CR1_SPE;
217 PHAL_reEnable(spi->tx_dma_cfg);
218
219 return true;
220}

245static void handleTxComplete(DMA_TypeDef *dma_periph, uint8_t channel) {
246 volatile SPI_InitConfig_t **active_table =
247 (dma_periph == DMA1) ? dma1_active_tx : dma2_active_tx;
248 SPI_InitConfig_t *transfer = (channel < 8) ? (SPI_InitConfig_t *)active_table[channel] : NULL;
249 if (transfer == NULL)
250 return;
251
252 uint32_t tcif_mask = DMA_ISR_TCIF1 << (4 * (channel - 1));
253 uint32_t teif_mask = DMA_ISR_TEIF1 << (4 * (channel - 1));
254 uint32_t htif_mask = DMA_ISR_HTIF1 << (4 * (channel - 1));
255 uint32_t gif_mask = DMA_ISR_GIF1 << (4 * (channel - 1));
256
257 if (dma_periph->ISR & teif_mask) {
258 dma_periph->IFCR |= teif_mask;
259 transfer->_error = true;
260 }
261 if (dma_periph->ISR & tcif_mask) {
262 // Wait for TXE and not busy
263 while (!(transfer->periph->SR & SPI_SR_TXE) || (transfer->periph->SR & SPI_SR_BSY))
264 ;
265
266 // If RX DMA is used, wait until its TC flag also asserts before teardown
267 if (transfer->rx_dma_cfg) {
268 DMA_TypeDef *rx_dma = transfer->rx_dma_cfg->periph;
269 uint8_t rx_ch = transfer->rx_dma_cfg->channel_idx;
270 uint32_t rx_tc_mask = DMA_ISR_TCIF1 << (4 * (rx_ch - 1));
271 // Busy-wait for RX complete
272 while (!(rx_dma->ISR & rx_tc_mask))
273 ;
274 // Clear RX flags and stop RX
275 rx_dma->IFCR |= rx_tc_mask;
276 PHAL_stopTxfer(transfer->rx_dma_cfg);
277 }
278
279 // Deassert CS after both TX and RX complete
280 if (transfer->nss_sw)
281 PHAL_writeGPIO(transfer->nss_gpio_port, transfer->nss_gpio_pin, 1);
282
283 if (transfer->tx_dma_cfg)
284 PHAL_stopTxfer(transfer->tx_dma_cfg);
285
286 transfer->periph->CR1 &= ~SPI_CR1_SPE;
287 transfer->periph->CR2 &= ~(SPI_CR2_TXDMAEN | SPI_CR2_RXDMAEN);
288
289 transfer->_busy = false;
290 transfer->_error = false;
291 transfer->_direct_mode_error = false;
292 transfer->_fifo_overrun = false;
293
294 dma_periph->IFCR |= tcif_mask;
295 dma_periph->IFCR |= gif_mask;
296 active_table[channel] = NULL;
297 }
298
299 if (dma_periph->ISR & htif_mask) {
300 dma_periph->IFCR |= htif_mask;
301 }
302}

317uint8_t PHAL_SPI_readByte(SPI_InitConfig_t *spi, uint8_t address, bool skipDummy) {
318 static uint8_t tx_cmd[4] = {(1 << 7), 0, 0};
319 static uint8_t rx_dat[4] = {1, 1, 1, 1};
320 tx_cmd[0] |= (address & 0x7F);
321
322 while (PHAL_SPI_busy(spi))
323 ;
324 if (spi->rx_dma_cfg != NULL)
325 PHAL_SPI_transfer(spi, tx_cmd, skipDummy ? 2 : 3, rx_dat);
326 else
327 PHAL_SPI_transfer_noDMA(spi, tx_cmd, 1, skipDummy ? 1 : 2, rx_dat);
328 while (PHAL_SPI_busy(spi))
329 ;
330
331 return skipDummy ? rx_dat[1] : rx_dat[2];
332}

334uint8_t PHAL_SPI_writeByte(SPI_InitConfig_t *spi, uint8_t address, uint8_t writeDat) {
335 uint8_t tx_cmd[3] = {0};
336 uint8_t rx_dat[3] = {0};
337 tx_cmd[0] |= (address & 0x7F);
338 tx_cmd[1] |= (writeDat);
339
340 while (PHAL_SPI_busy(spi))
341 ;

343 PHAL_SPI_transfer(spi, tx_cmd, 2, rx_dat);
344 else
345 PHAL_SPI_transfer_noDMA(spi, tx_cmd, 2, 0, rx_dat);
346 while (PHAL_SPI_busy(spi))
347 ;
348
