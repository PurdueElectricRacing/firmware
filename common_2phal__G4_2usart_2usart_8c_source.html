33bool PHAL_initUSART(usart_init_t* handle, const uint32_t fck) {
34 uint32_t div;
35
36 // Add Handle to active peripheral set for keeping track of activity.
37 active_uarts[handle->usart_active_num].active_handle = handle;
38
39 // Disable peripheral until properly configured
40 handle->periph->CR1 &= ~USART_CR1_UE;
41
42 // Enable peripheral clock in RCC for G4.
43 switch ((ptr_int)handle->periph) {
44 case USART1_BASE:
45 RCC->APB2ENR |= RCC_APB2ENR_USART1EN;
46 break;
47 case USART2_BASE:
48 RCC->APB1ENR1 |= RCC_APB1ENR1_USART2EN;
49 break;
50 case USART3_BASE:
51 RCC->APB1ENR1 |= RCC_APB1ENR1_USART3EN;
52 break;
53 case UART4_BASE:
54 RCC->APB1ENR1 |= RCC_APB1ENR1_UART4EN;
55 break;
56 case LPUART1_BASE:
57 RCC->APB1ENR2 |= RCC_APB1ENR2_LPUART1EN;
58 break;
59 default:
60 return false;
61 }
62
63 if (handle->ovsample == OV_16) {
64 div = (fck + (handle->baud_rate / 2U)) / handle->baud_rate;
65 handle->periph->BRR = div;
66 } else {
67 div = (2U * fck + (handle->baud_rate / 2U)) / handle->baud_rate;
68 uint32_t mantissa = div & 0xFFF0;
69 uint32_t fraction = (div & 0x000F) >> 1;
70 handle->periph->BRR = mantissa | fraction;
71 }
72
73 // Set CR1 parameters
74 handle->periph->CR1 = 0U;
75 handle->periph->CR1 |= (handle->parity != PT_NONE) << USART_CR1_PCE_Pos;
76
77 switch (handle->word_length) {
78 case WORD_7:
79 handle->periph->CR1 |= USART_CR1_M1; // bit 28 = 1, bit 12 = 0
80 break;
81
82 case WORD_9:
83 handle->periph->CR1 |= USART_CR1_M0; // bit 12 = 1, bit 28 = 0
84 break;
85
86 default: // WL_8B
87 // both bits 0
88 break;
89 }
90
91 handle->periph->CR1 |= (handle->parity >> 2) << USART_CR1_PS_Pos;
92 handle->periph->CR1 |= handle->ovsample << USART_CR1_OVER8_Pos;
93 handle->periph->CR1 |= USART_CR1_RXNEIE | USART_CR1_IDLEIE;
94
95 // Set CR2 parameters
96 handle->periph->CR2 = 0U;
97 handle->periph->CR2 |= (handle->address & 0xF) << USART_CR2_ADD_Pos;
98 handle->periph->CR2 |= handle->stop_bits << USART_CR2_STOP_Pos;
99
100 // Set CR3 parameters
101 handle->periph->CR3 |= handle->obsample << USART_CR3_ONEBIT_Pos;
102
103 // Enable peripheral for use
104 handle->periph->CR1 |= USART_CR1_UE;
105
106 // Blocking is currently not supported without DMA configuration
107 if (!handle->rx_dma_cfg || !handle->tx_dma_cfg)
108 return false;
109
110 // Configure DMA
111 if (!PHAL_initDMA(handle->tx_dma_cfg) || !PHAL_initDMA(handle->rx_dma_cfg)) {
112 return false;
113 }
114
115 return true;
116}

160bool PHAL_usartTxDma(usart_init_t* handle, uint8_t* data, uint32_t len) {
161 if (active_uarts[handle->usart_active_num].active_handle != handle)
162 return false;
163
164 // Ensure any RX data is not overwritten before continuing with transfer
165 while (
166 (active_uarts[handle->usart_active_num].active_handle->periph->ISR & USART_ISR_RXNE_RXFNE))
167 ;
168
169 // Enable the correct DMA interrupt for the G4
170 if (handle->tx_dma_cfg->periph == DMA1) {
171 NVIC_EnableIRQ(DMA1_Channel1_IRQn + (handle->tx_dma_cfg->channel_idx - 1));
172 } else if (handle->tx_dma_cfg->periph == DMA2) {
173 NVIC_EnableIRQ(DMA2_Channel1_IRQn + (handle->tx_dma_cfg->channel_idx - 1));
174 } else {
175 return false;
176 }
177
178 PHAL_stopTxfer(handle->tx_dma_cfg);
179
180 PHAL_DMA_setTxferLength(handle->tx_dma_cfg, len);
181 PHAL_DMA_setMemAddress(handle->tx_dma_cfg, (uint32_t)data);
182
183 PHAL_reEnable(handle->tx_dma_cfg);
184
185 active_uarts[handle->usart_active_num]._tx_busy = 1;
186
187 handle->periph->CR3 |= USART_CR3_DMAT;
188 handle->periph->CR1 |= USART_CR1_TE;
189
190 PHAL_startTxfer(handle->tx_dma_cfg);
191 return true;

326static void handleUsartIRQ(USART_TypeDef* periph, uint8_t idx) {
327 uint32_t isr = periph->ISR;
328
329 // USART RX Not Empty interrupt flag
330 if (isr & USART_ISR_RXNE_RXFNE) {
331 active_uarts[idx]._rx_busy = 1;
332 PHAL_DMA_setTxferLength(active_uarts[idx].active_handle->rx_dma_cfg,
333 active_uarts[idx].rxfer_size);
334 PHAL_reEnable(active_uarts[idx].active_handle->rx_dma_cfg);
335 // Read RDR to clear RXNE flag if set
336 (void)active_uarts[idx].active_handle->periph->RDR;
337 active_uarts[idx].active_handle->periph->RQR = USART_RQR_RXFRQ;
338 active_uarts[idx].active_handle->periph->CR1 &= ~USART_CR1_RXNEIE;
339 // Clear any errors that may have been set in the previous Rx
340 active_uarts[idx].active_handle->rx_errors.framing_error = 0;
341 active_uarts[idx].active_handle->rx_errors.noise_detected = 0;
342 active_uarts[idx].active_handle->rx_errors.overrun = 0;
343 active_uarts[idx].active_handle->rx_errors.parity_error = 0;
344 }
345
346 // Overrun Error Flag
347 if (isr & USART_ISR_ORE) {
348 active_uarts[idx].active_handle->rx_errors.overrun = 1;
349 periph->ICR |= USART_ICR_ORECF;
350 }
351 // Noise Error Flag
352 if (isr & USART_ISR_NE) {
353 active_uarts[idx].active_handle->rx_errors.noise_detected = 1;
354 periph->ICR |= USART_ICR_NECF;
355 }
356 // Framing Error Flag
357 if (isr & USART_ISR_FE) {
358 active_uarts[idx].active_handle->rx_errors.framing_error = 1;
359 periph->ICR |= USART_ICR_FECF;
360 }
361 // Parity Error Flag
362 if (isr & USART_ISR_PE) {
363 active_uarts[idx].active_handle->rx_errors.parity_error = 1;
364 periph->ICR |= USART_ICR_PECF;
365 }
366
367 // Idle line detected
368 if (isr & USART_ISR_IDLE) {
369 PHAL_stopTxfer(active_uarts[idx].active_handle->rx_dma_cfg);
370 if (active_uarts[idx].cont_rx) {
371 // Read RDR to clear RXNE before re-enabling RXNEIE
372 if (periph->ISR & USART_ISR_RXNE_RXFNE) {
373 (void)periph->RDR;
374 }
375 active_uarts[idx].active_handle->periph->CR1 |= USART_CR1_RXNEIE;
376 } else {
377 active_uarts[idx].active_handle->periph->CR1 &= ~USART_CR1_RE;
378 }
379 active_uarts[idx]._rx_busy = 0;
380 periph->ICR |= USART_ICR_IDLECF;
381 usart_receive_complete_callback(active_uarts[idx].active_handle);
382 }
383}

396static void
397handleDMAxComplete(DMA_TypeDef* dma_periph, uint8_t channel, uint8_t dma_type, uint8_t idx) {
398 // The bit masks for each channel's flags
399 uint32_t tcif_mask = DMA_ISR_TCIF1 << (4 * (channel - 1));
400 uint32_t teif_mask = DMA_ISR_TEIF1 << (4 * (channel - 1));
401 uint32_t htif_mask = DMA_ISR_HTIF1 << (4 * (channel - 1));
402 uint32_t gif_mask = DMA_ISR_GIF1 << (4 * (channel - 1));
403
404 // Check for a Transfer Complete interrupt
405 if (dma_periph->ISR & tcif_mask) {
406 // Clear the transfer complete flag
407 dma_periph->IFCR |= tcif_mask;
408
409 if (dma_type == USART_DMA_TX) {
410 PHAL_stopTxfer(active_uarts[idx].active_handle->tx_dma_cfg);
411 active_uarts[idx]._tx_busy = 0;
412 }
413 }
414
415 // Check for a Transfer Error interrupt
416 if (dma_periph->ISR & teif_mask) {
417 // Clear the transfer error flag
418 dma_periph->IFCR |= teif_mask;
419
420 if (dma_type == USART_DMA_TX) {
421 active_uarts[idx].active_handle->tx_errors.dma_transfer_error = 1;
422 } else {
423 active_uarts[idx].active_handle->rx_errors.dma_transfer_error = 1;
424 }
425 }
426
427 // Check for Half Transfer Complete interrupt (if enabled)
428 if (dma_periph->ISR & htif_mask) {
429 dma_periph->IFCR |= htif_mask;
430 }
431
432 // Clear any other global flags for this channel
433 dma_periph->IFCR |= gif_mask;
434}
