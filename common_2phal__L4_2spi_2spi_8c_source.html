20bool PHAL_SPI_init(SPI_InitConfig_t* cfg) {
21 zero = 0;
22 if (cfg->periph == SPI1) {
23 // Enable RCC Clock
24 RCC->APB2ENR |= RCC_APB2ENR_SPI1EN;
25 }
26#ifdef SPI2
27 else if (cfg->periph == SPI2) {
28 RCC->APB1ENR1 |= RCC_APB1ENR1_SPI2EN;
29 }
30#endif
31 else if (cfg->periph == SPI3) {
32 RCC->APB1ENR1 |= RCC_APB1ENR1_SPI3EN;
33 } else {
34 return false;
35 }
36
37 // Setup for Master, positive polarity
38 cfg->periph->CR1 |= SPI_CR1_MSTR | SPI_CR1_SPE | SPI_CR1_SSM | SPI_CR1_SSI;
39 cfg->periph->CR1 &= ~(SPI_CR1_CPOL);
40 cfg->periph->CR2 &= ~(SPI_CR2_NSSP | SPI_CR2_SSOE);
41
42 // Data Size
43 cfg->periph->CR2 &= ~(SPI_CR2_DS_Msk);
44 cfg->periph->CR2 |= (CLAMP(cfg->data_len, 4, 16) - 1) << SPI_CR2_DS_Pos;
45
46 // RX Fifo full on 8 bits
47 cfg->periph->CR2 |= SPI_CR2_FRXTH;
48 // SPI1->CR1 |= SPI_CR1_LSBFIRST;
49
50 // Data Rate
51 // Divisor is a power of 2, find the closest power of 2 limited to log2(256)
52 uint32_t f_div = 0;
53 if (cfg->periph == SPI1)
54 f_div = LOG2_DOWN(APB2ClockRateHz / cfg->data_rate) - 1;
55 // Both SPI1 and SPI2 are on APB1
56 else
57 f_div = LOG2_DOWN(APB1ClockRateHz / cfg->data_rate) - 1;
58 f_div = CLAMP(f_div, 0, 0b111);
59 cfg->periph->CR1 &= ~SPI_CR1_BR_Msk;
60 cfg->periph->CR1 |= f_div << SPI_CR1_BR_Pos;
61
62 // Setup DMA channels
63 if (cfg->rx_dma_cfg && !PHAL_initDMA(cfg->rx_dma_cfg))
64 return false;
65
66 if (cfg->tx_dma_cfg && !PHAL_initDMA(cfg->tx_dma_cfg))
67 return false;
68
69 PHAL_writeGPIO(cfg->nss_gpio_port, cfg->nss_gpio_pin, 1);
70
71 cfg->_busy = false;
72 cfg->_error = false;
73}

75bool PHAL_SPI_transfer_noDMA(SPI_InitConfig_t* spi, const uint8_t* out_data, uint32_t txlen, uint32_t rxlen, uint8_t* in_data) {
76 // SPI1->CR1 &= ~(1 << 11);
77 in_data += txlen;
78 if (PHAL_SPI_busy(spi))
79 return false;
80 active_transfer = spi;
81 spi->_busy = true;
82 // Enable SPI
83 spi->periph->CR1 |= SPI_CR1_SPE;
84 // Select peripheral
85 if (spi->nss_sw)
86 PHAL_writeGPIO(spi->nss_gpio_port, spi->nss_gpio_pin, 0);
87 // Add messages to TX FIFO
88 for (uint8_t i = 0; i < txlen; i++) {
89 while (!(spi->periph->SR & SPI_SR_TXE))
90 ;
91 if (i + 1 < txlen) {
92 uint16_t data = out_data[i + 1] << 8 | (uint16_t)out_data[i];
93 spi->periph->DR = data;
94 i++;
95 } else {
96 spi->periph->DR = out_data[i];
97 }
98 }
99 // Wait till TX FIFO is empty and spi is not active
100 while (!(spi->periph->SR & SPI_SR_TXE) || (spi->periph->SR & SPI_SR_BSY))
101 ;
102 // Clear overrun
103 uint8_t trash = spi->periph->DR;
104 trash = spi->periph->SR;
105
106 // RX
107 for (uint8_t i = 0; i < rxlen; i++) {
108 // Wait till SPI is not in active transaction, send dummy
109 while ((spi->periph->SR & SPI_SR_BSY))
110 ;
111 spi->periph->DR = 0;
112 // With for RX FIFO to recieve a message, read it in
113 while (!(spi->periph->SR & SPI_SR_RXNE))
114 ;
115 in_data[i] = (uint8_t)(spi->periph->DR);
116 }
117 // Stop message
118 if (spi->nss_sw)
119 PHAL_writeGPIO(spi->nss_gpio_port, spi->nss_gpio_pin, 1);
120 // Wait till transaction is finished, disable spi, and clear the queue
121 while ((spi->periph->SR & SPI_SR_BSY))
122 ;
123 spi->periph->CR1 &= ~SPI_CR1_SPE;
124 while ((spi->periph->SR & SPI_SR_FRLVL)) {
125 trash = spi->periph->DR;
126 }
127 spi->_busy = false;
128 return true;
129}

131bool PHAL_SPI_transfer(SPI_InitConfig_t* spi, const uint8_t* out_data, const uint32_t data_len, const uint8_t* in_data) {
132 /*
133 Each DMA channel is enabled if a data buffer is provided.
134 Only the Tx DMA channel has transfer complete interrupts enabled, as the same data length is used for both channels
135 */
136
137 if (PHAL_SPI_busy(spi))
138 return false;
139
140 active_transfer = spi;
141
142 if (spi->nss_sw)
143 PHAL_writeGPIO(spi->nss_gpio_port, spi->nss_gpio_pin, 0);
144
145 spi->_busy = true;
146
147 spi->periph->CR2 |= SPI_CR2_TXDMAEN;
148 if (!out_data) {
149 spi->tx_dma_cfg->channel->CCR &= ~DMA_CCR_MINC;
150 PHAL_DMA_setMemAddress(spi->tx_dma_cfg, (uint32_t)&zero);
151 } else {
152 PHAL_DMA_setMemAddress(spi->tx_dma_cfg, (uint32_t)out_data);
153 }
154 PHAL_DMA_setTxferLength(spi->tx_dma_cfg, data_len);
155
156 spi->periph->CR2 |= SPI_CR2_RXDMAEN;
157 if (!in_data) {
158 spi->rx_dma_cfg->channel->CCR &= ~DMA_CCR_MINC;
159 PHAL_DMA_setMemAddress(spi->rx_dma_cfg, (uint32_t)&trash_can);
160 } else {
161 PHAL_DMA_setMemAddress(spi->rx_dma_cfg, (uint32_t)in_data);
162 }
163 PHAL_DMA_setTxferLength(spi->rx_dma_cfg, data_len);
164
165 PHAL_startTxfer(spi->rx_dma_cfg);
166
167 // Enable the DMA IRQ
168 if (spi->periph == SPI1) {
169 NVIC_EnableIRQ(DMA1_Channel2_IRQn);
170 NVIC_EnableIRQ(DMA1_Channel3_IRQn);
171 }
172#ifdef SPI2
173 else if (spi->periph == SPI2) {
174 NVIC_EnableIRQ(DMA1_Channel4_IRQn);
175 NVIC_EnableIRQ(DMA1_Channel5_IRQn);
176 }
177#endif
178 else if (spi->periph == SPI3) {
179 NVIC_EnableIRQ(DMA2_Channel1_IRQn);
180 NVIC_EnableIRQ(DMA2_Channel2_IRQn);
181 }
182
183 // Start transaction
184 spi->periph->CR1 |= SPI_CR1_SPE;
185
186 // STM32 HAL Libraries start TX Dma transaction last
187 PHAL_startTxfer(spi->tx_dma_cfg);
188
189 return true;
190}

192bool PHAL_SPI_busy(SPI_InitConfig_t* cfg) {
193 // Latch in case active_transfer cleared during interrupt
194 volatile SPI_InitConfig_t* act = active_transfer;
195 if (act && cfg->periph == act->periph)
196 return act->_busy;
197
198 return false;

273 active_transfer->_busy = false;

354uint8_t PHAL_SPI_readByte(SPI_InitConfig_t* spi, uint8_t address, bool skipDummy) {
355 static uint8_t tx_cmd[4] = {(1 << 7), 0, 0};
356 static uint8_t rx_dat[4] = {1, 1, 1, 1};
357 tx_cmd[0] |= (address & 0x7F);
358
359 while (PHAL_SPI_busy(spi))
360 ;
361 if (spi->rx_dma_cfg != NULL)
362 PHAL_SPI_transfer(spi, tx_cmd, skipDummy ? 2 : 3, rx_dat);
363 else
364 PHAL_SPI_transfer_noDMA(spi, tx_cmd, 1, skipDummy ? 1 : 2, rx_dat);
365 while (PHAL_SPI_busy(spi))
366 ;
367
368 return skipDummy ? rx_dat[1] : rx_dat[2];
369}

371uint8_t PHAL_SPI_writeByte(SPI_InitConfig_t* spi, uint8_t address, uint8_t writeDat) {
372 uint8_t tx_cmd[3] = {0};
373 uint8_t rx_dat[3] = {0};
374 tx_cmd[0] |= (address & 0x7F);
375 tx_cmd[1] |= (writeDat);
376
377 while (PHAL_SPI_busy(spi))
378 ;
379 if (spi->tx_dma_cfg != NULL)
380 PHAL_SPI_transfer(spi, tx_cmd, 2, rx_dat);
381 else
382 PHAL_SPI_transfer_noDMA(spi, tx_cmd, 2, 0, rx_dat);
383 while (PHAL_SPI_busy(spi))
384 ;
385
386 return rx_dat[1];
387}
