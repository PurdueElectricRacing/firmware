18bool PHAL_initUSART(USART_TypeDef* instance, usart_init_t* handle, const uint32_t fck) {
19 // Locals
20 uint16_t div;
21
22 // Disable peripheral until properly configured
23 instance->CR1 &= ~USART_CR1_UE;
24
25 // Enable peripheral clock in RCC
26 switch ((ptr_int)instance) {
27 case USART1_BASE:
28 RCC->APB2ENR |= RCC_APB2ENR_USART1EN;
29 break;
30
31 case USART2_BASE:
32 RCC->APB1ENR1 |= RCC_APB1ENR1_USART2EN;
33
34 case LPUART1_BASE:
35 RCC->APB1ENR2 |= RCC_APB1ENR2_LPUART1EN;
36 break;
37
38 default:
39 return false;
40 }
41
42 // Calculate prescalar for given baud rate
43 div = (fck / handle->baud_rate) * ((handle->ovsample == OV_16) ? 1 : 2);
44
45 // Apply prescalar
46 switch (handle->ovsample) {
47 case OV_16:
48 instance->BRR = div;
49 break;
50
51 case OV_8:
52 instance->BRR = ((div & 0b111) >> 1) | (div & 0xfff0);
53 break;
54 }
55
56 // Set CR1 parameters
57 instance->CR1 = 0U;
58 instance->CR1 |= (handle->word_length & 2) << USART_CR1_M1_Pos;
59 instance->CR1 |= (handle->word_length & 1) << USART_CR1_M0_Pos;
60 instance->CR1 |= handle->ovsample << USART_CR1_OVER8_Pos;
61 instance->CR1 |= handle->parity << USART_CR1_PS_Pos;
62 instance->CR1 |= handle->mode << USART_CR1_RE_Pos;
63 instance->CR1 |= USART_CR1_RXNEIE | USART_CR1_TCIE | USART_CR1_IDLEIE;
64
65 // Set CR2 parameters
66 instance->CR2 = 0U;
67 instance->CR2 |= handle->adv_feature.ab_mode << USART_CR2_ABRMODE_Pos;
68 instance->CR2 |= handle->adv_feature.auto_baud << USART_CR2_ABREN_Pos;
69 instance->CR2 |= handle->adv_feature.msb_first << USART_CR2_MSBFIRST_Pos;
70 instance->CR2 |= handle->adv_feature.data_inv << USART_CR2_DATAINV_Pos;
71 instance->CR2 |= handle->adv_feature.tx_inv << USART_CR2_TXINV_Pos;
72 instance->CR2 |= handle->adv_feature.rx_inv << USART_CR2_RXINV_Pos;
73 instance->CR2 |= handle->adv_feature.tx_rx_swp << USART_CR2_SWAP_Pos;
74 instance->CR2 |= handle->stop_bits << USART_CR2_STOP_Pos;
75
76 // Set CR3 parameters
77 instance->CR3 = 0U;
78 instance->CR3 |= handle->adv_feature.dma_on_rx_err << USART_CR3_DDRE_Pos;
79 instance->CR3 |= handle->adv_feature.overrun << USART_CR3_OVRDIS_Pos;
80 instance->CR3 |= handle->obsample << USART_CR3_ONEBIT_Pos;
81 instance->CR2 |= USART_CR3_EIE;
82
83 // Enable peripheral for use
84 instance->CR1 |= USART_CR1_UE;
85
86 // Configure DMA
87 if (handle->tx_dma_cfg && !PHAL_initDMA(handle->tx_dma_cfg)) {
88 return false;
89 }
90
91 if (handle->rx_dma_cfg && !PHAL_initDMA(handle->rx_dma_cfg)) {
92 return false;
93 }
94
95 // Ensure interrupts are masked
96 usart_mask_irq(instance);
97
98 return true;
99}

101void PHAL_usartTxBl(USART_TypeDef* instance, const uint16_t* data, uint32_t len) {
102 uint8_t i;
103
104 instance->CR3 &= ~USART_CR3_DMAT;
105
106 for (i = 0; i < len; i++) {
107 while (!(instance->ISR & USART_ISR_TXE))
108 ;
109
110 instance->TDR = data[i] & 0x01ff;
111 }
112
113 while (!(instance->ISR & USART_ISR_TC))
114 ;
115}
