49static void calculate_cooling_periodic() {
50 // GPS not stale implies GPS fix
51 bool not_moving = !can_data.gps_speed.stale && can_data.gps_speed.gps_speed <= GPS_SPEED_MOVING;
52
53 // PUMP1: Battery pumps
54 // Enable if above 30C
55 // TODO handle stale
56 if (!can_data.max_cell_temp.stale && (can_data.max_cell_temp.max_temp >= BATT_COOLING_ENABLE_TEMP * 10)) {
57 cr.pump1_status = true;
58 } else {
59 cr.pump1_status = false;
60 }
61
62 // FAN1: Battery fans
63 // Enable battery fans only when car stopped since there is little airflow
64 if (not_moving) {
65 cr.fan1_status = true;
66 if (!cr.fan1_speed) // Don't override dash request values
67 cr.fan1_speed = 100; // Default to 100% duty
68 } else {
69 cr.fan1_status = false;
70 cr.fan1_speed = 0;
71 }
72
73 // PUMP2: Motor pumps
74 // Enable motor pumps if above 60C
75 if (!can_data.rear_motor_temps.stale && ((can_data.rear_motor_temps.left_mot_temp >= MOTOR_COOLING_ENABLE_TEMP || can_data.rear_motor_temps.right_mot_temp >= MOTOR_COOLING_ENABLE_TEMP))) {
76 cr.pump2_status = true;
77 } else {
78 cr.pump2_status = false;
79 }
80
81 // FAN2: Motor fans
82 // Enable motor fans if (above 100C) || (car stopped && temp > 60)
83 if (!can_data.rear_motor_temps.stale && ((can_data.rear_motor_temps.left_mot_temp >= MOTOR_COOLING_MAX_TEMP || can_data.rear_motor_temps.right_mot_temp >= MOTOR_COOLING_MAX_TEMP) || ((can_data.rear_motor_temps.left_mot_temp >= MOTOR_COOLING_ENABLE_TEMP || can_data.rear_motor_temps.right_mot_temp >= MOTOR_COOLING_ENABLE_TEMP) && not_moving))) {
84 cr.fan2_status = true;
85 if (!cr.fan2_speed) // Don't override dash request values
86 cr.fan2_speed = 100;
87 } else {
88 cr.fan2_status = false;
89 cr.fan2_speed = 0;
90 }
91}
