14void PHAL_CRC32_Reset(void) {
15 // CRC initializaion
16#if defined(STM32F732xx)
17 RCC->AHB1ENR |= RCC_AHB1ENR_CRCEN; // Clock the CRC peripheral
18 CRC->INIT = 0xFFFFFFFF; // Reset initial value
19 CRC->CR &= ~CRC_CR_POLYSIZE_Msk; // Set 32 bit (00)
20 CRC->POL = 0x04C11DB7; // CRC-32b (Ethernet Polynomial)
21 CRC->CR |= CRC_CR_RESET; // Reset CRC
22#else
23 RCC->AHB1ENR |= RCC_AHB1ENR_CRCEN; // F4 only supports CRC-32b
24 CRC->CR = CRC_CR_RESET;
25#endif
26}
