184static void configure_interrupts(void) {
185 // Configure exti interupt for power loss pin (PE15)
186 // Enable the SYSCFG clock for interrupts
187 RCC->APB2ENR |= RCC_APB2ENR_SYSCFGEN;
188 SYSCFG->EXTICR[3] |= SYSCFG_EXTICR4_EXTI15_PE; // Map PE15 to EXTI 15
189 EXTI->IMR |= EXTI_IMR_MR15; // Unmask EXTI15
190 EXTI->FTSR |= EXTI_FTSR_TR15; // Enable the falling edge trigger (active low reset)
191 NVIC_SetPriority(EXTI15_10_IRQn, 15); // allow other interrupts to preempt this one (especially systick and dma)
192
193 NVIC_SetPriority(CAN1_RX0_IRQn, 6); // highest RTOS priority
194 NVIC_SetPriority(CAN2_RX0_IRQn, 7);
195 NVIC_SetPriority(CAN1_SCE_IRQn, 10);
196
197 NVIC_EnableIRQ(CAN1_RX0_IRQn);
198 NVIC_EnableIRQ(CAN1_SCE_IRQn);
199 NVIC_EnableIRQ(CAN2_RX0_IRQn);
200 NVIC_EnableIRQ(EXTI15_10_IRQn);
201}

203static void can_rx_irq_handler(CAN_TypeDef* can_h) {
204 portBASE_TYPE xHigherPriorityTaskWoken;
205 xHigherPriorityTaskWoken = pdFALSE;
206
207 // TODO: track FIFO overrun and full errors
208 if (can_h->RF0R & CAN_RF0R_FOVR0) // FIFO Overrun
209 can_h->RF0R &= ~(CAN_RF0R_FOVR0);
210
211 if (can_h->RF0R & CAN_RF0R_FULL0) // FIFO Full
212 can_h->RF0R &= ~(CAN_RF0R_FULL0);
213
214 if (can_h->RF0R & CAN_RF0R_FMP0_Msk) // Release message pending
215 {
216 timestamped_frame_t* rx;
217 uint32_t cont;
218 if (bGetHeadForWrite(&b_rx_can, (void**)&rx, &cont) == 0) {
219 rx->frame_type = DAQ_FRAME_CAN_RX; // msg generated by CAN interrupt
220 rx->tick_ms = getTick();
221
222 rx->bus_id = (can_h == CAN1) ? BUS_ID_CAN1 : BUS_ID_CAN2;
223
224 // Get either StdId or ExtId
225 if (CAN_RI0R_IDE & can_h->sFIFOMailBox[0].RIR) {
226 rx->msg_id = CAN_EFF_FLAG | (((CAN_RI0R_EXID | CAN_RI0R_STID) & can_h->sFIFOMailBox[0].RIR) >> CAN_RI0R_EXID_Pos);
227 } else {
228 rx->msg_id = (CAN_RI0R_STID & can_h->sFIFOMailBox[0].RIR) >> CAN_TI0R_STID_Pos;
229 }
230
231 rx->dlc = (CAN_RDT0R_DLC & can_h->sFIFOMailBox[0].RDTR) >> CAN_RDT0R_DLC_Pos;
232
233 rx->data[0] = (uint8_t)(can_h->sFIFOMailBox[0].RDLR >> 0) & 0xFF;
234 rx->data[1] = (uint8_t)(can_h->sFIFOMailBox[0].RDLR >> 8) & 0xFF;
235 rx->data[2] = (uint8_t)(can_h->sFIFOMailBox[0].RDLR >> 16) & 0xFF;
236 rx->data[3] = (uint8_t)(can_h->sFIFOMailBox[0].RDLR >> 24) & 0xFF;
237 rx->data[4] = (uint8_t)(can_h->sFIFOMailBox[0].RDHR >> 0) & 0xFF;
238 rx->data[5] = (uint8_t)(can_h->sFIFOMailBox[0].RDHR >> 8) & 0xFF;
239 rx->data[6] = (uint8_t)(can_h->sFIFOMailBox[0].RDHR >> 16) & 0xFF;
240 rx->data[7] = (uint8_t)(can_h->sFIFOMailBox[0].RDHR >> 24) & 0xFF;
241
242 bCommitWrite(&b_rx_can, 1);
243
244 CanMsgTypeDef_t msg;
245 msg.Bus = can_h;
246 msg.IDE = (rx->msg_id & CAN_EFF_FLAG) ? 1 : 0;
247 if (msg.IDE == 1) msg.ExtId = rx->msg_id & ~CAN_EFF_FLAG;
248 else msg.StdId = rx->msg_id;
249 msg.DLC = rx->dlc;
250 memcpy(msg.Data, rx->data, 8);
251 qSendToBack(&q_rx_can, &msg);
252
253 if(msg.ExtId == GPS_TIME_MSG_ID) rtc_config_cb(&msg);
254 } else {
255 daq_hub.bcan_rx_overflow++;
256 }
257 can_h->RF0R |= (CAN_RF0R_RFOM0);
258 }
259
260 portYIELD_FROM_ISR(xHigherPriorityTaskWoken);
261}

294bool can_parse_error_status(uint32_t err, timestamped_frame_t* frame) {
295 //frame->echo_id = 0xFFFFFFFF;
296 frame->bus_id = 0;
297 frame->msg_id = CAN_ERR_FLAG | CAN_ERR_CRTL;
298 frame->dlc = CAN_ERR_DLC;
299 frame->data[0] = CAN_ERR_LOSTARB_UNSPEC;
300 frame->data[1] = CAN_ERR_CRTL_UNSPEC;
301 frame->data[2] = CAN_ERR_PROT_UNSPEC;
302 frame->data[3] = CAN_ERR_PROT_LOC_UNSPEC;
303 frame->data[4] = CAN_ERR_TRX_UNSPEC;
304 frame->data[5] = 0;
305 frame->data[6] = 0;
306 frame->data[7] = 0;
307
308 if ((err & CAN_ESR_BOFF) != 0) {
309 frame->msg_id |= CAN_ERR_BUSOFF;
310 }
311
312 /*
313 uint8_t tx_error_cnt = (err>>16) & 0xFF;
314 uint8_t rx_error_cnt = (err>>24) & 0xFF;
315 */
316
317 if (err & CAN_ESR_EPVF) {
318 frame->data[1] |= CAN_ERR_CRTL_RX_PASSIVE | CAN_ERR_CRTL_TX_PASSIVE;
319 } else if (err & CAN_ESR_EWGF) {
320 frame->data[1] |= CAN_ERR_CRTL_RX_WARNING | CAN_ERR_CRTL_TX_WARNING;
321 }
322
323 uint8_t lec = (err >> 4) & 0x07;
324 if (lec != 0) { /* protocol error */
325 switch (lec) {
326 case 0x01: /* stuff error */
327 frame->msg_id |= CAN_ERR_PROT;
328 frame->data[2] |= CAN_ERR_PROT_STUFF;
329 break;
330 case 0x02: /* form error */
331 frame->msg_id |= CAN_ERR_PROT;
332 frame->data[2] |= CAN_ERR_PROT_FORM;
333 break;
334 case 0x03: /* ack error */
335 frame->msg_id |= CAN_ERR_ACK;
336 break;
337 case 0x04: /* bit recessive error */
338 frame->msg_id |= CAN_ERR_PROT;
339 frame->data[2] |= CAN_ERR_PROT_BIT1;
340 break;
341 case 0x05: /* bit dominant error */
342 frame->msg_id |= CAN_ERR_PROT;
343 frame->data[2] |= CAN_ERR_PROT_BIT0;
344 break;
345 case 0x06: /* CRC error */
346 frame->msg_id |= CAN_ERR_PROT;
347 frame->data[3] |= CAN_ERR_PROT_LOC_CRC_SEQ;
348 break;
349 default:
350 break;
351 }
352 }
353
354 return true;
355}
