77static bool readPin(uint8_t bank_pin, daq_tx_frame_writer_t* tx_msg)
78{
79 if (DAQ_PIN_VAL_LENGTH + DAQ_CMD_LENGTH + DAQ_BANK_LENGTH + DAQ_PIN_LENGTH > 64 - tx_msg->curr_bit)
80 {
81 // make room so entire message will fit
82 flushDaqFrame(tx_msg);
83 }
84 // add bank and pin
85 appendDataToFrame(tx_msg, (bank_pin << DAQ_CMD_LENGTH) | DAQ_RPLY_READ_PIN,
86 DAQ_CMD_LENGTH + DAQ_BANK_LENGTH + DAQ_PIN_LENGTH);
87
88 // get data
89 uint8_t val = 0;
90 GPIO_TypeDef * bank;
91 switch(bank_pin & DAQ_BANK_MASK)
92 {
93 case 0:
94 bank = GPIOA;
95 break;
96 case 1:
97 bank = GPIOB;
98 break;
99 case 2:
100 bank = GPIOC;
101 break;
102#ifdef GPIOD
103 case 3:
104 bank = GPIOD;
105 break;
106#endif
107#ifdef GPIOE
108 case 4:
109 bank = GPIOE;
110 break;
111#endif
112#ifdef GPIOF
113 case 5:
114 bank = GPIOF;
115 break;
116#endif
117 default:
118 bank = NULL;
119 }
120 if (bank == NULL)
121 {
122 val = DAQ_PIN_VAL_ERROR;
123 }
124 else
125 {
126 val = PHAL_readGPIO(bank, (bank_pin >> DAQ_BANK_LENGTH) & DAQ_PIN_MASK);
127 }
128
129 // place data, based on error checking, should be within the same frame
130 appendDataToFrame(tx_msg, val, DAQ_PIN_VAL_LENGTH);
131
132 return false;
133}

137static bool readVar(uint8_t var_id, daq_tx_frame_writer_t* tx_msg)
138{
139 if (tracked_vars[var_id].bit_length + DAQ_CMD_LENGTH + DAQ_ID_LENGTH > 64)
140 {
141 // var too large
142 appendDataToFrame(tx_msg, (var_id << DAQ_CMD_LENGTH) | DAQ_RPLY_READ_ERROR,
143 DAQ_CMD_LENGTH + DAQ_ID_LENGTH);
144 return true;
145 }
146 if (tracked_vars[var_id].bit_length + DAQ_CMD_LENGTH + DAQ_ID_LENGTH > 64 - tx_msg->curr_bit)
147 {
148 // make room so entire message will fit
149 flushDaqFrame(tx_msg);
150 }
151 // add var id
152 appendDataToFrame(tx_msg, (var_id << DAQ_CMD_LENGTH) | DAQ_RPLY_READ,
153 DAQ_CMD_LENGTH + DAQ_ID_LENGTH);
154
155 // get data
156 uint64_t temp_storage = 0;
157 if (tracked_vars[var_id].has_read_func)
158 {
159 tracked_vars[var_id].read_func_a(&temp_storage);
160 }
161 else
162 {
163 v_memcpy(&temp_storage, tracked_vars[var_id].read_var_a, (tracked_vars[var_id].bit_length + 7) / 8);
164 }
165
166 // place data, based on error checking, should be within the same frame
167 appendDataToFrame(tx_msg, temp_storage, tracked_vars[var_id].bit_length);
168
169 return false;

172// write variable
173// returns true if error
174static bool writeVar(uint8_t var_id, daq_rx_frame_reader_t* rx_msg, daq_tx_frame_writer_t* tx_msg)
175{
176 if (tracked_vars[var_id].is_read_only)
177 {
178 // can't write to read only variable
179 appendDataToFrame(tx_msg, (var_id << DAQ_CMD_LENGTH) | DAQ_RPLY_WRITE_ERROR,
180 DAQ_CMD_LENGTH + DAQ_ID_LENGTH);
181 return true;
182 }
183
184 rx_msg->curr_bit += DAQ_ID_LENGTH;
185
186 // get data
187 uint64_t temp_storage = (*(rx_msg->raw_data_a) >> rx_msg->curr_bit) &
188 ~(0xFFFFFFFFFFFFFFFF << (tracked_vars[var_id].bit_length));
189 // place data
190 if (tracked_vars[var_id].has_write_func)
191 {
192 tracked_vars[var_id].write_func_a(&temp_storage);
193 }
194 else
195 {
196 v_memcpy(tracked_vars[var_id].write_var_a, &temp_storage,
197 (tracked_vars[var_id].bit_length + 7) / 8);
198 }
199
200 rx_msg->curr_bit += tracked_vars[var_id].bit_length;
201
202 return false;
203}

266void daq_command_callback(CanMsgTypeDef_t* msg_header_a)
267{
268 daq_tx_frame_writer_t tx_writer = {.data={0}, .curr_bit=0};
269 daq_rx_frame_reader_t rx_reader = {.curr_bit=0};
270 rx_reader.raw_data_a = (uint64_t *) msg_header_a->Data;
271
272 while (rx_reader.curr_bit <= msg_header_a->DLC * 8 - DAQ_CMD_LENGTH - DAQ_ID_LENGTH)
273 {
274 uint8_t command = (*(rx_reader.raw_data_a) >> rx_reader.curr_bit) & DAQ_CMD_MASK;
275 rx_reader.curr_bit += DAQ_CMD_LENGTH;
276 switch(command)
277 {
278 case DAQ_CMD_READ:
279 readVar((*(rx_reader.raw_data_a) >> rx_reader.curr_bit) & DAQ_ID_MASK, &tx_writer);
280 rx_reader.curr_bit += DAQ_ID_LENGTH;
281 break;
282 case DAQ_CMD_WRITE:
283 writeVar((*(rx_reader.raw_data_a) >> rx_reader.curr_bit) & DAQ_ID_MASK, &rx_reader, &tx_writer);
284 // writeVar increments the curr_bit internally
285 break;
286 case DAQ_CMD_LOAD: // FORMAT: 4 character file name
287 loadFile(&rx_reader, &tx_writer);
288 // loadFile increments the curr_bit internally
289 break;
290 case DAQ_CMD_SAVE:
291 saveFile(&rx_reader, &tx_writer);
292 // saveFile increments the curr_bit internally
293 break;
294 case DAQ_CMD_PUB_START:
295 pubVarStart((*(rx_reader.raw_data_a) >> rx_reader.curr_bit) & DAQ_ID_MASK, &rx_reader);
296 // pubVarStart increments the curr_bit internally
297 break;
298 case DAQ_CMD_PUB_STOP:
299 pubVarStop((*(rx_reader.raw_data_a) >> rx_reader.curr_bit) & DAQ_ID_MASK);
300 rx_reader.curr_bit += DAQ_ID_LENGTH;
301 break;
302 case DAQ_CMD_READ_PIN:
303 readPin((*(rx_reader.raw_data_a) >> rx_reader.curr_bit) & (DAQ_BANK_MASK | (DAQ_PIN_MASK << DAQ_PIN_LENGTH)), &tx_writer);
304 rx_reader.curr_bit += DAQ_BANK_LENGTH + DAQ_PIN_LENGTH;
305 default:
306 __asm__("nop"); // Do nothing so we can place a breakpoint
307 }
308 }
309 flushDaqFrame(&tx_writer);
310}

315void daqPeriodicBase()
316{

318 daq_tick += 1;
319
320 // publish
321 for (uint8_t i = 0; i < num_vars; i++)
322 {
323 if (tracked_vars[i].pub_period != 0 &&
324 daq_tick % tracked_vars[i].pub_period == 0)
325 {
326 // read var and publish
327
328 if (tracked_vars[i].bit_length + DAQ_CMD_LENGTH + DAQ_ID_LENGTH > 64 - tx_msg.curr_bit)
329 {

331 flushDaqFrame(&tx_msg);
332 }
333
334 // add var id

357// add data to a tx frame
358// returns true if errror (i.e. bit_length too long for one frame)
359static bool appendDataToFrame(daq_tx_frame_writer_t* tx_msg, uint64_t data, uint8_t bit_length)
360{
361 // data too long?
362 if (bit_length > 64)
363 {
364 return true;
365 }
366 // is there enough room?
367 if (bit_length > 64 - tx_msg->curr_bit)
368 {
369 flushDaqFrame(tx_msg);
370 }
371
372 tx_msg->raw_data |= ((data & ~(0xFFFFFFFFFFFFFFFF << bit_length)) << tx_msg->curr_bit);
373 tx_msg->curr_bit += bit_length;
374
375 return false;
376}
