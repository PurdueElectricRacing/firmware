60void eth_update_periodic(void) {
61 static uint8_t eth_startup = 0;
62 if (!eth_startup) {
63 mDelay(250); // Wait for module to kick up
64 eth_startup = 1;
65 }
66
67 switch (daq_hub.eth_state) {
68 case ETH_LINK_IDLE:
69 if (eth_init() == ETH_ERROR_NONE) {
70 // Chip initialized, setup UDP socket
71 if (eth_udp_init() == ETH_ERROR_NONE)
72 daq_hub.eth_state = ETH_LINK_STARTING;
73 else
74 daq_hub.eth_state = ETH_LINK_FAIL;
75 } else {
76 daq_hub.eth_state = ETH_LINK_FAIL;
77 }
78 break;

83 debug_printf("UDP UP!\n");
84 bActivateTail(&b_rx_can, RX_TAIL_UDP);
85 daq_hub.eth_state = ETH_LINK_UP;
86 }
87 break;

90 daq_hub.eth_state = ETH_LINK_UP;
91 PHAL_writeGPIO(CONNECTION_LED_PORT, CONNECTION_LED_PIN, 1);

93 eth_udp_send_periodic();

97 PHAL_writeGPIO(CONNECTION_LED_PORT, CONNECTION_LED_PIN, 0);

115static void eth_reset_error(void) {
116 // Do not retry immediately
117 if (!(getTick() - daq_hub.eth_last_error_time > ETH_ERROR_RETRY_MS))
118 return;
119
120 if (daq_hub.eth_tcp_state == ETH_TCP_FAIL) {
121 daq_hub.eth_tcp_state = ETH_TCP_IDLE; // Retry TCP only
122 daq_hub.eth_last_err = ETH_ERROR_NONE;
123 daq_hub.eth_last_err_res = 0;
124 PHAL_writeGPIO(ERROR_LED_PORT, ERROR_LED_PIN, 0);
125 } else if (daq_hub.eth_last_err != ETH_ERROR_NONE || daq_hub.eth_state == ETH_LINK_FAIL) {
126 daq_hub.eth_state = ETH_LINK_IDLE; // Retry
127 daq_hub.eth_last_err = ETH_ERROR_NONE;
128 daq_hub.eth_last_err_res = 0;
129 PHAL_writeGPIO(ERROR_LED_PORT, ERROR_LED_PIN, 0);
130 }
131
132 if (daq_hub.eth_state == ETH_LINK_UP) {
133 if (daq_hub.eth_tcp_state == ETH_TCP_ESTABLISHED)
134 PHAL_toggleGPIO(CONNECTION_LED_PORT, CONNECTION_LED_PIN); // Blink on TCP
135 else
136 PHAL_writeGPIO(CONNECTION_LED_PORT, CONNECTION_LED_PIN, 1);
137 } else {
138 PHAL_writeGPIO(CONNECTION_LED_PORT, CONNECTION_LED_PIN, 0);
139 }
140}

143static int8_t eth_init(void) {
144 PHAL_writeGPIO(ETH_RST_PORT, ETH_RST_PIN, 0);
145 osDelay(ETH_PHY_RESET_PERIOD_MS);
146 PHAL_writeGPIO(ETH_RST_PORT, ETH_RST_PIN, 1);
147 osDelay(ETH_PHY_RESET_PERIOD_MS); // datasheet says 50 ms
148
149 // Check for sign of life
150 uint8_t ret = getVERSIONR();
151 if (ret != (uint8_t)ETH_PHY_VERSION_ID) {
152 eth_handle_error(ETH_ERROR_VERS, ret);
153 return ETH_ERROR_VERS;
154 }
155
156 // W5500 has 16kB memory for RX/TX each internally. It doesnt matter how the block is distributed
157 // across its 8 internal sockets, so allocate all 16kB to the sockets we actually use.
158 // 0: UDP broadcast (don't need RX buffer, need big TX)
159 // 1: TCP/UDS (need big RX buffer for bootloader, only need small TX for ack messages)
160 // 2/3/4: FTP
161 uint8_t rxBufSizes[] = {8, 8, 0, 0, 0, 0, 0, 0}; // kB
162 uint8_t txBufSizes[] = {8, 8, 0, 0, 0, 0, 0, 0}; // kB
163
164 if (wizchip_init(txBufSizes, rxBufSizes)) {
165 eth_handle_error(ETH_ERROR_INIT, 0);
166 return ETH_ERROR_INIT;
167 }
168
169 wizchip_setnetinfo(&eth_config.net_info);
170
171 return ETH_ERROR_NONE;
172}

193static void eth_udp_send_periodic(void) {
194 int32_t ret;
195 timestamped_frame_t* buf;
196 timestamped_frame_t* frame;
197 uint32_t consecutive_items;
198
199 if (daq_hub.eth_state == ETH_LINK_UP) {
200 if (bGetTailForRead(&b_rx_can, RX_TAIL_UDP, (void**)&buf, &consecutive_items) == 0) {
201 if (consecutive_items > UDP_MAX_WRITE_COUNT)
202 consecutive_items = UDP_MAX_WRITE_COUNT; // limit
203 // Write time :D
204 ret = sendto(eth_config.udp_bc_sock, (uint8_t*)buf, consecutive_items * sizeof(*buf), eth_config.udp_bc_addr, eth_config.udp_bc_port);
205 if (ret < consecutive_items * sizeof(*buf)) {
206 eth_handle_error(ETH_ERROR_UDP_SEND, ret);
207 bCommitRead(&b_rx_can, RX_TAIL_UDP, (ret / sizeof(*buf)));
208 } else {
209 bCommitRead(&b_rx_can, RX_TAIL_UDP, consecutive_items);
210 }
211 }
212 }

226static int8_t eth_init_tcp(void) {
227 // Open socket
228 int8_t res;
229 res = socket(eth_config.tcp_sock, SOCK_STREAM, eth_config.tcp_port, 0);
230 if (res != eth_config.tcp_sock) {
231 eth_handle_error(ETH_ERROR_TCP_SOCK, res);
232 return ETH_ERROR_TCP_SOCK;
233 }
234
235 // Set to non-blocking
236 uint8_t io_mode = SOCK_IO_NONBLOCK;
237 ctlsocket(eth_config.tcp_sock, CS_SET_IOMODE, &io_mode);
238
239 // Set socket to listen -> verify by checking SR went to ESTABLISHED
240 res = listen(eth_config.tcp_sock);
241 if (res != SOCK_OK) {
242 close(eth_config.tcp_sock);
243 eth_handle_error(ETH_ERROR_TCP_LISTEN, res);
244 return ETH_ERROR_TCP_LISTEN;
245 }
246
247 return ETH_ERROR_NONE;
248}

250static void eth_tcp_update(void) {
251 if (daq_hub.eth_state != ETH_LINK_UP)
252 return;
253
254 uint8_t stat;
255 switch (daq_hub.eth_tcp_state) {
256 case ETH_TCP_IDLE:
257 if (eth_init_tcp() == ETH_ERROR_NONE)
258 daq_hub.eth_tcp_state = ETH_TCP_LISTEN;
259 else
260 daq_hub.eth_tcp_state = ETH_TCP_FAIL;
261 break;
262 case ETH_TCP_LISTEN:
263 stat = getSn_SR(eth_config.tcp_sock);
264 if (stat == SOCK_ESTABLISHED) {
265 debug_printf("TCP UP!\n");
266 daq_hub.eth_tcp_state = ETH_TCP_ESTABLISHED; // Connected
267 }
268 break;
269 case ETH_TCP_ESTABLISHED:
270 stat = getSn_SR(eth_config.tcp_sock);
271 if (stat == SOCK_CLOSE_WAIT) {
272 close(eth_config.tcp_sock);
273 daq_hub.eth_tcp_state = ETH_TCP_IDLE;
274 } else {
275 /* TCP UP routines */
276 eth_tcp_receive_periodic(); // RX
277 eth_tcp_send_periodic(); // TX
278 }
279 break;
280 }

283static void conv_tcp_frame_to_can_msg(timestamped_frame_t* tcp_frame, CanMsgTypeDef_t* can_msg) {
284 can_msg->Bus = tcp_frame->bus_id == BUS_ID_CAN1 ? CAN1 : CAN2;
285
286 if (tcp_frame->msg_id & CAN_EFF_FLAG) {
287 can_msg->IDE = 1;
288 can_msg->ExtId = tcp_frame->msg_id & CAN_EFF_MASK;
289 } else {
290 can_msg->IDE = 0;
291 can_msg->StdId = tcp_frame->msg_id & CAN_SFF_MASK;
292 }
293
294 can_msg->DLC = tcp_frame->dlc;
295 for (uint8_t i = 0; i < 8; ++i)
296 can_msg->Data[i] = tcp_frame->data[i];
297}
