57void eth_update_periodic(void) {
58 static uint8_t eth_startup = 0;
59 if (!eth_startup) {
60 mDelay(250); // Wait for module to kick up
61 eth_startup = 1;
62 }
63
64 switch (daq_hub.eth_state) {
65 case ETH_LINK_IDLE:
66 if (eth_init() == ETH_ERROR_NONE) {
67 // Chip initialized, setup UDP socket
68 if (eth_udp_init() == ETH_ERROR_NONE)
69 daq_hub.eth_state = ETH_LINK_STARTING;
70 else
71 daq_hub.eth_state = ETH_LINK_FAIL;
72 } else {
73 daq_hub.eth_state = ETH_LINK_FAIL;
74 }
75 break;
76 case ETH_LINK_DOWN: // intentional fall through
77 case ETH_LINK_STARTING:
78 if (eth_get_link_up()) {
79 PHAL_writeGPIO(CONNECTION_LED_PORT, CONNECTION_LED_PIN, 1);
80 debug_printf("UDP UP!\n");
81 daq_hub.eth_state = ETH_LINK_UP;
82 }
83 break;
84 case ETH_LINK_UP:
85 if (eth_get_link_up()) {
86 daq_hub.eth_state = ETH_LINK_UP;
87 PHAL_writeGPIO(CONNECTION_LED_PORT, CONNECTION_LED_PIN, 1);
88 /* Eth link UP routines */
89 eth_udp_send_periodic();
90 } else {
91 daq_hub.eth_state = ETH_LINK_DOWN;
92 PHAL_writeGPIO(CONNECTION_LED_PORT, CONNECTION_LED_PIN, 0);
93 }
94 break;
95 case ETH_LINK_FAIL:
96 // Stay in fail state until eth_reset_error() moves us out
97 break;
98 }
99
100 eth_reset_error();
101}

112static void eth_reset_error(void) {
113 // Do not retry immediately
114 if (!(getTick() - daq_hub.eth_last_error_time > ETH_ERROR_RETRY_MS))
115 return;
116
117 if (daq_hub.eth_tcp_state == ETH_TCP_FAIL) {
118 daq_hub.eth_tcp_state = ETH_TCP_IDLE; // Retry TCP only
119 daq_hub.eth_last_err = ETH_ERROR_NONE;
120 daq_hub.eth_last_err_res = 0;
121 PHAL_writeGPIO(ERROR_LED_PORT, ERROR_LED_PIN, 0);
122 } else if (daq_hub.eth_last_err != ETH_ERROR_NONE || daq_hub.eth_state == ETH_LINK_FAIL) {
123 daq_hub.eth_state = ETH_LINK_IDLE; // Retry
124 daq_hub.eth_last_err = ETH_ERROR_NONE;
125 daq_hub.eth_last_err_res = 0;
126 PHAL_writeGPIO(ERROR_LED_PORT, ERROR_LED_PIN, 0);
127 }
128
129 if (daq_hub.eth_state == ETH_LINK_UP) {
130 if (daq_hub.eth_tcp_state == ETH_TCP_ESTABLISHED)
131 PHAL_toggleGPIO(CONNECTION_LED_PORT, CONNECTION_LED_PIN); // Blink on TCP
132 else
133 PHAL_writeGPIO(CONNECTION_LED_PORT, CONNECTION_LED_PIN, 1);
134 } else {
135 PHAL_writeGPIO(CONNECTION_LED_PORT, CONNECTION_LED_PIN, 0);
136 }
137}

140static int8_t eth_init(void) {
141 PHAL_writeGPIO(ETH_RST_PORT, ETH_RST_PIN, 0);
142 osDelay(ETH_PHY_RESET_PERIOD_MS);
143 PHAL_writeGPIO(ETH_RST_PORT, ETH_RST_PIN, 1);
144 osDelay(ETH_PHY_RESET_PERIOD_MS); // datasheet says 50 ms
145
146 // Check for sign of life
147 uint8_t ret = getVERSIONR();
148 if (ret != (uint8_t)ETH_PHY_VERSION_ID) {
149 eth_handle_error(ETH_ERROR_VERS, ret);
150 return ETH_ERROR_VERS;
151 }
152
153 // W5500 has 16kB memory for RX/TX each internally. It doesnt matter how the block is distributed
154 // across its 8 internal sockets, so allocate all 16kB to the sockets we actually use.
155 // 0: UDP broadcast (don't need RX buffer, need big TX)
156 // 1: TCP/UDS (need big RX buffer for bootloader, only need small TX for ack messages)
157 // 2/3/4: FTP
158 uint8_t rxBufSizes[] = {8, 8, 0, 0, 0, 0, 0, 0}; // kB
159 uint8_t txBufSizes[] = {8, 8, 0, 0, 0, 0, 0, 0}; // kB
160
161 if (wizchip_init(txBufSizes, rxBufSizes)) {
162 eth_handle_error(ETH_ERROR_INIT, 0);
163 return ETH_ERROR_INIT;
164 }
165
166 wizchip_setnetinfo(&eth_config.net_info);
167
168 return ETH_ERROR_NONE;
169}

176static int8_t eth_udp_init(void) {
177 // UDP Broadcast address
178 // 255.255.255.255 is not forwarded by routers
179 // local address | (~subnet address) is forwarded
180 int8_t bc_sock;
181 bc_sock = socket(eth_config.udp_bc_sock, SOCK_DGRAM, eth_config.udp_bc_port, 0);
182 if (bc_sock != eth_config.udp_bc_sock) {
183 eth_handle_error(ETH_ERROR_UDP_SOCK, bc_sock);
184 return ETH_ERROR_UDP_SOCK;
185 }
186
187 return ETH_ERROR_NONE;
188}

203 eth_handle_error(ETH_ERROR_UDP_SEND, ret);
