78static void _sd_write_periodic(bool bypass) {
79 timestamped_frame_t* buf;
80 uint32_t consecutive_items;
81 UINT bytes_written;
82 FRESULT result;
83
84 if (daq_hub.sd_state != SD_STATE_ACTIVE)
85 return;
86
87 // Use the total item count, not contiguous for the threshold
88 consecutive_items = SPMC_master_get_total(&queue, &buf);
89 if (!(bypass || consecutive_items >= SD_MAX_WRITE_COUNT)) {
90 return;
91 }
92
93 if (SPMC_master_peek_batch(&queue, &buf) == 0) {
94 daq_hub.sd_rx_overflow++;
95 return;
96 }
97
98 consecutive_items = SPMC_master_peek_batch(&queue, &buf);
99 if (consecutive_items > SD_MAX_WRITE_COUNT) {
100 consecutive_items = SD_MAX_WRITE_COUNT; // enforce the limit
101 }
102
103 // Write time :D
104 PHAL_writeGPIO(SD_ACTIVITY_LED_PORT, SD_ACTIVITY_LED_PIN, 1);
105 result = f_write(&daq_hub.log_fp, buf, consecutive_items * sizeof(*buf), &bytes_written);
106 if (result != FR_OK) {
107 sd_handle_error(SD_ERROR_WRITE, result);
108 } else {
109 daq_hub.last_write_ms = getTick();
110 SPMC_master_commit(&queue, bytes_written/sizeof(*buf));
111 sd_file_sync(); // fsync takes only 4 ticks and ensures sure cache is flushed on close
112 }
113 PHAL_writeGPIO(SD_ACTIVITY_LED_PORT, SD_ACTIVITY_LED_PIN, 0);
114}

132 }

135void sd_update_periodic(void) {
136 FRESULT result;
137
138 daq_hub.log_enable_sw = PHAL_readGPIO(LOG_ENABLE_PORT, LOG_ENABLE_PIN);
139 if (!daq_hub.log_enable_sw) {
140 PHAL_writeGPIO(SD_DETECT_LED_PORT, SD_DETECT_LED_PIN, 0);
141 sd_shutdown();

143 } else {
144 PHAL_writeGPIO(SD_DETECT_LED_PORT, SD_DETECT_LED_PIN, 1);
145 }
146
147 switch (daq_hub.sd_state) {
148 case SD_STATE_IDLE:
149 if (SD_Detect() != SD_PRESENT) {
150 PHAL_writeGPIO(SD_DETECT_LED_PORT, SD_DETECT_LED_PIN, 0);
151 break;
152 }
153
154 result = f_mount(&daq_hub.fat_fs, "", 1);
155 if (result != FR_OK) {
156 sd_handle_error(SD_ERROR_MOUNT, result);
157 PHAL_writeGPIO(SD_DETECT_LED_PORT, SD_DETECT_LED_PIN, 0);
158 break;
159 }
160
161 daq_hub.sd_state = SD_STATE_MOUNTED;
162 PHAL_writeGPIO(SD_DETECT_LED_PORT, SD_DETECT_LED_PIN, 1);
163 debug_printf("SD UP!\n");
164 break;
165 case SD_STATE_MOUNTED:
166 result = sd_create_new_file();
167 if (result == FR_OK) {
168 daq_hub.sd_state = SD_STATE_ACTIVE;
169 daq_hub.log_start_ms = getTick();
170 }
171 break;
172 case SD_STATE_ACTIVE:
173 if (getTick() - daq_hub.last_write_ms > SD_WRITE_PERIOD_MS) {
174 sd_write_periodic();
175 }
176
177 if (getTick() - daq_hub.last_file_ms > SD_NEW_FILE_PERIOD_MS) {
178 sd_create_new_file();
179 }
180 break;
181 }
182
183 sd_reset_error();
184}
