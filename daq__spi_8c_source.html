22/*
23 * W5500 uses a custom framed multi-byte SPI format that requires
24 * CS to be low for the entire duration of the multi-byte transaction.
25 * Hence the SW CS. Since the W5500 driver pulls CS manually using this
26 * callback before calling SPI_transfer, and the SPI peripheral needs to be
27 * enabled before CS, we enable it here and use a special function in the PHAL
28 * that doesn't pull CS/enable SPI
29 */
30static void cs_sel(void) {
31 eth_spi_config.periph->CR1 |= SPI_CR1_SPE;
32 PHAL_writeGPIO(ETH_CS_PORT, ETH_CS_PIN, 0);
33}
34
35static void cs_desel(void) {
36 eth_spi_config.periph->CR1 &= ~SPI_CR1_SPE;
37 PHAL_writeGPIO(ETH_CS_PORT, ETH_CS_PIN, 1);
38}

41static void PHAL_SPI_transfer_noDMA_DAQW5500Only(SPI_InitConfig_t* spi, uint32_t addr, const uint8_t* tx_data, uint32_t txlen, uint32_t rxlen, uint8_t* rx_data) {
42 // crit_enter() and crit_exit() in the W5500 callback
43 // takes lock before entering this function, so this fn does not lock nor fail
44
45 rx_data += txlen;
46
47 // DO NOT Enable SPI here
48 // The CS must come after enabling SPI, and since W5500 driver selects CS, we manually add SPE to the CS sel hook
49 //spi->periph->CR1 |= SPI_CR1_SPE;
50
51 // Add messages to TX FIFO
52 // Write the first data item to be transmitted into the SPI_DR register (this clears the TXE flag).
53 // The sequence begins when data are written into the SPI_DR register (Tx buffer).
54 // 24-bit address selection (mandatory)
55 spi->periph->DR = (addr & 0x00FF0000) >> 16;
56 while (!(spi->periph->SR & SPI_SR_TXE))
57 ;
58 spi->periph->DR = (addr & 0x0000FF00) >> 8;
59 while (!(spi->periph->SR & SPI_SR_TXE))
60 ;
61 spi->periph->DR = (addr & 0x000000FF) >> 0;
62 while (!(spi->periph->SR & SPI_SR_TXE))
63 ;
64 for (uint32_t i = 0; i < txlen; i++) {
65 spi->periph->DR = tx_data[i];
66 while (!(spi->periph->SR & SPI_SR_TXE))
67 ;
68 }
69
70 /*
71 * During discontinuous communications, there is a 2 APB clock period delay
72 * between the write operation to the SPI_DR register and BSY bit setting.
73 * As a consequence it is mandatory to wait first until TXE is set and then
74 * until BSY is cleared after writing the last data.
75 */
76 while (((spi->periph->SR) & SPI_SR_BSY))
77 ;
78
79 // Clear overrun
80 (void)spi->periph->DR;
81 (void)spi->periph->SR;
82
83 // RX
84 for (uint32_t i = 0; i < rxlen; i++) {
85 // Wait till SPI is not in active transaction, send dummy
86 while (((spi->periph->SR) & SPI_SR_BSY))
87 ;
88 spi->periph->DR = 0;
89
90 // Wait for RX FIFO to recieve a message, read it in
91 while (!(spi->periph->SR & SPI_SR_RXNE))
92 ;
93 rx_data[i] = (uint8_t)(spi->periph->DR);
94 }
95
96 // Do NOT wait until done, there is nothing to wait for (RX already received) and it will hang
97 //Wait till transaction is finished, disable spi, and clear the queue
98 //while ((spi->periph->SR & SPI_SR_BSY));
99}
