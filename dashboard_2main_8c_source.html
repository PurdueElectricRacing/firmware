203void preflight_task() {
204 static uint8_t counter = 0;
205
206 // run animation until for at least 1.5 seconds
207 if (counter >= 150) {
208 PHAL_writeGPIO(HEART_LED_GPIO_Port, HEART_LED_Pin, 0);
209 PHAL_writeGPIO(CONN_LED_GPIO_Port, CONN_LED_Pin, 0);
210 PHAL_writeGPIO(ERROR_LED_GPIO_Port, ERROR_LED_Pin, 0);
211
212 // spawn the other threads
213 createThread(pedalsPeriodic);
214 createThread(can_worker_task);
215
216 createThread(updateFaultDisplay);
217 createThread(heartbeat_led);
218 createThread(service_button_inputs);
219 createThread(send_version);
220 createThread(updateTelemetryPages);
221 createThread(sendTVParameters);
222 osThreadExit(); // Self delete
223 return;
224 }
225
226 if (counter % 10 == 0) { // Run every 100ms
227 preflight_animation();
228 }
229
230 counter++;
231}

243void preflight_animation(void) {
244 // Controls external LEDs since they are more visible when dash is in car
245 static uint32_t external_index;
246 static uint32_t sweep_index;
247
248 switch (sweep_index++ % 3) { // Creates a sweeping pattern
249 case 0:
250 PHAL_writeGPIO(HEART_LED_GPIO_Port, HEART_LED_Pin, 1);
251 PHAL_writeGPIO(CONN_LED_GPIO_Port, CONN_LED_Pin, 0);
252 PHAL_writeGPIO(ERROR_LED_GPIO_Port, ERROR_LED_Pin, 0);
253 break;
254 case 1:
255 PHAL_writeGPIO(HEART_LED_GPIO_Port, HEART_LED_Pin, 0);
256 PHAL_writeGPIO(CONN_LED_GPIO_Port, CONN_LED_Pin, 1);
257 PHAL_writeGPIO(ERROR_LED_GPIO_Port, ERROR_LED_Pin, 0);
258 break;
259 case 2:
260 PHAL_writeGPIO(HEART_LED_GPIO_Port, HEART_LED_Pin, 0);
261 PHAL_writeGPIO(CONN_LED_GPIO_Port, CONN_LED_Pin, 0);
262 PHAL_writeGPIO(ERROR_LED_GPIO_Port, ERROR_LED_Pin, 1);
263 break;
264 }
265
266 switch (external_index++ % 2) { // Creates a 50/50 blinking pattern
267 case 0:
268 PHAL_writeGPIO(BMS_LED_GPIO_Port, BMS_LED_Pin, 0);
269 PHAL_writeGPIO(IMD_LED_GPIO_Port, IMD_LED_Pin, 0);
270 PHAL_writeGPIO(PRCHG_LED_GPIO_Port, PRCHG_LED_Pin, 0);
271 break;
272 case 1:
273 PHAL_writeGPIO(BMS_LED_GPIO_Port, BMS_LED_Pin, 0);
274 PHAL_writeGPIO(IMD_LED_GPIO_Port, IMD_LED_Pin, 0);
275 PHAL_writeGPIO(PRCHG_LED_GPIO_Port, PRCHG_LED_Pin, 0);
276 break;
277 }

292void heartbeat_led() {
293 static uint8_t imd_prev_latched;
294 static uint8_t bms_prev_latched;
295
296 PHAL_toggleGPIO(HEART_LED_GPIO_Port, HEART_LED_Pin);
297
298 if ((xTaskGetTickCount() - last_can_rx_time_ms) >= CONN_LED_MS_THRESH) {
299 PHAL_writeGPIO(CONN_LED_GPIO_Port, CONN_LED_Pin, 1);
300 } else {
301 PHAL_writeGPIO(CONN_LED_GPIO_Port, CONN_LED_Pin, 0);
302 }
303
304 if (!can_data.main_hb.stale && can_data.main_hb.precharge_state) {
305 PHAL_writeGPIO(PRCHG_LED_GPIO_Port, PRCHG_LED_Pin, 0);
306 } else {
307 PHAL_writeGPIO(PRCHG_LED_GPIO_Port, PRCHG_LED_Pin, 1);
308 }
309
310 if (!can_data.precharge_hb.stale) {
311 if (can_data.precharge_hb.IMD) {
312 imd_prev_latched = 1;
313 }
314
315 if (can_data.precharge_hb.BMS) {
316 bms_prev_latched = 1;
317 }
318 } else {
319 PHAL_writeGPIO(IMD_LED_GPIO_Port, IMD_LED_Pin, 0);
320 PHAL_writeGPIO(BMS_LED_GPIO_Port, BMS_LED_Pin, 0);
321 }
322
323 PHAL_writeGPIO(IMD_LED_GPIO_Port, IMD_LED_Pin, !imd_prev_latched);
324 PHAL_writeGPIO(BMS_LED_GPIO_Port, BMS_LED_Pin, !bms_prev_latched);
325}

409void config_button_irqs() {
410 // Enable the SYSCFG clock for interrupts
411 RCC->APB2ENR |= RCC_APB2ENR_SYSCFGEN;
412
413 // Map EXTI lines to correct GPIO ports
414 // PC6, PC7 (EXTI6, 7)
415 SYSCFG->EXTICR[1] |= (SYSCFG_EXTICR2_EXTI6_PC | SYSCFG_EXTICR2_EXTI7_PC);
416 // PC8, PC9 (EXTI8, 9)
417 SYSCFG->EXTICR[2] |= (SYSCFG_EXTICR3_EXTI8_PC | SYSCFG_EXTICR3_EXTI9_PC);
418 // PB14, PB15 (EXTI14, 15)
419 SYSCFG->EXTICR[3] |= (SYSCFG_EXTICR4_EXTI14_PB | SYSCFG_EXTICR4_EXTI15_PB);
420
421 // Unmask interrupts (EXTI lines 6,7,8,9,14,15)
422 EXTI->IMR1 |= (EXTI_IMR1_IM6 | EXTI_IMR1_IM7 | EXTI_IMR1_IM8 |
423 EXTI_IMR1_IM9 | EXTI_IMR1_IM14 | EXTI_IMR1_IM15);
424
425 // Falling edge trigger only (pull-up buttons)
426 EXTI->RTSR1 &= ~(EXTI_RTSR1_RT6 | EXTI_RTSR1_RT7 | EXTI_RTSR1_RT8 |
427 EXTI_RTSR1_RT9 | EXTI_RTSR1_RT14 | EXTI_RTSR1_RT15);
428
429 EXTI->FTSR1 |= (EXTI_FTSR1_FT6 | EXTI_FTSR1_FT7 | EXTI_FTSR1_FT8 |
430 EXTI_FTSR1_FT9 | EXTI_FTSR1_FT14 | EXTI_FTSR1_FT15);
431
432 NVIC_EnableIRQ(EXTI9_5_IRQn);
433 NVIC_EnableIRQ(EXTI15_10_IRQn);
434}
