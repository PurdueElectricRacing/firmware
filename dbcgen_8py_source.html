11def generate_dbcs(context: SystemContext):
12 """
13 Generates DBC files for each bus in the system.
14 """
15 print("Generating DBCs...")
16
17 # Ensure output directory exists and is clean
18 if DBC_DIR.exists():
19 for f in DBC_DIR.glob("*.dbc"):
20 f.unlink()
21 DBC_DIR.mkdir(exist_ok=True)
22
23 git_hash = context.version
24
25 for bus_name, view in context.busses.items():
26 can_db = db.Database()
27
28 # Add nodes
29 for node_name in sorted(view.nodes):
30 can_db.nodes.append(db.Node(name=node_name, comment=""))
31
32 # Add messages
33 for msg in view.messages:
34 signals = []
35
36 # Sort signals by bit offset for deterministic output
37 sorted_signals = sorted(msg.signals, key=lambda x: x.bit_offset)
38
39 for sig in sorted_signals:
40 # Resolve choices (enums) for VAL_ table in DBC
41 choices = None
42 if sig.choices:
43 choices = {i: c for i, c in enumerate(sig.choices)}
44 elif sig.datatype in context.custom_types:
45 type_info = context.custom_types[sig.datatype]
46 if 'choices' in type_info:
47 choices = {i: c for i, c in enumerate(type_info['choices'])}
48 elif sig.datatype == 'bool':
49 choices = {0: "OFF", 1: "ON"}
50
51 signals.append(db.Signal(
52 name=sig.name,
53 start=sig.bit_offset,
54 length=sig.length,
55 byte_order="little_endian",
56 is_signed=sig.is_signed,
57 initial=sig.offset,
58 scale=sig.scale,
59 offset=sig.offset,
60 minimum=sig.min_val,
61 maximum=sig.max_val,
62 unit=sig.unit if sig.unit else "",
63 comment=sig.desc,
64 choices=choices
65 ))
66
67 # Use pre-calculated sender mapping
68 sender = view.sender_map.get(msg.name, "Vector__XXX")
69
70 can_db.messages.append(db.Message(
71 frame_id=msg.final_id,
72 name=msg.name,
73 length=msg.get_dlc(context.custom_types),
74 signals=signals,
75 comment=msg.desc,
76 is_extended_frame=msg.is_extended,
77 senders=[sender]
78 ))
79
80 filename = f"{bus_name}_{git_hash}.dbc"
81 output_path = DBC_DIR / filename
82 with open(output_path, 'w', newline='\n') as f:
83 f.write(can_db.as_dbc_string())
84
85 print_as_ok(f"Generated {filename}")
86
87 print_as_success("Successfully generated DBC files")
