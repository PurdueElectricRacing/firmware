104static DRESULT _sdio_disk_read(
105 BYTE* buff, /* Data buffer to store read data */
106 DWORD sector, /* Sector address (LBA) */
107 UINT count /* Number of sectors to read (1..128) */
108) {
109 SD_Error Status = SD_OK;
110
111 if ((SD_SDIO_Stat & STA_NOINIT)) {
112 return RES_NOTRDY;
113 }
114
115 if ((DWORD)buff & 3) {
116 DRESULT res = RES_OK;
117 DWORD scratch[BLOCK_SIZE / 4];
118
119 while (count--) {
120 res = _sdio_disk_read((void*)scratch, sector++, 1);
121
122 if (res != RES_OK) {
123 break;
124 }
125
126 memcpy(buff, scratch, BLOCK_SIZE);
127
128 buff += BLOCK_SIZE;
129 }
130
131 return res;
132 }
133
134 Status = SD_ReadMultiBlocks(buff, sector << 9, BLOCK_SIZE, count);
135
136 if (Status == SD_OK) {
137 SDTransferState State;
138
139 Status = SD_WaitReadOperation();
140
141 while ((State = SD_GetStatus()) == SD_TRANSFER_BUSY)
142 ;
143
144 if ((State == SD_TRANSFER_ERROR) || (Status != SD_OK)) {
145 DISKIO_LOG_RED("Read transfer error, state %d, status %d\n", State, Status);
146 return RES_ERROR;
147 } else {
148 return RES_OK;
149 }
150 } else {
151 return RES_ERROR;
152 }
153}

179static DRESULT _sdio_disk_write(
180 const BYTE* buff, /* Data to be written */
181 DWORD sector, /* Sector address (LBA) */
182 UINT count /* Number of sectors to write (1..128) */
183) {
184 SD_Error Status = SD_OK;

186
187 if (SD_Detect() != SD_PRESENT) {
188 return RES_NOTRDY;

190

192 if ((DWORD)buff & 3) {
193 DRESULT res = RES_OK;
194 DWORD scratch[BLOCK_SIZE / 4];
195
196 while (count--) {
197 memcpy(scratch, buff, BLOCK_SIZE);
198 res = _sdio_disk_write((void*)scratch, sector++, 1);
199

202 }

204 buff += BLOCK_SIZE;

216

263DWORD get_fattime(void) {
264 /* Returns current time packed into a DWORD variable */
265 RTC_timestamp_t time;
266 DWORD t;
267 if (PHAL_getTimeRTC(&time)) {
268 t = (time.date.year_bcd + 20) << 25;
269 t |= (time.date.month_bcd) << 21;
270 t |= (time.date.day_bcd) << 16;
271 t |= (time.time.hours_bcd) << 11;
272 t |= (time.time.minutes_bcd) << 5;
273 t |= (time.time.seconds_bcd) >> 1;
274 return t;
275 } else {
276 /* hard-coded date */
277 return ((DWORD)(2013 - 1980) << 25) /* Year 2013 */
278 | ((DWORD)7 << 21) /* Month 7 */
279 | ((DWORD)28 << 16) /* Mday 28 */
280 | ((DWORD)0 << 11) /* Hour 0 */
281 | ((DWORD)0 << 5) /* Min 0 */
282 | ((DWORD)0 >> 1); /* Sec 0 */
283 }
