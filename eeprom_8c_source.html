19// @funcname: initMem
20//
21// @brief: Initializes chip metadata. Attemps to read
22// and load all metadata from last run. If memory
23// isn't initialized, force_init can be set to
24// set a default version of 1 and mark the chip
25// as officially in use
26//
27// @param: wc_gpio_port: Mem lock port0
28// @param: wc_gpio_pin: Mem lock pin
29// @param: version: Version of app code
30// @param: force_init: Force initialization if memory isn't
31// initialized. Only set if version is truly 1
32int initMem(GPIO_TypeDef* wc_gpio_port, uint32_t wc_gpio_pin, uint16_t version, bool force_init) {
33 int ret;
34 uint16_t i, size, end;
35 uint8_t page[MICRO_PG_SIZE];
36
37 // Set WC pins
38 mem.wc_gpio_port = wc_gpio_port;
39 mem.wc_gpio_pin = wc_gpio_pin;
40
41 // Read EEPROM metadata
42 ret = readMem(0, (uint8_t*)&mem.phys, sizeof(struct phys_mem));
43 ee_memset(mem_zero, MICRO_PG_SIZE, 0);
44
45 // Ensure there are no errors
46 if (ret < 0) {
47 return ret;
48 }
49
50 // Check if the memory isn't initialized
51 if (!mem.phys.init) {
52 // It isn't, and we want it to be, so set to default values
53 if (force_init) {
54 mem.phys.init = true;
55 mem.phys.version = 1;
56 mem.init_physical = true;
57
58 return E_SUCCESS;
59 } else {
60 return -E_NO_INIT;
61 }
62 }
63
64 // Check if we have a version match
65 ret = checkVersion(version);
66
67 // Return error code if version is off
68 if (ret < 0) {
69 return -E_V_MISMATCH;
70 }
71
72 // Start loading process
73 ee_memset(page, NAME_LEN, 0);
74
75 for (i = 0; i < MAX_PAGES; i++) {
76 if (ee_memcheck(page, (char*)&mem.phys.filename[i], NAME_LEN) < 0) {
77 ret = readMem(mem.phys.pg_bound[i], mem.pg_addr[i], mem.pg_size[i]);
78
79 if (ret < 0) {
80 return ret;
81 }
82 }
83 }
84
85 mem.init_physical = true;
86
87 return E_SUCCESS;
88}

90// @funcname: checkVersion
91//
92// @brief: Checks version to ensure app code is new enough
93//
94// @param: dest: Pointer to location to set
95// @param: len: Length to set
96// @param: value: Value to set each memory address to
97// @param: Difference between current and chip versions
98// if app code is newer, -E_V_MISMATCH if app
99// code is old. -E_NO_INIT if memory hasn't
100// been initialized yet
101int checkVersion(uint16_t version) {
102 // Check if we're even initialized
103 if (!mem.phys.init) {
104 return -E_NO_INIT;
105 }
106
107 // Check if our software is a later version
108 if (mem.phys.version < version) {
109 return -E_V_MISMATCH;
110 } else {
111 return mem.phys.version - version;
112 }
113}

115// @funcname: mapMem
116//
117// @brief: Maps local address to chip address
118//
119// @param: addr: Pointer to local address
120// @param: len: Length of data
121// @param: fname: File name (NAME_LEN characters)
122// @param: bcmp: Backwards compatibility enabled. Disable for temp storage
123int mapMem(uint8_t* addr, uint16_t len, uint8_t* fname, bool bcmp) {
124 int i;
125 uint8_t null_name[NAME_LEN];
126
127 ee_memset(null_name, NAME_LEN, 0U);
128
129 // Check if we already found this memory on EEPROM
130 i = fnameSearch(fname);
131
132 if (i >= 0) {
133 mem.pg_addr[i] = addr;
134 mem.pg_size[i] = len;
135
136 return E_SUCCESS;
137 }
138
139 // Check for the first free location
140 for (i = 0; i < MAX_PAGES; i++) {
141 if (ee_memcheck(null_name, &mem.phys.filename[i * NAME_LEN], NAME_LEN) == 0) {
142 break;
143 }
144 }
145
146 // Check if we found a free location
147 if (i == MAX_PAGES) {
148 return -E_NO_MEM;
149 }
150
151 // Copy over new metadata
152 ee_memcpy(fname, &mem.phys.filename[i * NAME_LEN], NAME_LEN);
153 mem.phys.mem_size[i] = len;
154 mem.phys.bcmp[i] = bcmp;
155 mem.phys.pg_bound[i] = (i == 0) ? MACRO_PG_SIZE : ROUNDUP(mem.phys.pg_bound[i - 1] + mem.phys.mem_size[i - 1], MICRO_PG_SIZE);
156 mem.pg_addr[i] = addr;
157 mem.pg_size[i] = len;
158
159 return E_SUCCESS;
160}

274// @funcname: readMem
275//
276// @brief: Reads memory across micro page boundaries
277//
278// @param: phys_addr: On chip address of data
279// @param: loc_addr: Local address of data
280// @param: len: Length of data
281//
282// @return: E_SUCCESS if read, error code if failed
283static int readMem(uint16_t phys_addr, uint8_t* loc_addr, uint16_t len) {
284 int ret;
285 uint8_t page[MICRO_PG_SIZE];
286 uint16_t i, size, end;
287
288 // Calculate ending address
289 end = ROUNDUP(phys_addr + len, MICRO_PG_SIZE);
290
291 // Read page by page
292 for (i = phys_addr; i < end; i += MICRO_PG_SIZE) {
293 ret = readPage(i, page);
294
295 if (ret < 0) {
296 return ret;
297 }
298
299 // Check if micro page would overflow during memcpy
300 size = (i + 32 == end) ? MICRO_PG_SIZE - (len % MICRO_PG_SIZE) : MICRO_PG_SIZE;
301 ee_memcpy(page, (uint8_t*)(loc_addr + i), size);
302 }
303
304 return E_SUCCESS;
305}

307// @funcname: writePage
308//
309// @brief: Writes a single micro page to chip
310//
311// @param: addr: On chip address of data
312// @param: page: Pointer to data to write
313// @param: size: Length of write (capped at MICRO_PG_SIZE)
314//
315// @note: E_SUCCESS if written, -E_I2C if failed
316static int writePage(uint16_t addr, uint8_t* page, uint8_t size) {
317 uint8_t ret;
318
319 size = (size > MICRO_PG_SIZE) ? MICRO_PG_SIZE : size;
320
321 // Pull WC to allow for a write
322 PHAL_writeGPIO(mem.wc_gpio_port, mem.wc_gpio_pin, 0);
323
324 // Complete write
325 ret += !PHAL_I2C_gen_start(I2C1, MEM_ADDR | MODE_W, size + 2, PHAL_I2C_MODE_TX);
326 ret += !PHAL_I2C_write(I2C1, addr >> 8);
327 ret += !PHAL_I2C_write(I2C1, addr & 0xff);
328 ret += !PHAL_I2C_write_multi(I2C1, page, size);
329 ret += !PHAL_I2C_gen_stop(I2C1);
330
331 // Stop mem writes
332 PHAL_writeGPIO(mem.wc_gpio_port, mem.wc_gpio_pin, 1);
333
334 return ret ? -E_I2C : E_SUCCESS;
335}
