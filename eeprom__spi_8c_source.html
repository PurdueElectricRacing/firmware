153int mapMem(uint8_t* addr, uint16_t len, uint8_t* fname, bool bcmp) {
154 int i;
155 uint8_t null_name[NAME_LEN], ret;
156
157 ee_memset(null_name, NAME_LEN, 0U);
158
159 // Check if we already found this memory on EEPROM
160 i = fnameSearch(fname);
161
162 if (i >= 0) {
163 mem.pg_addr[i] = addr;
164 mem.pg_size[i] = len;
165 if (len < mem.phys.mem_size[i]) {
166 // Variable deleted, update size
167 mem.phys.mem_size[i] = mem.pg_size[i];
168 ee_request_flush_physical();
169 ret = readMem(mem.phys.pg_bound[i], mem.pg_addr[i], mem.pg_size[i]);
170 if (ret < 0)
171 return ret;
172 } else if (len > mem.phys.mem_size[i]) {
173 // Variable added, ensure enough space left in page
174 if ((mem.phys.mem_size[i] % MICRO_PG_SIZE) != 0 && (mem.phys.mem_size[i] % MICRO_PG_SIZE) + (len - mem.phys.mem_size[i]) <= MICRO_PG_SIZE) {
175 // Read portion that was saved before
176 ret = readMem(mem.phys.pg_bound[i], mem.pg_addr[i], mem.phys.mem_size[i]);
177 if (ret < 0)
178 return ret;
179 mem.phys.mem_size[i] = mem.pg_size[i];
180 ee_request_flush_physical();
181 // Make sure the new defaults get saved
182 requestFlush(fname);
183 } else
184 return -12;
185 } else {
186 // Files equal lengths, normal operation
187 ret = readMem(mem.phys.pg_bound[i], mem.pg_addr[i], mem.pg_size[i]);
188 if (ret < 0)
189 return ret;
190 }
191
192 return E_SUCCESS;
193 }
194
195 // Check for the first free location
196 for (i = 0; i < MAX_PAGES; i++) {
197 if (ee_memcheck(null_name, &mem.phys.filename[i * NAME_LEN], NAME_LEN) == 0) {
198 break;
199 }
200 }
201
202 // Check if we found a free location
203 if (i == MAX_PAGES) {
204 return -E_NO_MEM;
205 }
206
207 // Copy over new metadata
208 ee_memcpy(fname, &mem.phys.filename[i * NAME_LEN], NAME_LEN);
209 mem.phys.mem_size[i] = len;
210 mem.phys.bcmp[i] = bcmp;
211 mem.phys.pg_bound[i] = (i == 0) ? MACRO_PG_SIZE : ROUNDUP(mem.phys.pg_bound[i - 1] + mem.phys.mem_size[i - 1], MICRO_PG_SIZE);
212 mem.pg_addr[i] = addr;
213 mem.pg_size[i] = len;
214 ee_request_flush_physical();
215 requestFlush(fname); // Write to default values
216
217 return E_SUCCESS;

240void memBg(void) {
241 int ret;
242 static uint8_t page[MICRO_PG_SIZE];
243 uint16_t len, end;
244
245 // Check if we're waiting on foreground to execute a write
246 if (mem.write_pending || !mem.init_physical) {
247 return;
248 }
249
250 // Check for action
251 if (bg_state == BG_IDLE) {
252 if (mem.zero_req)
253 bg_state = BG_ZERO;
254 else if (mem.flush_physical)
255 bg_state = BG_FLUSH_META;
256 else if (mem.req_flush[0] | mem.req_flush[1])
257 bg_state = BG_FLUSH;
258 else if (mem.req_load[0] | mem.req_load[1])
259 bg_state = BG_LOAD;
260 else
261 return; // Nothing to do
262 addr = 0; // restart
263 curr_page = 0;
264 }
265
266 switch (bg_state) {
267 case BG_IDLE:
268 break;
269 case BG_ZERO:
270 mem.write_pending = true;
271 mem.source_loc = mem_zero;
272 mem.dest_loc = addr;
273 mem.update_len = MICRO_PG_SIZE;
274 addr += MICRO_PG_SIZE;
275 if (addr >= CHIP_SIZE) {
276 mem.zero_req = false;
277 bg_state = BG_IDLE;
278 }
279 break;
280 case BG_LOAD:
281 curr_page = ee_get_idx(mem.req_load);
282 addr = mem.phys.pg_bound[curr_page];
283 // Loads do all pages at once (don't want half data in memory when using it)
284 readMem(addr, mem.pg_addr[curr_page], mem.phys.mem_size[curr_page]);
285 ee_clear_idx(mem.req_load, curr_page);
286 bg_state = BG_IDLE;
287 addr = 0;
288 curr_page = 0;
289 break;
290 case BG_FLUSH:
291 if (curr_page == 0) {
292 curr_page = ee_get_idx(mem.req_flush);
293 addr = mem.phys.pg_bound[curr_page];
294 }
295 end = mem.phys.pg_bound[curr_page] + mem.pg_size[curr_page];
296 len = (addr + MICRO_PG_SIZE > end) ? (mem.pg_size[curr_page] % MICRO_PG_SIZE) : MICRO_PG_SIZE;
297 ret = readMem(addr, page, len);
298
299 if (ret < 0) {
300 return;
301 }
302
303 ret = ee_memcheck(page, (uint8_t*)(mem.pg_addr[curr_page] + addr - mem.phys.pg_bound[curr_page]), len);
304
305 if (ret < 0) {
306 mem.write_pending = true;
307 mem.dest_loc = addr;
308 mem.source_loc = (uint8_t*)(mem.pg_addr[curr_page] + addr - mem.phys.pg_bound[curr_page]);
309 mem.update_len = len;
310 }
311
312 addr += MICRO_PG_SIZE;
313
314 if (addr >= end) {
315 ee_clear_idx(mem.req_flush, curr_page);
316 addr = 0;
317 curr_page = 0;
318 bg_state = BG_IDLE;
319 }
320 break;
321 case BG_FLUSH_META:
322 // assumes curr_page and addr reset to 0 before initial
323 end = sizeof(struct phys_mem);
324 len = (addr + MICRO_PG_SIZE > end) ? (sizeof(struct phys_mem) % MICRO_PG_SIZE) : MICRO_PG_SIZE;
325 ret = readMem(addr, page, len);
326
327 if (ret < 0) {
328 return;
329 }
330
331 ret = ee_memcheck(page, ((uint8_t*)&mem.phys) + addr, len);
332
333 if (ret < 0) {
334 mem.write_pending = true;
335 mem.dest_loc = addr;
336 mem.source_loc = ((uint8_t*)&mem.phys) + addr;
337 mem.update_len = len;
338 }
339
340 addr += MICRO_PG_SIZE;
341
342 if (addr >= sizeof(struct phys_mem)) {
343 addr = 0;
344 curr_page = 0;
345 mem.flush_physical = 0;
346 bg_state = BG_IDLE;
347 }
348 break;
349 default:
350 return;
351 break;
352 }
353}

381static int readMem(uint16_t phys_addr, uint8_t* loc_addr, uint16_t len) {
382 int ret;
383 uint8_t page[MICRO_PG_SIZE];
384 uint16_t i, size, end;
385
386 // Calculate ending address
387 end = phys_addr + len; // (non-inclusive)
388
389 // Read page by page
390 for (i = phys_addr; i < end; i += MICRO_PG_SIZE) {
391 while (PHAL_SPI_busy(mem.spi))
392 ; // Block
393 ret = readPage(i, page);
394 while (PHAL_SPI_busy(mem.spi))
395 ; // Block
396
397 if (ret < 0) {
398 return ret;
399 }
400
401 // Check if micro page would overflow during memcpy
402 size = (i + MICRO_PG_SIZE > end) ? (len % MICRO_PG_SIZE) : MICRO_PG_SIZE;
403 ee_memcpy(page, (uint8_t*)(loc_addr + (i - phys_addr)), size);
404 }
405
406 return E_SUCCESS;
407}

420int writePage(uint16_t addr, uint8_t* page, uint8_t size) {
421 uint8_t ret;
422
423 size = (size > MICRO_PG_SIZE) ? MICRO_PG_SIZE : size;
424
425 static uint8_t tx_command[3];
426 static uint8_t rx_buff[3];
427
428 if (PHAL_SPI_busy(mem.spi))
429 return -E_SPI;
430
431 // Must send a E_WREN before each write command (NSS must go low and back high to latch)
432 tx_command[0] = E_WREN;
433 mem.spi->nss_sw = true; // SPI library can deal with this
434 ret = !PHAL_SPI_transfer(mem.spi, tx_command, 1, NULL);
435 while (PHAL_SPI_busy(mem.spi))
436 ;
437
438 tx_command[0] = E_WRITE;
439 tx_command[1] = addr >> 8;
440 tx_command[2] = addr & 0xFF;
441
442 // Complete write
443 mem.spi->nss_sw = false; // Take control for sending address
444 PHAL_writeGPIO(mem.spi->nss_gpio_port, mem.spi->nss_gpio_pin, 0); // start transfer
445 ret |= !PHAL_SPI_transfer(mem.spi, tx_command, 3, NULL);
446 while (PHAL_SPI_busy(mem.spi))
447 ;
448 mem.spi->nss_sw = true; // Give back control (DMA interrupt will set NSS back high in the interrupt)
449 ret |= !PHAL_SPI_transfer(mem.spi, page, size, NULL);
450
451 return ret ? -E_SPI : E_SUCCESS;
452}
