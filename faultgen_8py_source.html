11def validate_fault_configs(fault_modules: List[FaultModule]):
12 """Semantic validation for faults."""
13 print("Validating fault configurations...")
14 fault_names = set()
15
16 for module in fault_modules:
17 if len(module.faults) > 64:
18 print_as_error(f"Node '{module.node_name}' has {len(module.faults)} faults, exceeding the 64-bit bitfield limit.")
19 raise ValueError("Too many faults for bitfield sync")
20
21 for fault in module.faults:
22 # Enforce global fault name uniqueness
23 name_upper = fault.name.upper()
24 if name_upper in fault_names:
25 print_as_error(f"Duplicate fault name detected across system: {name_upper}")
26 raise ValueError(f"Global fault name collision: {name_upper}")
27 fault_names.add(name_upper)
28
29 if fault.min_val >= fault.max_val:
30 print_as_error(f"Fault '{fault.name}' in node '{module.node_name}' has invalid limits: min({fault.min_val}) >= max({fault.max_val})")
31 raise ValueError("Invalid fault limits")
32
33 print_as_success("Fault configurations validated")

35def validate_fault_injection(nodes: List[Node], fault_modules: List[FaultModule], bus_configs: Dict) -> str:
36 """
37 Check for potential issues that would prevent successful injection.
38 Runs BEFORE inject_fault_messages to ensure 'clean' state.
39 """
40 print("Pre-flight validation for fault injection...")
41
42 # 1. Check for host bus library
43 fault_bus_name = None
44 for b_name, b_conf in bus_configs.items():
45 if b_conf.get('host_fault_library', False):
46 fault_bus_name = b_name
47 break
48
49 if not fault_bus_name:
50 print_as_error("No bus configured with 'host_fault_library': true. This is required for decentralised faults.")
51 raise ValueError("Missing host_fault_library configuration")
52
53 node_map = {n.name.upper(): n for n in nodes if not n.is_external}
54
55 for module in fault_modules:
56 m_name_upper = module.node_name.upper()
57 if m_name_upper not in node_map:
58 continue
59
60 node = node_map[m_name_upper]
61
62 # 2. Check if node is on the fault bus
63 if fault_bus_name not in node.busses:
64 print_as_error(f"Node '{node.name}' has faults but no {fault_bus_name} bus. This bus is required for fault communication.")
65 raise ValueError(f"Node {node.name} missing fault bus {fault_bus_name}")
66
67 # 3. Check for message shadowing
68 bus = node.busses[fault_bus_name]
69 existing_msg_names = {m.name.lower() for m in bus.tx_messages}
70
71 for suffix in ["_fault_event", "_fault_sync"]:
72 msg_name = f"{node.name.lower()}{suffix}"
73 if msg_name in existing_msg_names:
74 print_as_error(f"Conflict: Node {node.name} already has a message named {msg_name}. Shadowing injected faults is forbidden.")
75 raise ValueError(f"Message name shadowing: {msg_name}")
76
77 return fault_bus_name

79def inject_fault_messages(nodes: List[Node], fault_modules: List[FaultModule], bus_configs: Dict, fault_bus_name: str, custom_types: Dict):
80 """Inject TX and RX messages for faults into Node objects."""
81
82 print("Injecting fault communication messages into pipeline...")
83 node_map = {n.name.upper(): n for n in nodes if not n.is_external}
84
85 all_fault_sync_msgs = []
86 all_fault_event_msgs = []
87
88 # 1. Inject TX Messages
89 is_extended = bus_configs.get(fault_bus_name, {}).get('is_extended_id', False)
90
91 for module in fault_modules:
92 m_name_upper = module.node_name.upper()
93 if m_name_upper not in node_map:
94 continue
95
96 node = node_map[m_name_upper]
97 bus = node.busses[fault_bus_name]
98
99 # Fault Event (Priority 0) - Fired on state change
100 event_name = f"{node.name.lower()}_fault_event"
101 event_msg = Message(
102 name=event_name,
103 desc=f"Immediate fault event signal for {node.name}",
104 priority=0,
105 is_extended=is_extended,
106 signals=[
107 Signal(name="idx", datatype="fault_id_t", desc="Global Fault Index"),
108 Signal(name="val", datatype="uint16_t", desc="Trigger Value"),
109 Signal(name="state", datatype="bool", desc="Latch State (0=unlatched, 1=latched)")
110 ]
111 )
112 event_msg.resolve_layout(custom_types)
113 bus.tx_messages.append(event_msg)
114 all_fault_event_msgs.append(event_name)
115
116 # Fault Sync (Priority 1) - Periodic bitfield
117 sync_name = f"{node.name.lower()}_fault_sync"
118 sync_msg = Message(
119 name=sync_name,
120 desc=f"Periodic fault synchronization for {node.name}",
121 priority=1,
122 is_extended=is_extended,
123 period=100,
124 signals=[
125 Signal(name=f.name, datatype="bool", length=1) for f in module.faults
126 ]
127 )
128 sync_msg.resolve_layout(custom_types)
129 bus.tx_messages.append(sync_msg)
130 all_fault_sync_msgs.append(sync_name)
131
132 # 2. Global Subscription
133 # Every internal node subscribes to every other node's sync and event messages
134 all_msgs = all_fault_event_msgs + all_fault_sync_msgs
135
136 for node in nodes:
137 if node.is_external or fault_bus_name not in node.busses or not node.fault_library_enabled:
138 continue
139
140 bus = node.busses[fault_bus_name]
141 tx_names = {m.name for m in bus.tx_messages}
142 rx_names = {m.name for m in bus.rx_messages}
143
144 for msg_name in all_msgs:
145 if msg_name not in tx_names and msg_name not in rx_names:
146 bus.rx_messages.append(RxMessage(name=msg_name, callback=True))
147
148 print_as_success(f"Injected {len(all_fault_event_msgs)} events and {len(all_fault_sync_msgs)} sync messages on {fault_bus_name}")

150def inject_fault_types(custom_types: Dict, fault_modules: List[FaultModule]):
151 """Inject fault_id_t enum into the common types list."""
152 choices = []
153 global_idx = 0
154 for module in fault_modules:
155 for fault in module.faults:
156 # The prefix will be FAULT_ID_
157 choices.append(f"{module.node_name}_{fault.name}")
158 fault.absolute_index = global_idx
159 global_idx += 1
160
161 # Preserve base_type if it exists in the original definition from common_types.json
162 base_type = custom_types.get("fault_id_t", {}).get("base_type", "uint16_t")
163
164 custom_types["fault_id_t"] = {
165 "name": "fault_id_t",
166 "choices": choices,
167 "base_type": base_type
168 }
169 print_as_ok(f"Injected fault_id_t with {global_idx} total faults")

171def augment_system_with_faults(nodes: List[Node], bus_configs: Dict, custom_types: Dict):
172 """
173 Middleware: Enriches Nodes with fault communication protocols and
174 registers fault types in the global type system. (Change B)
175 """
176 fault_modules = [
177 FaultModule(n.name, n.generate_fault_strings, n.faults)
178 for n in nodes if n.faults and n.fault_library_enabled
179 ]
180
181 if not fault_modules:
182 return
183
184 validate_fault_configs(fault_modules)
185 fault_bus_name = validate_fault_injection(nodes, fault_modules, bus_configs)
186 inject_fault_types(custom_types, fault_modules)
187 inject_fault_messages(nodes, fault_modules, bus_configs, fault_bus_name, custom_types)
