32void update_fault(fault_id_t fault_id, uint16_t value) {
33 if ((fault_id < MY_FAULT_START) || (fault_id > MY_FAULT_END)) {
34 return;
35 }
36
37 fault_t *fault = &faults[fault_id];
38 uint32_t now = OS_TICKS;
39 // min is inclusive, max is exclusive, so the healthy range is [min, max)
40 bool is_out_of_bounds = (value >= fault->max_value) || (value < fault->min_value);
41
42 // Implementation of the FSM diagram in common/can_library/README
43 // I know this FSM is not "mathematically pure", but it must be implemented in this way
44 // to satisfy timing constraints/defensive programming
45 switch (fault->state) {
46 case FAULT_STATE_OK: {
47 if (is_out_of_bounds) {
48 fault->state = FAULT_STATE_PENDING;
49 fault->start_time_ms = now;
50 }
51 break;
52 }
53 case FAULT_STATE_PENDING: {
54 if (!is_out_of_bounds) {
55 fault->state = FAULT_STATE_OK;
56 break;
57 }
58
59 uint32_t elapsed = now - fault->start_time_ms;
60 // do not update the start_time
61 if (elapsed >= fault->latch_time_ms) {
62 fault->state = FAULT_STATE_LATCHED;
63 tx_fault_event(fault_id, value);
64 }
65 break;
66 }
67 case FAULT_STATE_LATCHED: {
68 if (!is_out_of_bounds) {
69 fault->state = FAULT_STATE_RECOVERING;
70 fault->start_time_ms = now;
71 }
72
73 break;
74 }
75 case FAULT_STATE_RECOVERING: {
76 if (is_out_of_bounds) {
77 fault->state = FAULT_STATE_LATCHED;
78 break;
79 }
80
81 uint32_t elapsed = now - fault->start_time_ms;
82 // do not update the start_time
83 if (elapsed >= fault->unlatch_time_ms) {
84 fault->state = FAULT_STATE_OK;
85 }
86 break;
87 }
88 }
89}

91void fault_library_periodic() {
92 // Reset priority counters
93 for (int i = 0; i < NUM_FAULT_PRIOS; i++) {
94 fault_counters[i] = 0;
95 }
96
97 // Count latched faults
98 for (int i = 0; i < TOTAL_NUM_FAULTS; i++) {
99 if (is_latched(i)) {
100 fault_counters[faults[i].priority]++;
101 }
102 }
103
104 // Broadcast our local state to the rest of the car
105 tx_fault_sync();
106}
