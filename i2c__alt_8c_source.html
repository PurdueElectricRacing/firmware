11int PHAL_I2C_init(I2C_TypeDef* instance, i2c_handle_t* handle, uint32_t freq, const uint32_t fck) {
12 uint8_t clk_tim;
13 uint8_t presc = 0;
14
15 // Enable I2C clock
16 if (instance == I2C1) {
17 RCC->APB1ENR1 |= RCC_APB1ENR1_I2C1EN;
18 } else if (instance == I2C3) {
19 RCC->APB1ENR1 |= RCC_APB1ENR1_I2C3EN;
20 } else {
21 return -E_CONFIG;
22 }
23
24 // Disable peripheral
25 instance->CR1 &= ~I2C_CR1_PE;
26
27 // Enable DMA
28 instance->CR1 |= I2C_CR1_TXDMAEN | I2C_CR1_RXDMAEN | I2C_CR1_TCIE | I2C_CR1_TXIE | I2C_CR1_RXIE | I2C_CR1_ERRIE;
29
30 // Set clock timing
31 instance->TIMINGR &= 0x0f000000;
32 clk_tim = fck / (freq * 2) - 1;
33
34 if (clk_tim > 0xff) {
35 presc = 0xff - clk_tim;
36
37 if (presc > 0xff) {
38 return -E_CONFIG;
39 }
40 }
41
42 instance->TIMINGR |= (presc << I2C_TIMINGR_PRESC_Pos) | (clk_tim << I2C_TIMINGR_SCLH_Pos) | (clk_tim << I2C_TIMINGR_SCLL_Pos);
43
44 // Enable peripheral
45 instance->CR1 |= I2C_CR1_PE;
46
47 // Configure DMA
48 if (handle->tx_dma_cfg && !PHAL_initDMA(handle->tx_dma_cfg)) {
49 return false;
50 }
51
52 if (handle->rx_dma_cfg && !PHAL_initDMA(handle->rx_dma_cfg)) {
53 return false;
54 }
55
56 // Set core and unmask interrupts
57 core = handle;
58 i2c_unmask_irq(instance);
59
60 return 0;
61}

63/*
64 What I'm going for here...
65
66 DMA starts on TX/RX. If DMA hasn't finished from the last pass, return busy.
67 Turn on all event interrupts. If there's an error on the bus, stop transfer, store the error code,
68 and try to get everything reset for the next go. I think DMA might be the key here. There's no notion
69 of checking when the address is actually sent like it is on the F4 variants, so letting DMA decide
70 when to pack data might be useful. We can then check in on how far the DMA progress is going with
71 calls to read/write with a return of -E_BUSY, and terminate the read/write if it's taking too long.
72 So the timeout moves to app code and we aren't in danger of hitting a WWDG reset waiting for something
73 to unblock. At least in theory...
74
75 Also, set STOP bit when transfer is complete using DMA interrupt. So a few interrupts going across
76 the board.
77
78 It's late and I want some sleep. If only DoorDash could deliver food on time... :/
79*/

100 PHAL_DMA_setMemAddress(core->tx_dma_cfg, (uint32_t)data);

196static void I2C_IRQHandler(I2C_TypeDef* instance) {
197 // Check if bus is busy but we don't think we should be busy
198 if (instance->ISR & I2C_ISR_BUSY && (!core->_tx_busy && !core->_rx_busy)) {
199 core->error[core->err_idx++] = E_BAD_BUSY;
200 core->err_idx = (core->err_idx == MAX_ERR) ? 0 : core->err_idx;
201 }
202
203 // Check if the bus is not busy and unlock
204 if (!(instance->ISR & I2C_ISR_BUSY)) {
205 core->_tx_busy = core->_rx_busy = false;
206 }
207
208 // Check for arbitration loss
209 if (instance->ISR & I2C_ISR_ARLO) {
210 core->error[core->err_idx++] = E_ARLO;
211 core->err_idx = (core->err_idx == MAX_ERR) ? 0 : core->err_idx;
212 }
213
214 // Check for bad start/stop bits
215 if (instance->ISR & I2C_ISR_BERR) {
216 core->error[core->err_idx++] = E_BERR;
217 core->err_idx = (core->err_idx == MAX_ERR) ? 0 : core->err_idx;
218 }
219
220 // Check for slave device NAK
221 if (instance->ISR & I2C_ISR_NACKF) {
222 core->error[core->err_idx++] = E_NACK;
223 core->err_idx = (core->err_idx == MAX_ERR) ? 0 : core->err_idx;
224 }
225
226 // Send a stop if the last byte was just transmitted
227 if (instance->ISR & I2C_ISR_TC) {
228 if (core->stop) {
229 PHAL_I2C_stop(instance);
230 }
231
232 core->_tx_busy = core->_rx_busy = false;
233 }
234
235 // Clear ISR so we don't trigger again from NVIC
236 instance->ICR = 0b11111100111000;
237}

239static int gen_start(I2C_TypeDef* instance, uint8_t addr, uint8_t len, bool read) {
240 // Clear from last read/write
241 instance->CR2 &= 0x07ffffff;
242
243 // Set NBYTES, read/write, and address, then start
244 instance->CR2 |= (len << I2C_CR2_NBYTES_Pos) | (read << I2C_CR2_RD_WRN_Pos) | (addr << 1);
245 instance->CR2 |= I2C_CR2_START;
246
247 return 0;
248}
