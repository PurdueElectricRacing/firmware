416void CAN1_RX0_IRQHandler() {
417 if (CAN1->RF0R & CAN_RF0R_FOVR0) // FIFO Overrun
418 CAN1->RF0R &= !(CAN_RF0R_FOVR0);
419
420 if (CAN1->RF0R & CAN_RF0R_FULL0) // FIFO Full
421 CAN1->RF0R &= !(CAN_RF0R_FULL0);
422
423 if (CAN1->RF0R & CAN_RF0R_FMP0_Msk) // Release message pending
424 {
425 CanMsgTypeDef_t rx;
426 rx.Bus = CAN1;
427
428 // Get either StdId or ExtId
429 if (CAN_RI0R_IDE & CAN1->sFIFOMailBox[0].RIR) {
430 rx.ExtId = ((CAN_RI0R_EXID | CAN_RI0R_STID) & CAN1->sFIFOMailBox[0].RIR) >> CAN_RI0R_EXID_Pos;
431 } else {
432 rx.StdId = (CAN_RI0R_STID & CAN1->sFIFOMailBox[0].RIR) >> CAN_TI0R_STID_Pos;
433 }
434
435 rx.DLC = (CAN_RDT0R_DLC & CAN1->sFIFOMailBox[0].RDTR) >> CAN_RDT0R_DLC_Pos;
436
437 rx.Data[0] = (uint8_t)(CAN1->sFIFOMailBox[0].RDLR >> 0) & 0xFF;
438 rx.Data[1] = (uint8_t)(CAN1->sFIFOMailBox[0].RDLR >> 8) & 0xFF;
439 rx.Data[2] = (uint8_t)(CAN1->sFIFOMailBox[0].RDLR >> 16) & 0xFF;
440 rx.Data[3] = (uint8_t)(CAN1->sFIFOMailBox[0].RDLR >> 24) & 0xFF;
441 rx.Data[4] = (uint8_t)(CAN1->sFIFOMailBox[0].RDHR >> 0) & 0xFF;
442 rx.Data[5] = (uint8_t)(CAN1->sFIFOMailBox[0].RDHR >> 8) & 0xFF;
443 rx.Data[6] = (uint8_t)(CAN1->sFIFOMailBox[0].RDHR >> 16) & 0xFF;
444 rx.Data[7] = (uint8_t)(CAN1->sFIFOMailBox[0].RDHR >> 24) & 0xFF;
445
446 CAN1->RF0R |= (CAN_RF0R_RFOM0);
447
448 qSendToBack(&q_rx_can, &rx); // Add to queue (qSendToBack is interrupt safe)
449 }
450}
