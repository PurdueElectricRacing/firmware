170void updatePage() {
171 // If we do not detect a page update, do nothing
172 if (curr_page == prev_page) {
173 return;
174 }
175
176 // Set the page on display
177 switch (curr_page) {
178 case PAGE_RACE:
179 NXT_setPage(RACE_STRING);
180 break;
181 case PAGE_FAULTS:
182 NXT_setPage(FAULT_STRING);
183 break;
184 case PAGE_CALIBRATION:
185 NXT_setPage(CALIBRATION_STRING);
186 break;
187 default:
188 curr_page = PAGE_RACE; // something probably went wrong
189 break;
190 }
191
192 prev_page = curr_page;
193
194 // Bounds Check
195 if (curr_page >= NUM_PAGES) {
196 return;
197 }
198
199 // Call update handler if available
200 if (page_handlers[curr_page].update != NULL) {
201 page_handlers[curr_page].update();
202 }
203}
