415void updatePage() {
416 // Only update if we are on a "selectable" page
417 bool is_error_page = (curr_page == PAGE_ERROR) || (curr_page == PAGE_WARNING) || (curr_page == PAGE_FATAL);
418
419 // Only update prev_page for non-error pages
420 if (!is_error_page) {
421 fault_time_displayed = 0;
422 }
423
424 // If we do not detect a page update, do nothing
425 if (curr_page == prev_page) {
426 return;
427 }
428
429 if (!is_error_page) { // Must come after redundant page check
430 prev_page = curr_page;
431 }
432
433 // Set the page on display
434 switch (curr_page) {
435 case PAGE_PREFLIGHT:
436 NXT_setPage(PREFLIGHT_STRING);
437 break;
438 case PAGE_RACE:
439 NXT_setPage(RACE_STRING);
440 break;
441 case PAGE_COOLING:
442 NXT_setPage(COOLING_STRING);
443 break;
444 case PAGE_TVSETTINGS:
445 NXT_setPage(TVSETTINGS_STRING);
446 break;
447 case PAGE_FAULTS:
448 NXT_setPage(FAULT_STRING);
449 break;
450 case PAGE_SDCINFO:
451 NXT_setPage(SDCINFO_STRING);
452 break;
453 case PAGE_DRIVER:
454 NXT_setPage(DRIVER_STRING);
455 break;
456 case PAGE_PROFILES:
457 NXT_setPage(DRIVER_CONFIG_STRING);
458 break;
459 case PAGE_LOGGING:
460 NXT_setPage(LOGGING_STRING);
461 break;
462 case PAGE_CALIBRATION:
463 NXT_setPage(APPS_STRING);
464 break;
465 case PAGE_WARNING:
466 NXT_setPage(WARN_STRING);
467 NXT_setText(ERR_TXT, errorText);
468 return;
469 case PAGE_ERROR:
470 NXT_setPage(ERR_STRING);
471 NXT_setText(ERR_TXT, errorText);
472 return;
473 case PAGE_FATAL:
474 NXT_setPage(FATAL_STRING);
475 NXT_setText(ERR_TXT, errorText);
476 return;
477 default:
478 curr_page = PAGE_PREFLIGHT; // something probably went wrong
479 break;
480 }
481
482 // Bounds Check
483 if (curr_page > PAGE_COUNT && curr_page < 0) {
484 return;
485 }
486
487 // Call update handler if available
488 if (page_handlers[curr_page].update != NULL) {
489 page_handlers[curr_page].update();
490 }
491
492 lcd_tx_cmd();
493}
