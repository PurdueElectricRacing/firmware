95void updatePage() {
96 // If we do not detect a page update, do nothing
97 if (curr_page == prev_page) {
98 return;
99 }
100
101 // Set the page on display
102 NXT_setPage(page_handlers[curr_page].string);
103 prev_page = curr_page;
104
105 // Call update handler if available
106 if (page_handlers[curr_page].update != nullptr) {
107 page_handlers[curr_page].update();
108 }
109}
