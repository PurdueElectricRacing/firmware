144void updatePage() {
145 // If we do not detect a page update, do nothing
146 if (curr_page == prev_page) {
147 return;
148 }
149
150 // Set the page on display
151 switch (curr_page) {
152 case PAGE_RACE:
153 NXT_setPage(RACE_STRING);
154 break;
155 break;
156 case PAGE_FAULTS:
157 NXT_setPage(FAULT_STRING);
158 break;
159 case PAGE_SDC_INFO:
160 NXT_setPage(SDCINFO_STRING);
161 break;
162 break;
163 case PAGE_CALIBRATION:
164 NXT_setPage(APPS_STRING);
165 break;
166 default:
167 curr_page = PAGE_RACE; // something probably went wrong
168 break;
169 }
170
171 // Bounds Check
172 if (curr_page > NUM_PAGES && curr_page < 0) {
173 return;
174 }
175
176 // Call update handler if available
177 if (page_handlers[curr_page].update != NULL) {
178 page_handlers[curr_page].update();
179 }
180}
