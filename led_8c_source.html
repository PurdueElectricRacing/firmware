20bool LED_control(int led, enum LED_state state) {
21 uint16_t LED_control_data_new = 0x0000;
22
23 /* Get data offset to account for Endianness:
24 *
25 * SPI transfers the LSB, followed by the MSB.
26 * However, the LED driver wants the MSB transferred first.
27 *
28 * Therefore, write data for the first 8 LEDs to the second
29 * byte and data for the next 8 LEDs to the first byte
30 */
31
32 int offset;
33 if (led < 8) {
34 offset = 8;
35 } else {
36 offset = -8;
37 }
38
39 // Create 16-bit LED control data to write
40 switch (state) {
41 case LED_OFF: // Clear bit
42 LED_control_data_new = LED_control_data &= ~(1 << (led + offset));
43 LED_toggle &= ~(1 << led);
44 break;
45 case LED_ON: // Set bit
46 LED_control_data_new = LED_control_data |= (1 << (led + offset));
47 LED_toggle &= ~(1 << led);
48 break;
49 case LED_BLINK: // Toggle bit
50 LED_control_data_new = LED_control_data ^= (1 << (led + offset));
51 LED_toggle |= (1 << led);
52 break;
53 }
54
55 // Update persisting control data for next time an LED is controlled
56 LED_control_data = LED_control_data_new;
57
58 while (PHAL_SPI_busy(&spi_config))
59 ;
60
61 // Transfer LED control data to the ON/OFF Control Shift Register
62 PHAL_SPI_transfer(&spi_config, (uint8_t*)&LED_control_data_new, 2, NULL);
63
64 while (PHAL_SPI_busy(&spi_config))
65 ;
66
67 for (int i = 0; i < 80; i++) {
68 __asm__("nop");
69 }
70
71 // Latch LED control data into the ON/OFF Control Data Latch at rising edge of LAT
72 PHAL_writeGPIO(LED_CTRL_LAT_GPIO_Port, LED_CTRL_LAT_Pin, 1);
73
74 for (int i = 0; i < 80; i++) {
75 __asm__("nop");
76 }
77
78 PHAL_writeGPIO(LED_CTRL_LAT_GPIO_Port, LED_CTRL_LAT_Pin, 0);
79
80 // Set Blank low so corresponding output is turned on if data in the ON/OFF control data latch are '1'
81 PHAL_writeGPIO(LED_CTRL_BLANK_GPIO_Port, LED_CTRL_BLANK_Pin, 0);
82
83 return true;
84}
