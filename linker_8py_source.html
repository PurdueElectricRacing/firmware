29 def link(self):
30 """Execute the two-pass linking algorithm"""
31 self._validate_overrides() # Pass 1
32
33 # Pass 2: Assign IDs priority by priority
34 # We must process priorities in order (0, 1, 2...)
35 sorted_priorities = sorted(self.msgs_by_prio.keys())
36 for prio in sorted_priorities:

39 def _validate_overrides(self):
40 """
41 Pass 1: Validate that overrides respect priority monotonicity.
42 An override in Priority 0 (e.g. 0x10) must be < an override in Priority 1 (e.g. 0x20).
43 """
44 sorted_priorities = sorted(self.overrides_by_prio.keys())
45 max_override_seen = -1
46
47 for prio in sorted_priorities:
48 overrides = self.overrides_by_prio[prio]
49 if not overrides:
50 continue
51
52 current_min = min(overrides)
53 current_max = max(overrides)
54
55 # Check if this priority group starts before the previous one ended
56 if current_min <= max_override_seen:
57 print_as_error(f"Bus '{self.bus_name}': Priority {prio} override {hex(current_min)} violates priority order (previous max was {hex(max_override_seen)})")
58 raise ValueError("Priority Inversion in Overrides")
59
60 max_override_seen = max(max_override_seen, current_max)
61
62 def _link_priority_group(self, priority: int):
63 """
64 Pass 2: Assign IDs to dynamic messages in this priority group.
65 Respects the 'Water Level' (next_avail_id) and skips reserved overrides.
66 """
67 msgs = self.msgs_by_prio[priority]
68 # Get overrides for THIS priority group only
69 group_overrides = self.overrides_by_prio[priority]
70
71 # 1. Crash Check: Did previous groups push us past our own overrides?
72 if group_overrides:
73 min_override = min(group_overrides)
74 if self.next_avail_id > min_override:
75 print_as_error(f"Bus '{self.bus_name}': Dynamic IDs from higher priorities pushed water level to {hex(self.next_avail_id)}, colliding with Priority {priority} override at {hex(min_override)}")
76 raise ValueError("Dynamic ID Overflow")
77
78 # 2. Assign IDs
79 # To ensure absolute determinism across systems, we sort messages
80 # by name within the priority group before assigning IDs.
81 # This keeps IDs stable even if node discovery order or JSON insertion order changes.
82 sorted_msgs = sorted(msgs, key=lambda x: x.name)
83
84 for msg in sorted_msgs:
85 if msg.id_override:
86 msg.final_id = int(msg.id_override, 0)
87 else:
88 # Find next free slot
89 # Skip IDs used by overrides in ANY priority group
90 while self.next_avail_id in self.all_used_idsall_used_ids:
91 self.next_avail_id += 1
92
93 msg.final_id = self.next_avail_id
94 self.all_used_idsall_used_ids.add(msg.final_id)
95 self.next_avail_id += 1
96
97 # ID Range Validation
98 limit = 0x1FFFFFFF if msg.is_extended else 0x7FF
99 if msg.final_id > limit:
100 type_str = "Extended" if msg.is_extended else "Standard"
101 raise ValueError(f"Bus '{self.bus_name}': Message '{msg.name}' ID {hex(msg.final_id)} exceeds {type_str} CAN limit ({hex(limit)})")
102
103 # 3. Update Water Level
104 # Ensure next priority group starts AFTER the highest ID used in this group
105 if group_overrides:
106 max_override = max(group_overrides)
107 self.next_avail_id = max(self.next_avail_id, max_override + 1)

109def link_all(nodes: List[Node]) -> List[Node]:
110 """
111 Main entry point for the Linker stage.
112 Resolves CAN IDs for all messages across all buses.
113 """
114 print("Linking CAN IDs...")
115
116 # 1. Group messages by Bus
117 linkers: Dict[str, BusLinker] = {}
118
119 for node in nodes:
120 for bus_name, bus in node.busses.items():
121 if bus_name not in linkers:
122 linkers[bus_name] = BusLinker(bus_name)
123
124 for msg in bus.tx_messages:
125 linkers[bus_name].add_message(msg)
126
127 # 2. Execute Linker for each bus
128 for bus_name, linker in linkers.items():
129 try:
130 linker.link()
131 print_as_ok(f"Bus '{bus_name}' linked successfully")
132 except ValueError:
133 print_as_error(f"Failed to link bus '{bus_name}'")
134 raise
135
136 # 3. Resolve RX Messages
137 # Build global message map
138 all_tx_messages: Dict[str, Message] = {}
139 for node in nodes:
140 for bus in node.busses.values():
141 for msg in bus.tx_messages:
142 if msg.name in all_tx_messages:
143 # This should be caught by validator, but good to check
144 print_as_error(f"Duplicate message name '{msg.name}' found during linking")
145 all_tx_messages[msg.name] = msg
146
147 for node in nodes:
148 for bus_name, bus in node.busses.items():
149 for rx_msg in bus.rx_messages:
150 if rx_msg.name in all_tx_messages:
151 rx_msg.resolved_message = all_tx_messages[rx_msg.name]
152 else:
153 print_as_error(f"Node '{node.name}': RX message '{rx_msg.name}' not found in any bus")
154 raise ValueError(f"Unresolved RX message: {rx_msg.name}")
155
156 # 4. Final Validation
157 for node in nodes:
158 for bus in node.busses.values():
159 for msg in bus.tx_messages:
160 if msg.final_id > 0x7FF and not msg.is_extended:
161 print_as_error(f"Message '{msg.name}' has ID {hex(msg.final_id)} but is not marked as extended.")
162 raise ValueError(f"Standard CAN ID exceeds 0x7FF: {hex(msg.final_id)}")
163
164 print_as_success("Successfully linked all CAN IDs")
165 return nodes
