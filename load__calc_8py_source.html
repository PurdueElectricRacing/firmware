10# CAN 2.0 Base Overhead (including 3-bit Inter-Frame Space)
11# Standard: SOF(1), ID(11), RTR(1), IDE(1), r0(1), DLC(4), CRC(15), ACK(2), EOF(7), IFS(3) = 47
12# Extended: SOF(1), ID(11), SRR(1), IDE(1), ID(18), RTR(1), r1(1), r0(1), DLC(4), CRC(15), ACK(2), EOF(7), IFS(3) = 67
13STANDARD_OVERHEAD = 47
14EXTENDED_OVERHEAD = 67
15
16# Typical bit-stuffing factor (varies by data, but 1.2 is a safe average)
17STUFFING_FACTOR = 1.2
18
19def calculate_bus_load(context: SystemContext):
20 """
21 Analyzes all busses in the system context and prints estimated bus load.
22 Load calculation: (Total Bits Per Second / Baud Rate) * 100
23 """
24 print("Bus Load Analysis:")
25
26 for bus_name, bus_view in sorted(context.busses.items()):
27 total_bits_per_sec = 0
28
29 bus_config = context.bus_configs.get(bus_name, {})
30 baud_rate = bus_config.get("baud_rate")
31
32 for msg in bus_view.messages:
33 # 1. Calculate frames per second
34 # Treat asynchronous messages (period=0) as 1Hz for estimation
35 fps = 1000.0 / msg.period if msg.period > 0 else 1.0
36
37 # 2. Bits per frame (Overhead + Data)
38 overhead = EXTENDED_OVERHEAD if msg.is_extended else STANDARD_OVERHEAD
39 dlc = msg.get_dlc(context.custom_types)
40 bits_per_frame = overhead + (dlc * 8)
41
42 # 3. Apply bit-stuffing factor
43 stuffed_bits = bits_per_frame * STUFFING_FACTOR
44
45 total_bits_per_sec += (stuffed_bits * fps)
46
47 # Calculate final percentage
48 load_percent = (total_bits_per_sec / baud_rate) * 100
49
50 # Determine color formatting
51 color = bcolors.GREEN
52 if load_percent >= 70:
53 color = bcolors.RED
54 elif load_percent >= 50:
55 color = bcolors.ORANGE
56
57 if color != bcolors.GREEN:
58 print_as_warning(f"{bus_name}: {color}{load_percent:.2f}%{bcolors.ENDC}")
59 else:
60 print_as_ok(f"{bus_name}: {color}{load_percent:.2f}%{bcolors.ENDC}")
