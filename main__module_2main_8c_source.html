83void heartbeat_task() {
84 // preflight animation for the first 1.5 seconds after boot
85 if (OS_TICKS <= PREFLIGHT_DURATION_MS) {
86 static uint32_t sweep_index = 0;
87
88 // Creates a sweeping pattern
89 switch (sweep_index++ % 3) {
90 case 0:
91 PHAL_writeGPIO(HEARTBEAT_LED_PORT, HEARTBEAT_LED_PIN, 1);
92 PHAL_writeGPIO(CONNECTION_LED_PORT, CONNECTION_LED_PIN, 0);
93 PHAL_writeGPIO(ERROR_LED_PORT, ERROR_LED_PIN, 0);
94 break;
95 case 1:
96 PHAL_writeGPIO(HEARTBEAT_LED_PORT, HEARTBEAT_LED_PIN, 0);
97 PHAL_writeGPIO(CONNECTION_LED_PORT, CONNECTION_LED_PIN, 1);
98 PHAL_writeGPIO(ERROR_LED_PORT, ERROR_LED_PIN, 0);
99 break;
100 case 2:
101 PHAL_writeGPIO(HEARTBEAT_LED_PORT, HEARTBEAT_LED_PIN, 0);
102 PHAL_writeGPIO(CONNECTION_LED_PORT, CONNECTION_LED_PIN, 0);
103 PHAL_writeGPIO(ERROR_LED_PORT, ERROR_LED_PIN, 1);
104 break;
105 }
106
107 return;
108 }
109
110 PHAL_toggleGPIO(HEARTBEAT_LED_PORT, HEARTBEAT_LED_PIN);
111
112 if (OS_TICKS - last_can_rx_time_ms >= CONN_LED_TIMEOUT_MS) {
113 PHAL_writeGPIO(CONNECTION_LED_PORT, CONNECTION_LED_PIN, 1);
114 } else {
115 PHAL_writeGPIO(CONNECTION_LED_PORT, CONNECTION_LED_PIN, 0);
116 }
117}

138int main(void) {
139 // Hardware Initialization
140 if (0 != PHAL_configureClockRates(&clock_config)) {
141 HardFault_Handler();
142 }
143 if (false == PHAL_initGPIO(gpio_config, sizeof(gpio_config) / sizeof(GPIOInitConfig_t))) {
144 HardFault_Handler();

146 if (false == PHAL_FDCAN_init(FDCAN2, false, VCAN_BAUD_RATE)) {
147 HardFault_Handler();
148 }
149 if (false == PHAL_FDCAN_init(FDCAN3, false, MCAN_BAUD_RATE)) {
150 HardFault_Handler();
151 }
152
153 NVIC_EnableIRQ(FDCAN2_IT0_IRQn);
154 NVIC_EnableIRQ(FDCAN3_IT0_IRQn);
155 CAN_library_init();
156
157 car_init();
158
159 // Software Initialization
160 osKernelInitialize();
161
162 createThread(heartbeat_task);
163 createThread(background_can_update);
164 createThread(update_SDC);
165 createThread(fsm_periodic);
166 createThread(AMK_task);
167 createThread(fault_library_periodic);
168
169 // no way home
170 osKernelStart();
171
172 return 0;
173}
