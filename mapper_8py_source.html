46def map_hardware(nodes: List[Node], bus_configs: Dict) -> Dict[str, NodeMapping]:
47 """
48 Hardware Mapper stage.
49 Assigns physical resources (like bxCAN filter banks or FDCAN filter lists) to nodes.
50 """
51 mappings = {}
52 for node in nodes:
53 if node.is_external:
54 continue
55 mappings[node.name] = map_node_hardware(node, bus_configs)
56 return mappings

64def map_node_hardware(node: Node, bus_configs: Dict) -> NodeMapping:
65 mapping = NodeMapping(node_name=node.name)
66
67 peripherals = sorted(list(set(bus.peripheral for bus in node.busses.values())))
68
69 # Group RX messages by peripheral
70 periph_to_msgs = {p: [] for p in peripherals}
71 for bus_name, bus in node.busses.items():
72 for rx_msg in bus.rx_messages:
73 if rx_msg.resolved_message:
74 periph_to_msgs[bus.peripheral].append(
75 (rx_msg.resolved_message, bus_name)
76 )
77
78 for periph in peripherals:
79 msgs = periph_to_msgs[periph]
80 mapping.accept_all[periph] = any(
81 bus.accept_all_messages
82 for bus in node.busses.values()
83 if bus.peripheral == periph
84 )
85
86 if is_fdcan_peripheral(periph):
87 # FDCAN filter mapping (G4)
88 mapping.fdcan_filters[periph] = map_fdcan_filters(node.name, periph, msgs)
89 mapping.filters[periph] = [] # Empty for FDCAN
90 else:
91 # bxCAN filter bank mapping (F4/F7/L4)
92 mapping.filters[periph] = map_bxcan_filters(node.name, periph, msgs)
93 mapping.fdcan_filters[periph] = FDCANFilters() # Empty for bxCAN
94
95 return mapping

98def map_fdcan_filters(node_name: str, periph: str, msgs: List) -> FDCANFilters:
99 """Map messages to FDCAN standard and extended ID filter lists"""
100 filters = FDCANFilters()
101
102 if not msgs:
103 return filters
104
105 for msg, bus_name in msgs:
106 if msg.is_extended:
107 filters.ext_ids.append(msg)
108 else:
109 filters.std_ids.append(msg)
110
111 # Check filter limits
112 if len(filters.std_ids) > MAX_FDCAN_SID_FILTERS:
113 raise ValueError(
114 f"Node '{node_name}' exceeds FDCAN standard ID filter limit for {periph} "
115 f"({len(filters.std_ids)} > {MAX_FDCAN_SID_FILTERS})"
116 )
117 if len(filters.ext_ids) > MAX_FDCAN_XID_FILTERS:
118 raise ValueError(
119 f"Node '{node_name}' exceeds FDCAN extended ID filter limit for {periph} "
120 f"({len(filters.ext_ids)} > {MAX_FDCAN_XID_FILTERS})"
121 )
122
123 return filters

126def map_bxcan_filters(node_name: str, periph: str, msgs: List) -> List[FilterBank]:
127 """Map messages to bxCAN filter banks"""
128 banks = []
129
130 if not msgs:
131 return banks
132
133 # bxCAN filter bank assignment
134 # CAN1: 0-13, CAN2: 14-27
135 bank_offset = 0 if periph == "CAN1" else 14
136 max_bank = 13 if periph == "CAN1" else 27
137
138 for i in range(0, len(msgs), 2):
139 bank_idx = bank_offset + (i // 2)
140
141 if bank_idx > max_bank:
142 raise ValueError(
143 f"Node '{node_name}' exceeds available bxCAN filter banks for {periph} (limit 14 filters)."
144 )
145
146 msg1, bus1 = msgs[i]
147 msg2, bus2 = msgs[i + 1] if i + 1 < len(msgs) else (None, None)
148
149 is_ext1 = msg1.is_extended
150 is_ext2 = False
151 if msg2:
152 is_ext2 = msg2.is_extended
153
154 fb = FilterBank(
155 bank_idx=bank_idx, msg1=msg1, msg2=msg2, is_ext1=is_ext1, is_ext2=is_ext2
156 )
157 banks.append(fb)
158
159 return banks
