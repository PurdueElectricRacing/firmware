91void MS_select(menu_page_t* page) {
92 menu_element_t* current = &page->elements[page->current_index];
93
94 if (page->is_element_selected) {
95 // Deselect element
96 page->is_element_selected = false;
97 MS_setStyleHover(current);
98
99 // Call onChange if defined
100 if (current->on_change != NULL) {
101 current->on_change();
102 }
103 return;
104 }
105
106 // Select element if it's a selectable type
107 switch (current->type) {
108 case ELEMENT_BUTTON:
109 if (current->on_change != NULL) {
110 current->on_change();
111 }
112 break;
113 case ELEMENT_LIST:
114 // Clear other options
115 for (uint8_t i = 0; i < page->num_elements; i++) {
116 if (page->elements[i].type == ELEMENT_LIST) {
117 page->elements[i].current_value = 0;
118 MS_setStyleNormal(&page->elements[i]);
119 }
120 }
121 current->current_value = 1;
122 if (current->on_change != NULL) {
123 current->on_change();
124 }
125 MS_setStyleSelected(current);
126 break;
127 case ELEMENT_OPTION:
128 current->current_value ^= 1;
129 NXT_setValue(current->object_name, current->current_value);
130 if (current->on_change != NULL) {
131 current->on_change();
132 }
133 break;
134 case ELEMENT_FLT: // Fall through
135 case ELEMENT_VAL:
136 page->is_element_selected = true;
137 MS_setStyleSelected(current);
138 break;
139 default:
140 break;
141 }
