97void MS_select(menu_page_t* page) {
98 menu_element_t* current = &page->elements[page->current_index];
99
100 if (page->is_element_selected) {
101 // Deselect element
102 page->is_element_selected = false;
103 MS_setStyleHover(current);
104
105 // Call onChange if defined
106 if (current->on_change != NULL) {
107 current->on_change();
108 }
109 return;
110 }
111
112 // Select element if it's a selectable type
113 switch (current->type) {
114 case ELEMENT_BUTTON:
115 if (current->on_change != NULL) {
116 current->on_change();
117 }
118 break;
119 case ELEMENT_LIST:
120 // Clear other options
121 for (uint8_t i = 0; i < page->num_elements; i++) {
122 if (page->elements[i].type == ELEMENT_LIST) {
123 page->elements[i].current_value = 0;
124 MS_setStyleNormal(&page->elements[i]);
125 }
126 }
127 current->current_value = 1;
128 if (current->on_change != NULL) {
129 current->on_change();
130 }
131 MS_setStyleSelected(current);
132 break;
133 case ELEMENT_OPTION:
134 current->current_value ^= 1;
135 NXT_setValue(current->object_name, current->current_value);
136 if (current->on_change != NULL) {
137 current->on_change();
138 }
139 break;
140 case ELEMENT_FLT: // Fall through
141 case ELEMENT_VAL:
142 page->is_element_selected = true;
143 MS_setStyleSelected(current);
144 break;
145 default:
146 break;
147 }
148}

209void MS_refreshPage(menu_page_t* page) {
210 page->is_element_selected = false;
211 page->current_index = 0;
212
213 for (uint8_t i = 0; i < page->num_elements; i++) {
214 menu_element_t* curr_element = &page->elements[i];
215 MS_setStyleNormal(curr_element);
216 switch (curr_element->type) {
217 case ELEMENT_LIST:
218 if (curr_element->current_value) {
219 MS_setStyleSelected(curr_element);
220 }
221 break;
222 case ELEMENT_VAL:
223 NXT_setTextFormatted(curr_element->object_name, "%d", curr_element->current_value);
224 break;
225 case ELEMENT_FLT: // Fall through
226 case ELEMENT_OPTION:
227 NXT_setValue(curr_element->object_name, curr_element->current_value);
228 break;
229 default:
230 break;
231 }
232 }
233
234 // Highlight the first element
235 MS_setStyleHover(&page->elements[0]);
236}
