103void MS_select(menu_page_t* page) {
104 menu_element_t* current = &page->elements[page->current_index];
105
106 if (page->is_element_selected) {
107 // Deselect element
108 page->is_element_selected = false;
109 MS_setStyleHover(current);
110
111 // Call onChange if defined
112 if (current->on_change != NULL) {
113 current->on_change();
114 }
115 return;
116 }
117
118 // Select element if it's a selectable type
119 switch (current->type) {
120 case ELEMENT_BUTTON:
121 if (current->on_change != NULL) {
122 current->on_change();
123 }
124 break;
125 case ELEMENT_LIST:
126 // Clear other options
127 for (uint8_t i = 0; i < page->num_elements; i++) {
128 if (page->elements[i].type == ELEMENT_LIST) {
129 page->elements[i].current_value = 0;
130 MS_setStyleNormal(&page->elements[i]);
131 }
132 }
133 current->current_value = 1;
134 if (current->on_change != NULL) {
135 current->on_change();
136 }
137 MS_setStyleSelected(current);
138 break;
139 case ELEMENT_OPTION:
140 current->current_value ^= 1;
141 NXT_setValue(current->object_name, current->current_value);
142 if (current->on_change != NULL) {
143 current->on_change();
144 }
145 break;
146 case ELEMENT_FLT: // Fall through
147 case ELEMENT_VAL:
148 page->is_element_selected = true;
149 MS_setStyleSelected(current);
150 break;
151 default:
152 break;
153 }
154}

215void MS_refreshPage(menu_page_t* page) {
216 page->is_element_selected = false;
217 page->current_index = 0;
218
219 for (uint8_t i = 0; i < page->num_elements; i++) {
220 menu_element_t* curr_element = &page->elements[i];

223 if (curr_element->current_value) {
224 MS_setStyleSelected(curr_element);
225 }
226 break;

229 break;
230 case ELEMENT_FLT: // Fall through
231 case ELEMENT_OPTION:
232 NXT_setValue(curr_element->object_name, curr_element->current_value);
233 break;
