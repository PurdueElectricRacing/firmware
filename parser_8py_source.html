78 def resolve_layout(self, custom_types: Dict) -> None:
79 """
80 Calculate bit offsets, shifts, and masks for all signals.
81 This is intrinsic to the message definition.
82 """
83 current_offset = 0
84 for sig in self.signals:
85 length = sig.get_bit_length(custom_types)
86 sig.length = length
87 sig.bit_offset = current_offset
88 sig.bit_shift = current_offset
89 sig.mask = (1 << length) - 1
90
91 # Resolve signedness
92 base_type = sig.datatype
93 if custom_types and sig.datatype in custom_types:
94 base_type = custom_types[sig.datatype]['base_type']
95 sig.is_signed = base_type.startswith('int')
96
97 current_offset += length
98
99 self.dlc = (current_offset + 7) // 8
100 self.layout_hash = get_layout_hash(self)

102 def validate_semantics(self, custom_types: Dict) -> None:
103 """
104 Perform semantic checks that require external context (like custom types).
105 Raises ValueError if invalid.
106 """
107 total_length = sum(sig.get_bit_length(custom_types) for sig in self.signals)
108
109 for sig in self.signals:
110 if sig.datatype not in CTYPE_SIZES and sig.datatype not in custom_types:
111 print_as_error(f"Signal '{sig.name}' in message '{self.name}' has unknown type '{sig.datatype}'")
112 raise ValueError("Unknown signal type")
113
114 if total_length > 64:
115 print_as_error(f"Message '{self.name}' exceeds 64 bits (has {total_length})")
116 raise ValueError("Message too long")
117
118 # Validate ID Override Range
119 if self.id_override:
120 try:
121 raw_id = int(self.id_override, 0)
122 if not self.is_extended and raw_id > 0x7FF:
123 print_as_error(f"Message '{self.name}' has override ID {hex(raw_id)} which exceeds 11-bit limit for standard bus.")
124 raise ValueError(f"ID override too large for standard bus: {self.name}")
125 if raw_id > 0x1FFFFFFF:
126 print_as_error(f"Message '{self.name}' has override ID {hex(raw_id)} which exceeds 29-bit CAN limit.")
127 raise ValueError(f"ID override exceeds CAN protocol limits: {self.name}")
128 except ValueError as e:
129 if "invalid literal" in str(e):
130 print_as_error(f"Message '{self.name}' has invalid ID override format: {self.id_override}")
131 raise ValueError("Invalid ID override format")
132 raise

226def create_system_context(nodes, mappings, bus_configs, custom_types) -> SystemContext:
227 """Aggregates all system data into a single context object and performs final validation."""
228
229 # Derivation: Populate fault modules from node objects (A)
230 fault_modules = []
231 for node in nodes:
232 if node.faults and node.fault_library_enabled:
233 fault_modules.append(FaultModule(
234 node_name=node.name,
235 generate_strings=node.generate_fault_strings,
236 faults=node.faults
237 ))
238
239 context = SystemContext(
240 nodes=nodes,
241 fault_modules=fault_modules,
242 mappings=mappings,
243 bus_configs=bus_configs,
244 custom_types=custom_types,
245 version=get_git_hash()
246 )
247
248 # Identify all busses across all nodes
249 all_bus_names = set()
250 for node in nodes:
251 all_bus_names.update(node.busses)
252
253 for bus_name in sorted(all_bus_names):
254 view = BusView(name=bus_name)
255
256 for node in nodes:
257 if bus_name in node.busses:
258 view.nodes.add(node.name)
259 for msg in node.busses[bus_name].tx_messages:
260 view.messages.append(msg)
261 view.sender_map[msg.name] = node.name
262
263 # Deterministic sorting for downstream generators
264 view.messages.sort(key=lambda x: (x.final_id or 0, x.name))
265 context.busses[bus_name] = view
266
267 return context
