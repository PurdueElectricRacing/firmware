72 def resolve_layout(self, custom_types: Dict) -> None:
73 """
74 Calculate bit offsets, shifts, and masks for all signals.
75 This is intrinsic to the message definition.
76 """
77 current_offset = 0
78 for sig in self.signals:
79 length = sig.get_bit_length(custom_types)
80 sig.length = length
81 sig.bit_offset = current_offset
82 sig.bit_shift = current_offset
83 sig.mask = (1 << length) - 1
84
85 # Resolve signedness
86 base_type = sig.datatype
87 if custom_types and sig.datatype in custom_types:
88 base_type = custom_types[sig.datatype]['base_type']
89 sig.is_signed = base_type.startswith('int')
90
91 current_offset += length
92
93 self.dlc = (current_offset + 7) // 8

95 def validate_semantics(self, custom_types: Dict) -> None:
96 """
97 Perform semantic checks that require external context (like custom types).
98 Raises ValueError if invalid.
99 """
100 total_length = sum(sig.get_bit_length(custom_types) for sig in self.signals)
101
102 for sig in self.signals:
103 if sig.datatype not in CTYPE_SIZES and sig.datatype not in custom_types:
104 print_as_error(f"Signal '{sig.name}' in message '{self.name}' has unknown type '{sig.datatype}'")
105 raise ValueError("Unknown signal type")
106
107 if total_length > 64:
108 print_as_error(f"Message '{self.name}' exceeds 64 bits (has {total_length})")
109 raise ValueError("Message too long")
110
111 # Validate ID Override Range
112 if self.id_override:
113 try:
114 raw_id = int(self.id_override, 0)
115 if not self.is_extended and raw_id > 0x7FF:
116 print_as_error(f"Message '{self.name}' has override ID {hex(raw_id)} which exceeds 11-bit limit for standard bus.")
117 raise ValueError(f"ID override too large for standard bus: {self.name}")
118 if raw_id > 0x1FFFFFFF:
119 print_as_error(f"Message '{self.name}' has override ID {hex(raw_id)} which exceeds 29-bit CAN limit.")
120 raise ValueError(f"ID override exceeds CAN protocol limits: {self.name}")
121 except ValueError as e:
122 if "invalid literal" in str(e):
123 print_as_error(f"Message '{self.name}' has invalid ID override format: {self.id_override}")
124 raise ValueError("Invalid ID override format")
125 raise

219def create_system_context(nodes, mappings, bus_configs, custom_types) -> SystemContext:
220 """Aggregates all system data into a single context object and performs final validation."""
221
222 # Derivation: Populate fault modules from node objects (A)
223 fault_modules = []
224 for node in nodes:
225 if node.faults and node.fault_library_enabled:
226 fault_modules.append(FaultModule(
227 node_name=node.name,
228 generate_strings=node.generate_fault_strings,
229 faults=node.faults
230 ))
231
232 context = SystemContext(
233 nodes=nodes,
234 fault_modules=fault_modules,
235 mappings=mappings,
236 bus_configs=bus_configs,
237 custom_types=custom_types,
238 version=get_git_hash()
239 )
240
241 # Identify all busses across all nodes
242 all_bus_names = set()
243 for node in nodes:
244 all_bus_names.update(node.busses)
245
246 for bus_name in sorted(all_bus_names):
247 view = BusView(name=bus_name)
248
249 for node in nodes:
250 if bus_name in node.busses:
251 view.nodes.add(node.name)
252 for msg in node.busses[bus_name].tx_messages:
253 view.messages.append(msg)
254 view.sender_map[msg.name] = node.name
255
256 # Deterministic sorting for downstream generators
257 view.messages.sort(key=lambda x: (x.final_id or 0, x.name))
258 context.busses[bus_name] = view
259
260 return context
