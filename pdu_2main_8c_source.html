239void heartbeat_task() {
240 // Preflight animation for the first PREFLIGHT_DURATION_MS after boot
241 if (OS_TICKS <= PREFLIGHT_ANIMATION_DURATION_MS) {
242 static uint32_t time;
243 static int led_number;
244 static bool led_decrement = false;
245
246 // Three-LED sweep
247 PHAL_writeGPIO(HEARTBEAT_GPIO_Port, HEARTBEAT_Pin, 0);
248 PHAL_writeGPIO(ERR_LED_GPIO_Port, ERR_LED_Pin, 0);
249 PHAL_writeGPIO(CONN_LED_GPIO_Port, CONN_LED_Pin, 0);
250
251 switch (time++ % 24) {
252 case 0:
253 case 5:
254 PHAL_writeGPIO(HEARTBEAT_GPIO_Port, HEARTBEAT_Pin, 1);
255 break;
256 case 1:
257 case 4:
258 PHAL_writeGPIO(CONN_LED_GPIO_Port, CONN_LED_Pin, 1);
259 break;
260 case 2:
261 case 3:
262 PHAL_writeGPIO(ERR_LED_GPIO_Port, ERR_LED_Pin, 1);
263 break;
264 }
265
266 if (led_number < MAX_NUM_LED && !led_decrement) {
267 led_number++;
268 LED_control(led_number, LED_ON);
269 } else if (led_number >= MAX_NUM_LED && !led_decrement) {
270 led_decrement = true;
271 } else {
272 led_number--;
273 LED_control(led_number, LED_OFF);
274 }
275
276 return;
277 }
278
279 PHAL_toggleGPIO(HEARTBEAT_GPIO_Port, HEARTBEAT_Pin);
280 if ((OS_TICKS - last_can_rx_time_ms) >= CONN_LED_MS_THRESH)
281 PHAL_writeGPIO(CONN_LED_GPIO_Port, CONN_LED_Pin, 0);
282 else
283 PHAL_writeGPIO(CONN_LED_GPIO_Port, CONN_LED_Pin, 1);
284
285 static uint8_t trig;
286 if (trig)
287 CAN_SEND_pdu_can_stats(can_stats.can_peripheral_stats[CAN1_IDX].tx_of,
288 can_stats.can_peripheral_stats[CAN1_IDX].tx_fail,
289 can_stats.rx_of,
290 can_stats.can_peripheral_stats[CAN1_IDX].rx_overrun);
291 trig = !trig;
292}

303void send_iv_readings() {
304 // Set LV Batt faults
305 // set_fault(FAULT_ID_PDU_LV_BATT_FIFTY, auto_switches.voltage.in_24v);
306 update_fault(FAULT_ID_LV_GETTING_LOW, auto_switches.voltage.in_24v);
307 update_fault(FAULT_ID_LV_CRITICAL_LOW, auto_switches.voltage.in_24v);
308 // Send CAN messages containing voltage and current data
309 CAN_SEND_v_rails(auto_switches.voltage.in_24v,
310 auto_switches.voltage.out_5v,
311 auto_switches.voltage.out_3v3,
312 0); // amk_24v removed: mux ch4/5 are current sense, not voltage
313 CAN_SEND_rail_currents(auto_switches.current[CS_24V], auto_switches.current[CS_5V]);
314 CAN_SEND_pump_and_fan_current(auto_switches.current[SW_PUMP_1],
315 auto_switches.current[SW_PUMP_2],
316 auto_switches.current[SW_FAN_1],
317 auto_switches.current[SW_FAN_2]);
318 CAN_SEND_fan_current2(auto_switches.current[SW_FAN_3], auto_switches.current[SW_FAN_4]);
319 CAN_SEND_other_currents(auto_switches.current[SW_SDC],
320 auto_switches.current[SW_HXFAN],
321 auto_switches.current[SW_DASH],
322 auto_switches.current[SW_ABOX],
323 auto_switches.current[SW_MAIN]);
324
325 // DS8626 Rev 10 pg 139
326 // Reference voltage is 3.3
327 // 12 bit adc max is 4095
328 // Voltage at 25 degrees c is 0.76
329 // Average slope is 2.5mV per degree c
330 float vsense = (adc_readings.internal_therm / (float)4095) * 3.3;
331 float temp = ((vsense - 0.76) / 0.0025) + 25;
332 CAN_SEND_pdu_temps((uint16_t)temp);
333}
