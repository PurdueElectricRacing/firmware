204 taskCreate(update_cooling_periodic, 100);
205 taskCreate(send_iv_readings, 500);

207 taskCreate(send_flowrates, 200);
208 schedStart();

212void preflightChecks(void) {
213 static uint8_t state;
214
215 switch (state++) {
216 case 0:
217 if (!PHAL_initCAN(CAN1, false, VCAN_BAUD_RATE)) {
218 HardFault_Handler();
219 }
220 NVIC_EnableIRQ(CAN1_RX0_IRQn);
221 break;
222 case 1:
223 CAN_library_init();
224 break;
225 case 2:
226 if (!PHAL_SPI_init(&spi_config)) {
227 HardFault_Handler();
228 }
229 PHAL_writeGPIO(LED_CTRL_BLANK_GPIO_Port, LED_CTRL_BLANK_Pin, 1);
230 break;
231 case 3:
232 fanControlInit();
233 break;
234 case 4:
235 coolingInit();
236 flowRateInit();
237 break;
238 case 5:
239 break;
240 default:
241 if (led_anim_complete) {
242 // Initialize default 'ON' rails
243 setSwitch(SW_SDC, 1);
244 setSwitch(SW_DAQ, 1);
245 setSwitch(SW_NCRIT_5V, 1);
246 setSwitch(SW_MAIN, 1);
247 setSwitch(SW_ABOX, 1);
248 setSwitch(SW_DASH, 1);
249 setSwitch(SW_CRIT_5V, 1);
250 setSwitch(SW_BLT, 1);
251 registerPreflightComplete(1);
252 }
253 state = 255; // prevent wrap around
254 break;
255 }
256}

258void preflight_animation(void) {
259 static uint32_t time;
260 static int led_number;
261 static bool led_decrement = false;
262
263 PHAL_writeGPIO(HEARTBEAT_GPIO_Port, HEARTBEAT_Pin, 0);
264 PHAL_writeGPIO(ERR_LED_GPIO_Port, ERR_LED_Pin, 0);
265 PHAL_writeGPIO(CONN_LED_GPIO_Port, CONN_LED_Pin, 0);
266
267 switch (time++ % 24) {
268 case 0:
269 case 5:
270 PHAL_writeGPIO(HEARTBEAT_GPIO_Port, HEARTBEAT_Pin, 1);
271 break;
272 case 1:
273 case 4:
274 PHAL_writeGPIO(CONN_LED_GPIO_Port, CONN_LED_Pin, 1);
275 break;
276 case 2:
277 case 3:
278 PHAL_writeGPIO(ERR_LED_GPIO_Port, ERR_LED_Pin, 1);
279 break;
280 }
281
282 if (led_number < MAX_NUM_LED && !led_decrement) {
283 led_number++;
284 LED_control(led_number, LED_ON);
285 } else if (led_number >= MAX_NUM_LED && !led_decrement) {
286 led_decrement = true;
287 } else {
288 led_number--;
289 LED_control(led_number, LED_OFF);
290 if (led_number == 0) {
291 led_anim_complete = 1;
292 }
293 }
