76 uint16_t remainder, addr;
77 remainder = 16; /* initialize the PEC */
78 for (uint8_t i = 0; i < len; i++) { /* loops for each byte in data array */
79 addr = (((remainder >> 7) ^ data[i]) & 0xff); /* calculate PEC table address */
80 remainder = ((remainder << 8) ^ adbms_pec15_table[addr]);
