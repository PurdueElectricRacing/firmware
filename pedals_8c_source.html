71 uint16_t throttle2 = 4095 - raw_adc_values.t2; // Invert value for t2 (pull-up resistor)

75 // FSAE 2026 T.4.2.10

77 update_fault(FAULT_ID_APPS_WIRING_T2, throttle2);
78
79 // Hard clamp the raw values to the min and max values to account for physical limits
80 throttle1 = clamp(throttle1, THROTTLE1_MIN, THROTTLE1_MAX);
81 throttle2 = clamp(throttle2, THROTTLE2_MIN, THROTTLE2_MAX);
82 brake1 = clamp(brake1, BRAKE1_MIN, BRAKE1_MAX);
83 brake2 = clamp(brake2, BRAKE2_MIN, BRAKE2_MAX);

86 throttle1 = normalize(throttle1, THROTTLE1_MIN, THROTTLE1_MAX);
87 throttle2 = normalize(throttle2, THROTTLE2_MIN, THROTTLE2_MAX);

89 brake2 = normalize(brake2, BRAKE2_MIN, BRAKE2_MAX);

93 pedal_values.brake = brake1;

101 throttle_diff = throttle1 - throttle2;
102 } else {
103 throttle_diff = throttle2 - throttle1;
104 }
105 update_fault(FAULT_ID_APPS_IMPLAUSIBLE, throttle_diff);
106 if (is_latched(FAULT_ID_APPS_IMPLAUSIBLE)) {
