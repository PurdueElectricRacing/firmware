70void pedalsPeriodic(void) {
71 // Get current values (don't want them changing mid-calculation)
72 t1_raw = raw_adc_values.t1;
73 t2_raw = 4095 - raw_adc_values.t2; // Invert value for t2 (pull-up resistor)
74 b1_raw = raw_adc_values.b1;
75 b2_raw = raw_adc_values.b2;
76
77 // Check for wiring faults
78 update_fault(FAULT_ID_DASHBOARD_APPS_WIRING_T1, t1_raw);
79 update_fault(FAULT_ID_DASHBOARD_APPS_WIRING_T2, t2_raw);
80
81 // Hard clamp the raw values to the min and max values to account for physical limits
82 uint16_t t1_clamped = CLAMP(t1_raw, pedal_calibration.t1_min, pedal_calibration.t1_max);
83 uint16_t t2_clamped = CLAMP(t2_raw, pedal_calibration.t2_min, pedal_calibration.t2_max);
84 uint16_t b1_clamped = CLAMP(b1_raw, pedal_calibration.b1_min, pedal_calibration.b1_max);
85 uint16_t b2_clamped = CLAMP(b2_raw, pedal_calibration.b2_min, pedal_calibration.b2_max);
86
87 // Normalize pedal signals to the 0-4095 range while preserving a linear relationship
88 t1_final = normalize(t1_clamped, pedal_calibration.t1_min, pedal_calibration.t1_max);
89 t2_final = normalize(t2_clamped, pedal_calibration.t2_min, pedal_calibration.t2_max);
90 b1_final = normalize(b1_clamped, pedal_calibration.b1_min, pedal_calibration.b1_max);
91 b2_final = normalize(b2_clamped, pedal_calibration.b2_min, pedal_calibration.b2_max);
92
93 // If both pedals are pressed, set a fault
94 if ((b1_final >= APPS_BRAKE_THRESHOLD && t1_final >= APPS_THROTTLE_FAULT_THRESHOLD) || (is_latched(FAULT_ID_DASHBOARD_APPS_BRAKE) && t1_final >= APPS_THROTTLE_CLEARFAULT_THRESHOLD)) {
95 // Set APPS to 0
96 t2_final = 0;
97 t1_final = 0;
98 update_fault(FAULT_ID_DASHBOARD_APPS_BRAKE, 1);
99 } else if (t1_final <= APPS_THROTTLE_CLEARFAULT_THRESHOLD) { // Clear fault if throttle is released
100 update_fault(FAULT_ID_DASHBOARD_APPS_BRAKE, 0);
101 }
102
103 // Check for APPS sensor deviations (10%)
104 update_fault(FAULT_ID_DASHBOARD_IMPLAUS_DETECTED, ABS((int16_t)t1_final - (int16_t)t2_final));
105
106 // Update the pedal values for external use
107 pedal_values.throttle = t1_final;
108 pedal_values.brake = b1_final;
109
110 // Send the normalized pedal values to Main and TV
111 CAN_SEND_filt_throttle_brake(t1_final, b1_final);
112}
