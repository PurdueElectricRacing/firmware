11bool PHAL_initADC(ADC_TypeDef* adc, ADCInitConfig_t* config, ADCChannelConfig_t channels[], uint8_t num_channels) {
12 // Enable clock to the selected peripheral
13 RCC->APB2ENR |= (1 << (RCC_APB2ENR_ADC1EN_Pos + config->adc_number - 1)) & (0x7UL << RCC_APB2ENR_ADC1EN_Pos);
14
15 // Set prescaler (todo maintain acceptable bounds)
16 ADC123_COMMON->CCR &= ~(ADC_CCR_ADCPRE_Msk);
17 ADC123_COMMON->CCR |= (config->clock_prescaler << ADC_CCR_ADCPRE_Pos) & ADC_CCR_ADCPRE_Msk;
18 ADC123_COMMON->CCR |= ADC_CCR_TSVREFE; // Enable internal temperature sensor
19
20 // Set conversion mode on regular channels
21 adc->CR2 &= ~(ADC_CR2_CONT);
22 adc->CR1 &= ~(ADC_CR1_DISCEN);
23 config->cont_conv_mode ? (adc->CR2 |= (ADC_CR2_CONT)) : (adc->CR1 |= (ADC_CR1_DISCEN));
24
25 // Set resolution
26 adc->CR1 &= ~(ADC_CR1_RES);
27 adc->CR1 |= (config->resolution << ADC_CR1_RES_Pos) & ADC_CR1_RES_Msk;
28
29 // Set data alignment
30 adc->CR2 &= ~(ADC_CR2_ALIGN);
31 adc->CR2 |= (config->data_align << ADC_CR2_ALIGN_Pos) & ADC_CR2_ALIGN_Msk;
32
33 // Regular channel sequence length
34 adc->SQR1 &= ~(ADC_SQR1_L);
35 adc->SQR1 |= ((num_channels - 1) << ADC_SQR1_L_Pos) & ADC_SQR1_L_Msk;
36
37 // DMA configuration
38 if (config->dma_mode != ADC_DMA_OFF) {
39 // Circular or one shot
40 adc->CR2 |= (ADC_CR2_DMA) | (((config->dma_mode == ADC_DMA_CIRCULAR) << ADC_CR2_DDS_Pos) & ADC_CR2_DDS_Msk);
41 } else {
42 // Disable ADC DMA Mode
43 adc->CR2 &= ~(ADC_CR2_DMA);
44 }
45
46 // Channel configuration
47 for (int i = 0; i < num_channels; i++) {
48 // Configure sample time: https://controllerstech.com/adc-conversion-time-frequency-calculation-in-stm32/
49 if (channels[i].channel < 10) {
50 adc->SMPR2 &= ~(ADC_SMPR2_SMP0_Msk << (ADC_SMPR2_SMP1_Pos * channels[i].channel));
51 adc->SMPR2 |= (channels[i].sampling_time & ADC_SMPR2_SMP0_Msk) << (ADC_SMPR2_SMP1_Pos * channels[i].channel);
52 } else if (channels[i].channel > 9 && channels[i].channel < 19) {
53 adc->SMPR1 &= ~(ADC_SMPR1_SMP10_Msk << (ADC_SMPR1_SMP11_Pos * (channels[i].channel - 10)));
54 adc->SMPR1 |= (channels[i].sampling_time & ADC_SMPR1_SMP10_Msk) << (ADC_SMPR1_SMP11_Pos * (channels[i].channel - 10));
55 }
56
57 // Sequence rank
58 if (channels[i].rank < 7) {
59 adc->SQR3 &= ~(ADC_SQR3_SQ1_Msk << ((channels[i].rank - 1) * 5));
60 adc->SQR3 |= (channels[i].channel & ADC_SQR3_SQ1_Msk) << ((channels[i].rank - 1) * 5);
61 } else if (channels[i].rank < 13) {
62 adc->SQR2 &= ~(ADC_SQR2_SQ7_Msk << ((channels[i].rank - 7) * 5));
63 adc->SQR2 |= (channels[i].channel & ADC_SQR2_SQ7_Msk) << ((channels[i].rank - 7) * 5);
64 } else if (channels[i].rank < 17) {
65 adc->SQR1 &= ~(ADC_SQR1_SQ13_Msk << ((channels[i].rank - 13) * 5));
66 adc->SQR1 |= (channels[i].channel & ADC_SQR1_SQ13_Msk) << ((channels[i].rank - 13) * 5);
67 }
68 }
69 // Enable scan mode to read multiple channels
70 if (num_channels > 1) {
71 adc->CR1 |= ADC_CR1_SCAN;
72 }
73
74 // Wake up from power down if necessary
75 if (!(adc->CR2 & ADC_CR2_ADON_Msk)) {
76 adc->CR2 |= (ADC_CR2_ADON);
77 }
78
79 return true;
80}
