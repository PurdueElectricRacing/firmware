111

113 bus->IER |= CAN_IER_FMPIE0;
114 bus->IER |= CAN_IER_FMPIE1;
115
116 // Enter NORMAL mode
117 bus->MCR &= ~CAN_MCR_INRQ;
118 while ((bus->MSR & CAN_MSR_INAK) && ++timeout < PHAL_CAN_INIT_TIMEOUT)
119 ;
120

224bool PHAL_rxCANMessage(CAN_TypeDef *bus, uint8_t fifo, CanMsgTypeDef_t *msg) {
225 if (fifo > 1) return false;
226 if (fifo == 0 && !(bus->RF0R & CAN_RF0R_FMP0)) return false;
227 if (fifo == 1 && !(bus->RF1R & CAN_RF1R_FMP1)) return false;
228
229 msg->Bus = bus;
230 if (bus->sFIFOMailBox[fifo].RIR & CAN_RI0R_IDE) {
231 msg->IDE = 1;
232 msg->ExtId = bus->sFIFOMailBox[fifo].RIR >> CAN_RI0R_EXID_Pos;
233 } else {
234 msg->IDE = 0;
235 msg->StdId = bus->sFIFOMailBox[fifo].RIR >> CAN_RI0R_STID_Pos;
236 }
237
238 msg->DLC = (bus->sFIFOMailBox[fifo].RDTR & CAN_RDT0R_DLC) >> CAN_RDT0R_DLC_Pos;
239
240 uint32_t low = bus->sFIFOMailBox[fifo].RDLR;
241 uint32_t high = bus->sFIFOMailBox[fifo].RDHR;
242
243 msg->Data[0] = (low >> 0) & 0xFF;
244 msg->Data[1] = (low >> 8) & 0xFF;
245 msg->Data[2] = (low >> 16) & 0xFF;
246 msg->Data[3] = (low >> 24) & 0xFF;
247 msg->Data[4] = (high >> 0) & 0xFF;
248 msg->Data[5] = (high >> 8) & 0xFF;
249 msg->Data[6] = (high >> 16) & 0xFF;
250 msg->Data[7] = (high >> 24) & 0xFF;
251
252 if (fifo == 0) bus->RF0R |= CAN_RF0R_RFOM0;
253 else bus->RF1R |= CAN_RF1R_RFOM1;
254
255 return true;
256}
