14bool PHAL_initDMA(dma_init_t* init) {
15 // Check we aren't going to break the peripheral
16 if (init->mem_to_mem && init->circular) {
17 return false;
18 } else if (init->dir > 1) {
19 return false;
20 } else if (init->priority > 3) {
21 return false;
22 } else if (init->mem_size > 2 || init->periph_size > 2) {
23 return false;
24 }
25
26 // Enable clock in RCC
27 if (init->periph == DMA1) {
28 RCC->AHB1ENR |= RCC_AHB1ENR_DMA1EN;
29 } else if (init->periph == DMA2) {
30 RCC->AHB1ENR |= RCC_AHB1ENR_DMA2EN;
31 } else {
32 return false;
33 }
34
35 // Ensure the stream is disabled, must be in order to configure the DMA control registers
36 init->stream->CR &= ~(DMA_SxCR_EN);
37 while (init->stream->CR &= DMA_SxCR_EN)
38 ;
39
40 // Clear any stream dedicated status flags that may have been set previously
41 switch (init->stream_idx) {
42 case 0:
43 init->periph->LIFCR |= 0x2D;
44 break;
45 case 1:
46 init->periph->LIFCR |= 0x2D << 6;
47 break;
48 case 2:
49 init->periph->LIFCR |= 0x2D << 16;
50 break;
51 case 3:
52 init->periph->LIFCR |= 0x2D << 22;
53 break;
54 case 4:
55 init->periph->HIFCR |= 0x2D;
56 break;
57 case 5:
58 init->periph->HIFCR |= 0x2D << 6;
59 break;
60 case 6:
61 init->periph->HIFCR |= 0x2D << 16;
62 break;
63 case 7:
64 init->periph->HIFCR |= 0x2D << 22;
65 break;
66 }
67
68 // Set peripheral port register address
69 init->stream->PAR = init->periph_addr;
70
71 // Set memory address
72 init->stream->M0AR = init->mem_addr;
73
74 init->stream->CR = 0; //Reset preconfigured CR values
75
76 // Set channel, priority, memory data size
77 init->stream->CR |= (init->dma_chan_request << DMA_SxCR_CHSEL_Pos)
78 | (init->priority << DMA_SxCR_PL_Pos) | (init->mem_size << DMA_SxCR_MSIZE_Pos);
79
80 // Set peripheral data size, memory increment, peripheral increment
81 init->stream->CR |= (init->periph_size << DMA_SxCR_PSIZE_Pos)
82 | (init->mem_inc << DMA_SxCR_MINC_Pos) | (init->periph_inc << DMA_SxCR_PINC_Pos);
83
84 // Set circular mode, direction, transfer error interrupt enable, and transfer complete interrupt enable
85 init->stream->CR |= (init->circular << DMA_SxCR_CIRC_Pos)
86 | (init->dir << DMA_SxCR_DIR_Pos) | (init->tx_isr_en << DMA_SxCR_TCIE_Pos) | (init->tx_isr_en << DMA_SxCR_TEIE_Pos);
87
88 // Set stream memory configuration
89 PHAL_DMA_setTxferLength(init, init->tx_size);
90
91 return true;
92}
