93uint8_t PHAL_flashWriteU32(uint32_t Address, uint32_t value) {
94 uint32_t timeout = 0;
95 uint8_t ret;
96 ret = flashUnlock();
97 if (ret != FLASH_OK)
98 return ret;
99
100 // Ensure word has been cleared
101 if (*(uint32_t*)Address != 0xFFFFFFFF)
102 return FLASH_ADDR_NOT_CLEARED;
103
104 // Wait until not busy
105 while ((FLASH->SR & FLASH_SR_BSY) && ++timeout < PHAL_FLASH_TIMEOUT)
106 __asm__("nop");
107 if (timeout == PHAL_FLASH_TIMEOUT)
108 return FLASH_TIMEOUT;
109 timeout = 0;
110
111 // Set 32 bit parallelism (see RM 0090 Page 85)
112 FLASH->CR &= ~(FLASH_CR_PSIZE_Msk);
113 FLASH->CR |= FLASH_CR_PSIZE_1;
114
115 // Enable flash programming
116 FLASH->CR |= FLASH_CR_PG;
117
118 // Write word
119 *(__IO uint32_t*)Address = value;
120 __DSB();
121
122 while ((FLASH->SR & FLASH_SR_BSY) && ++timeout < PHAL_FLASH_TIMEOUT)
123 __asm__("nop");
124 if (timeout == PHAL_FLASH_TIMEOUT)
125 return FLASH_TIMEOUT;
126
127 // Disable flash programming
128 FLASH->CR &= ~(FLASH_CR_PG);
129 flashLock();
130
131 // Check for proper flash write
132 if (*((uint32_t*)Address) != value) {
133 return FLASH_FAIL;
134 }
135
136 return FLASH_OK;
137}

185uint8_t PHAL_flashErasePage(uint8_t page) {
186 // Validate page request
187 if (page > MAX_FLASH_SECTOR) {
188 return FLASH_FAIL;
189 }
190
191 uint32_t timeout = 0;
192 uint8_t ret;
193 ret = flashUnlock();
194 if (ret != FLASH_OK)
195 return ret;
196
197 // Enable sector erase
198 FLASH->CR |= FLASH_CR_SER;
199
200 // Set the sector to be erased (0-11)
201 FLASH->CR &= ~FLASH_CR_SNB_Msk;
202 FLASH->CR |= ((page) << FLASH_CR_SNB_Pos) & FLASH_CR_SNB_Msk;
203
204 // Execute the erase
205 FLASH->CR |= FLASH_CR_STRT;
206
207 while ((FLASH->SR & FLASH_SR_BSY) && ++timeout < PHAL_FLASH_TIMEOUT)
208 __asm__("nop");
209 if (timeout == PHAL_FLASH_TIMEOUT) {
210 flashLock();
211 return FLASH_TIMEOUT;
212 }
213
214 // Disable page erase and clear the page request
215 FLASH->CR &= ~(FLASH_CR_SER | FLASH_CR_SNB_Msk);
216
217 flashLock();
218
219 return FLASH_OK;
220}

222uint8_t PHAL_flashErase(uint32_t* addr, uint32_t words) {
223 uint8_t ret;
224 if (words == 0)
225 return FLASH_OK;
226 uint32_t* end_addr_inclusive = addr + (words - 1);
227 if ((uint32_t)addr < FLASH_BASE || (uint32_t)end_addr_inclusive > (FLASH_END - 3))
228 return FLASH_FAIL;
229 uint8_t sector_start = flashGetSector((uint32_t)addr);
230 uint8_t sector_end = flashGetSector((uint32_t)end_addr_inclusive);
231 for (uint8_t i = sector_start; i <= sector_end; ++i) {
232 ret = PHAL_flashErasePage(i);
233 if (ret != FLASH_OK)
234 return ret;
235 }
236 return FLASH_OK;
237}
