276static inline bool PHAL_readGPIO(GPIO_TypeDef *bank, uint8_t pin)
277{

288static inline void PHAL_writeGPIO(GPIO_TypeDef *bank, uint8_t pin, bool value)
289{
290 bank->BSRR |= 1 << ((!value << 4) | pin); // BSRR has "set" as bottom 16 bits and "reset" as top 16
291}
