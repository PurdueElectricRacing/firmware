261inline bool PHAL_readGPIO(GPIO_TypeDef *bank, uint8_t pin)
262{
263 return (bank->IDR >> pin) & 0b1;
264}
265
266inline void PHAL_writeGPIO(GPIO_TypeDef *bank, uint8_t pin, bool value)
267{
268 bank->BSRR |= 1 << (pin + (16 * (!value))); // BSRR has "set" as bottom 16 bits and "reset" as top 16
269}
270
271inline void PHAL_toggleGPIO(GPIO_TypeDef *bank, uint8_t pin)
272{
273 PHAL_writeGPIO(bank, pin, !PHAL_readGPIO(bank, pin));
274}
