20uint8_t PHAL_configureClockRates(ClockRateConfig_t* config) {
21 // Nonzero Bit Encoded Error Code Indicates Error
22 uint8_t ret_code = 0;
23
24 /* Configure System Clock */
25 if (config->clock_source == CLOCK_SOURCE_HSE) {
26 ret_code |= (!PHAL_configureHSESystemClock()) << RCC_ERROR_HSE_INIT;
27 } else {
28 ret_code |= (!PHAL_configureHSISystemClock()) << RCC_ERROR_HSI_INIT;
29 }
30
31 /* Configure PLL */
32 if (config->use_pll) {
33 ret_code |= (!PHAL_configurePLLVCO(config->pll_src, config->vco_output_rate_target_hz)) << RCC_ERROR_PLLVCO_INIT;
34 ret_code |= (!PHAL_configurePLLSystemClock(config->system_clock_target_hz)) << RCC_ERROR_PLLSYS_INIT;
35 }
36 /* No PLL */
37 else {
38 if (config->clock_source == CLOCK_SOURCE_HSE) {
39 config->system_clock_target_hz = HSE_CLOCK_RATE_HZ;
40 } else {
41 config->system_clock_target_hz = HSI_CLOCK_RATE_HZ;
42 }
43 }
44
45 ret_code |= (!PHAL_configureAHBClock(config->ahb_clock_target_hz)) << RCC_ERROR_AHB_INIT; // Configure AHB Clock
46 ret_code |= (!PHAL_configureAPB1Clock(config->apb1_clock_target_hz)) << RCC_ERROR_APB1_INIT; // Configure APB1 Clock
47 ret_code |= (!PHAL_configureAPB2Clock(config->apb2_clock_target_hz)) << RCC_ERROR_APB2_INIT; // Configure APB2 Clock
48
49 return ret_code;
50}

52bool PHAL_configurePLLVCO(PLLSrc_t pll_source, uint32_t vco_output_rate_target_hz) {
53 // Ensure range for PLL output is 100Mhz <= PLL_CLK <= 432Mhz
54 vco_output_rate_target_hz = vco_output_rate_target_hz > RCC_MAX_VCO_RATE_HZ ? RCC_MAX_VCO_RATE_HZ : vco_output_rate_target_hz;
55 vco_output_rate_target_hz = vco_output_rate_target_hz < RCC_MIN_VCO_RATE_HZ ? RCC_MIN_VCO_RATE_HZ : vco_output_rate_target_hz;
56
57 // Turn off and wait for PLL to disable
58 RCC->CR &= ~RCC_CR_PLLON;
59 while ((RCC->CR & RCC_CR_PLLRDY))
60 ; // Wait for PLL to turn off
61 RCC->PLLCFGR &= ~(RCC_PLLCFGR_PLLSRC_Msk | RCC_PLLCFGR_PLLN_Msk | RCC_PLLCFGR_PLLM_Msk); // Clear any PLL source, N, and M configuration
62
63 // Select the PLL source and determine a desired input clock rate
64 uint32_t pll_input_f_hz;
65 switch (pll_source) {
66 // 16 MHz HSI
67 case PLL_SRC_HSI16:
68 RCC->PLLCFGR |= RCC_PLLCFGR_PLLSRC_HSI; // Select HSI source
69 while (!(RCC->CR & RCC_CR_HSIRDY)) // Wait for HSI to enable
70 ;
71 pll_input_f_hz = HSI_CLOCK_RATE_HZ; // Set local for PLL input clock rate
72 break;
73 case PLL_SRC_HSE:
74 RCC->PLLCFGR |= RCC_PLLCFGR_PLLSRC_HSE;
75 while (!(RCC->CR & RCC_CR_HSERDY))
76 ;
77 pll_input_f_hz = HSE_CLOCK_RATE_HZ;
78 break;
79 default:
80 return false; // Invalid PLL source
81 }
82
83 /* Search for a possible PLL configuration */
84 uint8_t pll_input_divisor = RCC_MIN_PLL_INPUT_DIVISOR; // PLLM
85 uint8_t pll_output_multiplier = RCC_MIN_PLL_OUTPUT_MULTIPLIER; // PLLN
86 bool valid_rate = false;
87 for (; pll_input_divisor <= RCC_MAX_PLL_INPUT_DIVISOR; pll_input_divisor++) // PLLM must be 2 to 63 (Pg. 227)
88 {
89 // VCO input frequency = PLL input clock frequency / PLLM with 2 <= PLLM <= 63
90 uint32_t pll_vco_in_rate = pll_input_f_hz / pll_input_divisor;
91 if (pll_vco_in_rate < 1000000 || pll_vco_in_rate > 2000000) // VCO input rate must be 1MHz to 2MHz (Pg. 227 PLLM)
92 {
93 continue;
94 }
95
96 // VCO output frequency = VCO input * PLLN
97 for (; pll_output_multiplier <= RCC_MAX_PLL_OUTPUT_MULTIPLIER; pll_output_multiplier++) // PLLN must be 50 to 432 (Pg. 227)
98 {
99 if ((pll_input_f_hz / pll_input_divisor) * pll_output_multiplier == vco_output_rate_target_hz) {
100 valid_rate = true;
101 break;
102 }
103 }
104 if (valid_rate)
105 break;
106 }
107
108 if (!valid_rate)
109 return false; // Unable to find a valid clock rate!
110
111 RCC->PLLCFGR |= ((pll_input_divisor) << RCC_PLLCFGR_PLLM_Pos) & RCC_PLLCFGR_PLLM_Msk; // Set PLLM
112 RCC->PLLCFGR |= ((pll_output_multiplier) << RCC_PLLCFGR_PLLN_Pos) & RCC_PLLCFGR_PLLN_Msk; // Set PLLN
113
114 // Update global variable used to reference the PLL
115 PLLClockRateHz = ((pll_input_f_hz / pll_input_divisor) * pll_output_multiplier);
116
117 SystemCoreClockUpdate(); // Must be called each time the core clock HCLK changes
118 return true;
119}

121bool PHAL_configurePLLSystemClock(uint32_t system_clock_target_hz) {
122 // Ensure sysetm clock target is valid (Must be under 168 MHz)
123 if (system_clock_target_hz > RCC_MAX_SYSCLK_TARGET_HZ) {
124 system_clock_target_hz = RCC_MAX_SYSCLK_TARGET_HZ;
125 }
126
127 // Valid number for PLLP divisor are 2,4,6,8 (2 bit encoded)
128 uint8_t pll_p_divisor = PLLClockRateHz / system_clock_target_hz;
129 if (pll_p_divisor == 0 || pll_p_divisor % 2 != 0 || pll_p_divisor > 8) {
130 return false;
131 }
132
133 uint8_t pll_q_divisor = PLLClockRateHz / 48000000;
134 if (pll_q_divisor <= 1 || pll_q_divisor > 15) {
135 return false;
136 }
137
138 // Set the PLLP and PLLQ divisors
139 RCC->PLLCFGR &= ~(RCC_PLLCFGR_PLLP_Msk | RCC_PLLCFGR_PLLQ_Msk);
140 RCC->PLLCFGR |= (((pll_p_divisor / 2) - 1) << RCC_PLLCFGR_PLLP_Pos) & RCC_PLLCFGR_PLLP_Msk; // Divisor value to PLLP bits (Pg. 227)
141 RCC->PLLCFGR |= (pll_q_divisor << RCC_PLLCFGR_PLLQ_Pos) & RCC_PLLCFGR_PLLQ_Msk;
142
143 __DSB(); // Wait for explicit memory accesses to finish
144
145#if defined STM32F732xx
146 bool enable_overdrive = false;
147 RCC->APB1ENR |= RCC_APB1ENR_PWREN;
148 // Voltage regulator scaling based on system clock, see p. 14: https://www.st.com/resource/en/product_training/STM32F7_System_PWR.pdf
149 if (system_clock_target_hz > 180000000) {
150 PWR->CR1 &= ~PWR_CR1_VOS;
151 PWR->CR1 |= PWR_CR1_VOS;
152 enable_overdrive = true;
153 } else if (system_clock_target_hz > 168000000) {
154 PWR->CR1 &= ~PWR_CR1_VOS; //No need to use overdrive, but select scale 1 on voltage regulator
155 PWR->CR1 |= PWR_CR1_VOS;
156 } else if (system_clock_target_hz > 144000000) {
157 PWR->CR1 &= ~PWR_CR1_VOS; //No need to use overdrive, but select scale 2 on voltage regulator
158 PWR->CR1 |= PWR_CR1_VOS_1;
159 } else {
160 PWR->CR1 &= ~PWR_CR1_VOS; //No need to use overdrive, but select scale 3 on voltage regulator
161 PWR->CR1 |= PWR_CR1_VOS_0;
162 }
163
164#endif
165 RCC->CR |= RCC_CR_PLLON; // Enable PLL
166 while (!(RCC->CR & RCC_CR_PLLRDY))
167 ; // Wait for PLL to turn on
168 __DSB();
169
170//Set Level of Internal Voltage Regulator, see ST RM 0090/0431
171#if defined STM32F732xx
172 if (enable_overdrive) {
173 PWR->CR1 |= PWR_CR1_ODEN;
174 while (!(PWR->CSR1 & PWR_CSR1_ODRDY)) //Wait for regulator output to turn on
175 ;
176 PWR->CR1 |= PWR_CR1_ODSWEN; //Enable Overdrive
177 while (!(PWR->CSR1 & PWR_CSR1_ODSWRDY))
178 ; //Wait for overdrive to turn on
179 }
180#endif
181
182 //Flash latency adjustment, see ST RM 0090 Pg. 80, ST RM 0431 Pg. 69
183 uint32_t flash_acr_temp = FLASH->ACR;
184 flash_acr_temp &= ~(FLASH_ACR_LATENCY_Msk);
185
186 if (system_clock_target_hz >= 210000000)
187 flash_acr_temp |= FLASH_ACR_LATENCY_7WS << FLASH_ACR_LATENCY_Pos;
188 else if (system_clock_target_hz >= 180000000)
189 flash_acr_temp |= FLASH_ACR_LATENCY_6WS << FLASH_ACR_LATENCY_Pos;
190 else if (system_clock_target_hz >= 150000000)
191 flash_acr_temp |= FLASH_ACR_LATENCY_5WS << FLASH_ACR_LATENCY_Pos;
192 else if (system_clock_target_hz >= 120000000)
193 flash_acr_temp |= FLASH_ACR_LATENCY_4WS << FLASH_ACR_LATENCY_Pos;
194 else if (system_clock_target_hz >= 90000000)
195 flash_acr_temp |= FLASH_ACR_LATENCY_3WS << FLASH_ACR_LATENCY_Pos;
196 else if (system_clock_target_hz >= 60000000)
197 flash_acr_temp |= FLASH_ACR_LATENCY_2WS << FLASH_ACR_LATENCY_Pos;
198 else if (system_clock_target_hz >= 30000000)
199 flash_acr_temp |= FLASH_ACR_LATENCY_1WS << FLASH_ACR_LATENCY_Pos;
200 else
201 flash_acr_temp |= FLASH_ACR_LATENCY_0WS << FLASH_ACR_LATENCY_Pos;
202 FLASH->ACR = flash_acr_temp;
203
204 __DSB(); // Wait for explicit memory accesses to finish
205 RCC->CFGR &= ~RCC_CFGR_SW;
206 RCC->CFGR |= RCC_CFGR_SW_PLL; // Set system clock switch register to PLL
207 while ((RCC->CFGR & RCC_CFGR_SWS_PLL) != RCC_CFGR_SWS_PLL) // Wait for PLL to be the new system clock
208 ;
209 __DSB(); // Wait for explicit memory accesses to finish
210
211 SystemCoreClockUpdate(); // Must be called each time the core clock HCLK changes
212 return true;
213}

267bool PHAL_configureAHBClock(uint32_t ahb_clock_target_hz) {
268 // Map a required prescaler to achieve the target AHB clock speed from the SystemCoreClock
269 // HCLK has a maximum frequency of 144MHz VOS = 0 and 168 MHz VOS = 1 RM 0090 pg. 80
270 uint32_t desired_psc = SystemCoreClock / ahb_clock_target_hz;
271 uint32_t sys_clk_div;
272
273 // CPU Clock Prescalar (1,2,4,8,16,64,128,256,512 allowed)
274 switch (desired_psc) // ST RM0090 pg. 230
275 {
276 case 1:
277 sys_clk_div = RCC_CFGR_HPRE_DIV1;
278 break;
279 case 2:
280 sys_clk_div = RCC_CFGR_HPRE_DIV2;
281 break;
282 case 4:
283 sys_clk_div = RCC_CFGR_HPRE_DIV4;
284 break;
285 case 8:
286 sys_clk_div = RCC_CFGR_HPRE_DIV8;
287 break;
288 case 16:
289 sys_clk_div = RCC_CFGR_HPRE_DIV16;
290 break;
291 case 64:
292 sys_clk_div = RCC_CFGR_HPRE_DIV64;
293 break;
294 case 128:
295 sys_clk_div = RCC_CFGR_HPRE_DIV128;
296 break;
297 case 256:
298 sys_clk_div = RCC_CFGR_HPRE_DIV256;
299 break;
300 case 512:
301 sys_clk_div = RCC_CFGR_HPRE_DIV512;
302 break;
303 default:
304 return false; // Return if invalid prescalar value
305 }
306
307 // Modify the RCC->CFGR HPRE bits to select prescaler
308 uint32_t rcc_cfgr_temp = RCC->CFGR;
309 rcc_cfgr_temp &= ~(RCC_CFGR_HPRE_Msk);
310 rcc_cfgr_temp |= (sys_clk_div);
311 RCC->CFGR = rcc_cfgr_temp;
312
313 AHBClockRateHz = ahb_clock_target_hz; // Set global for AHB Clock Rate
314 return true;
315}
