11// Oversample count must be 2,4,8,16,32,64,128,256
12// The shift is automatically set as log2(oversample_count)
13static bool PHAL_configureOversampling(ADCInitConfig_t* config) {
14 ADC_TypeDef* adc = config->periph;
15
16 uint16_t oversample_count = config->oversample;
17 if (oversample_count == ADC_OVERSAMPLE_NONE)
18 return true;
19 if (oversample_count < 2 || oversample_count > 256)
20 return false;
21
22 // Check power of two
23 if ((oversample_count & (oversample_count - 1)) != 0)
24 return false;
25
26 // Map oversample_count to OVSR encoding:
27 // OVSR = log2(oversample_count) - 1 (0 means no oversampling)
28 // e.g. 2 => OVSR=0, 4=>1, 8=>2, 16=>3, 32=>4, 64=>5, 128=>6, 256=>7
29 uint8_t ovsr = 0;
30 uint16_t tmp = oversample_count;
31 while (tmp > 2) {
32 tmp >>= 1;
33 ovsr++;
34 }
35
36 // Shift = log2(oversample_count) (to scale sum to average)
37 uint8_t ovss = 0;
38 tmp = oversample_count;
39 while (tmp > 1) {
40 tmp >>= 1;
41 ovss++;
42 }
43
44 // Clear previous oversampling bits
45 adc->CFGR2 &= ~(ADC_CFGR2_OVSR_Msk | ADC_CFGR2_OVSS_Msk);
46
47 // Set new oversampling ratio and shift
48 adc->CFGR2 |= (ovsr << ADC_CFGR2_OVSR_Pos) | (ovss << ADC_CFGR2_OVSS_Pos);
49
50 return true;
51}

53static bool PHAL_configureADCChannels(ADCInitConfig_t* config, ADCChannelConfig_t channels[], uint8_t num_channels) {
54 ADC_TypeDef* adc = config->periph;
55
56 // Clear all SQR ranks
57 adc->SQR1 = 0;
58 adc->SQR2 = 0;
59 adc->SQR3 = 0;
60 adc->SQR4 = 0;
61
62 // Set number of channels
63 adc->SQR1 |= (num_channels - 1) << ADC_SQR1_L_Pos;
64
65 // Configure sequence and sample time ---
66 for (uint8_t i = 0; i < num_channels; i++) {
67 uint8_t ch = channels[i].channel;
68 uint8_t rank = channels[i].rank;
69 if (rank <= 0)
70 return false;
71 rank = rank - 1; // PHAL does 1-based start
72
73 // Set rank for channel
74 if (rank < 4) {
75 adc->SQR1 &= ~(0b111111 << (6 + rank * 6));
76 adc->SQR1 |= ((uint32_t)ch << (6 + rank * 6));
77 } else if (rank < 9) {
78 adc->SQR2 &= ~(0b111111 << ((rank - 4) * 6));
79 adc->SQR2 |= ((uint32_t)ch << ((rank - 4) * 6));
80 } else if (rank < 14) {
81 adc->SQR3 &= ~(0b111111 << ((rank - 9) * 6));
82 adc->SQR3 |= ((uint32_t)ch << ((rank - 9) * 6));
83 } else if (rank < 16) {
84 adc->SQR4 &= ~(0b111111 << ((rank - 14) * 6));
85 adc->SQR4 |= ((uint32_t)ch << ((rank - 14) * 6));
86 } else {
87 return false;
88 }
89
90 // Set sampling time for channel
91 if (ch <= 9) {
92 adc->SMPR1 &= ~(0b111 << (ch * 3));
93 adc->SMPR1 |= (channels[i].sampling_time << (ch * 3));
94 } else {
95 uint8_t ch2 = ch - 10;
96 adc->SMPR2 &= ~(0b111 << (ch2 * 3));
97 adc->SMPR2 |= (channels[i].sampling_time << (ch2 * 3));
98 }
99 }
100
101 return true;
102}

104bool PHAL_initADC(ADCInitConfig_t* config, ADCChannelConfig_t channels[], uint8_t num_channels) {
105 if (num_channels >= 16)
106 return false;
107
108 ADC_TypeDef* adc = config->periph;
109 if (adc == ADC1 || adc == ADC2) {
110 // Enable clock to the selected peripheral
111 RCC->AHB2ENR |= RCC_AHB2ENR_ADC12EN;
112
113 RCC->CCIPR &= ~RCC_CCIPR_ADC12SEL; // Clear bits
114 RCC->CCIPR |= RCC_CCIPR_ADC12SEL_0; // Select system clock (PCLK) as ADC clock
115
116 ADC12_COMMON->CCR &= ~ADC_CCR_CKMODE;
117 ADC12_COMMON->CCR |= (0x1UL << ADC_CCR_CKMODE_Pos);
118 ADC12_COMMON->CCR &= ~ADC_CCR_PRESC_Msk;
119 ADC12_COMMON->CCR |= (config->prescaler << ADC_CCR_PRESC_Pos) & ADC_CCR_PRESC_Msk;
120 } else if (adc == ADC3 || adc == ADC4 || adc == ADC5) {
121 RCC->AHB2ENR |= RCC_AHB2ENR_ADC345EN;
122
123 ADC345_COMMON->CCR &= ~ADC_CCR_CKMODE;
124 ADC345_COMMON->CCR |= (0x1UL << ADC_CCR_CKMODE_Pos);
125 ADC345_COMMON->CCR &= ~(ADC_CCR_PRESC_Msk);
126 ADC345_COMMON->CCR |= (config->prescaler << ADC_CCR_PRESC_Pos) & ADC_CCR_PRESC_Msk;
127 } else {
128 return false;
129 }
130
131 adc->CR &= ~ADC_CR_DEEPPWD;
132 adc->CR |= ADC_CR_ADVREGEN;
133 for (volatile int i = 0; i < 1000; ++i)
134 ; // Short delay
135
136 // 1. Ensure ADC is disabled before calibration
137 if (adc->CR & ADC_CR_ADEN) {
138 adc->CR |= ADC_CR_ADDIS; // Disable ADC if it was enabled
139 while (adc->CR & ADC_CR_ADEN)
140 ; // Wait until disabled
141 }
142
143 // Calibrate ADC
144 adc->CR |= ADC_CR_ADCAL;
145 while (adc->CR & ADC_CR_ADCAL)
146 ; // Wait for calibration to finish
147
148 // Set conversion mode on regular channels
149 adc->CFGR &= ~(ADC_CFGR_CONT | ADC_CFGR_DISCEN);
150 config->cont_conv_mode ? (adc->CFGR |= (ADC_CFGR_CONT)) : (adc->CFGR |= (ADC_CFGR_DISCEN));
151
152 // Set resolution
153 adc->CFGR &= ~(ADC_CFGR_RES);
154 adc->CFGR |= (config->resolution << ADC_CFGR_RES_Pos) & ADC_CFGR_RES_Msk; // 12-bit resolution
155
156 // Set data alignment
157 adc->CFGR &= ~(ADC_CFGR_ALIGN);
158 adc->CFGR |= (config->data_align << ADC_CFGR_ALIGN_Pos) & ADC_CFGR_ALIGN_Msk;
159
160 if (!PHAL_configureADCChannels(config, channels, num_channels))
161 return false;
162
163 if (!PHAL_configureOversampling(config))
164 return false;
165
166 adc->CFGR &= ~(ADC_CFGR_CONT | ADC_CFGR_DMAEN | ADC_CFGR_DMACFG);
167 if (config->dma_mode == ADC_DMA_ONESHOT) {
168 adc->CFGR |= ADC_CFGR_DMAEN | ADC_CFGR_DMACFG;
169 } else if (config->dma_mode == ADC_DMA_CIRCULAR) {
170 adc->CFGR |= ADC_CFGR_CONT | ADC_CFGR_DMAEN | ADC_CFGR_DMACFG;
171 }
172
173 // Enable ADC
174 adc->ISR |= ADC_ISR_ADRDY; // Clear ready flag
175 adc->CR |= ADC_CR_ADEN; // Enable ADC
176 while (!(adc->ISR & ADC_ISR_ADRDY))
177 ; // Wait until ready
178
179 return true;
180}
