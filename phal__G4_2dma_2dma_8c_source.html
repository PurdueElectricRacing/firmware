14bool PHAL_initDMA(dma_init_t* dma) {
15 // Check we aren't going to break the peripheral
16 if (dma->mem_to_mem && dma->circular) {
17 return false;
18 } else if (dma->dir > 1) {
19 return false;
20 } else if (dma->priority > 3) {
21 return false;
22 } else if (dma->mem_size > 2 || dma->periph_size > 2) {
23 return false;
24 }
25
26 // Enable clock in RCC
27 if (dma->periph == DMA1) {
28 dma->channel = (DMA_Channel_TypeDef *)((uint32_t)DMA1_Channel1 + 0x14U * (dma->channel_idx - 1U));
29 RCC->AHB1ENR |= RCC_AHB1ENR_DMA1EN | RCC_AHB1ENR_DMAMUX1EN;
30 } else if (dma->periph == DMA2) {
31 dma->channel = (DMA_Channel_TypeDef *)((uint32_t)DMA2_Channel1 + 0x14U * (dma->channel_idx - 1U));
32 RCC->AHB1ENR |= RCC_AHB1ENR_DMA2EN | RCC_AHB1ENR_DMAMUX1EN;
33 } else {
34 return false;
35 }
36
37 // Ensure the stream is disabled, must be in order to configure the DMA control registers
38 dma->channel->CCR &= ~(DMA_CCR_EN);
39 while (dma->channel->CCR & DMA_CCR_EN)
40 ;
41
42 // Clear any stream dedicated status flags that may have been set previously
43 dma->periph->IFCR = (DMA_ISR_GIF1 | DMA_ISR_TCIF1 | DMA_ISR_HTIF1 | DMA_ISR_TEIF1)
44 << (4 * (dma->channel_idx - 1));
45
46 // Set peripheral address (src)
47 dma->channel->CPAR = dma->periph_addr;
48 // Set memory address (dst)
49 dma->channel->CMAR = dma->mem_addr;
50 dma->channel->CNDTR = dma->tx_size;
51
52 // Reset preconfigured CR values
53 dma->channel->CCR = 0;
54 // Set channel, priority, memory data size
55 dma->channel->CCR |= (dma->mem_size << DMA_CCR_MSIZE_Pos) & DMA_CCR_MSIZE_Msk
56 | (dma->periph_size << DMA_CCR_PSIZE_Pos) & DMA_CCR_PSIZE_Msk
57 | (dma->priority << DMA_CCR_PL_Pos) & DMA_CCR_PL_Msk
58 | (dma->mem_inc << DMA_CCR_MINC_Pos) & DMA_CCR_MINC_Msk
59 | (dma->periph_inc << DMA_CCR_PINC_Pos) & DMA_CCR_PINC_Msk
60 | (dma->circular << DMA_CCR_CIRC_Pos) & DMA_CCR_CIRC_Msk
61 | (dma->dir << DMA_CCR_DIR_Pos) & DMA_CCR_DIR_Msk
62 | (dma->tx_isr_en << DMA_CCR_TEIE_Pos) & DMA_CCR_TEIE_Msk
63 | (dma->tx_isr_en << DMA_CCR_TCIE_Pos) & DMA_CCR_TCIE_Msk
64 | (dma->mem_to_mem << DMA_CCR_MEM2MEM_Pos) & DMA_CCR_MEM2MEM_Msk;
65
66 /* DMA Mux */
67 // For category 3 and category 4 devices:
68 // DMAMUX channels 0 to 7 are connected to DMA1 channels 1 to 8
69 // DMAMUX channels 8 to 15 are connected to DMA2 channels 1 to 8
70 // DMAMUX Channel (usually equal to DMA channel number - 1)
71 DMAMUX_Channel_TypeDef* mux;
72 mux = (DMAMUX1_Channel0 + dma->channel_idx - 1);
73 mux->CCR = (mux->CCR & ~0x7F) | dma->mux_request;
74
75 return true;
76}
