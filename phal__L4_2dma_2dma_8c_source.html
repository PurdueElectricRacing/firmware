13bool PHAL_initDMA(dma_init_t* init) {
14 // Check we aren't going to break the peripheral
15 if (init->mem_to_mem && init->circular) {
16 return false;
17 } else if (init->dir > 1) {
18 return false;
19 } else if (init->priority > 3) {
20 return false;
21 } else if (init->mem_size > 2 || init->periph_size > 2) {
22 return false;
23 }
24
25 // Enable clock in RCC
26 if (init->periph == DMA1) {
27 RCC->AHB1ENR |= RCC_AHB1ENR_DMA1EN;
28 } else if (init->periph == DMA2) {
29 RCC->AHB1ENR |= RCC_AHB1ENR_DMA2EN;
30 } else {
31 return false;
32 }
33
34 // Channel configuration set
35 init->channel->CCR &= ~(DMA_CCR_EN_Msk);
36 init->channel->CCR &= ~(0x3FFF);
37
38 init->channel->CCR |= (init->mem_to_mem << 14) | (init->priority << 12) | (init->mem_size << 10);
39 init->channel->CCR |= (init->periph_size << 8) | (init->mem_inc << 7) | (init->periph_inc << 6);
40 init->channel->CCR |= (init->circular << 5) | (init->dir << 4) | (1U << 3) | (init->tx_isr_en << 1);
41
42 // Channel memory configuration set
43 PHAL_DMA_setTxferLength(init, init->tx_size);
44
45 init->channel->CPAR = init->periph_addr;
46 init->channel->CMAR = init->mem_addr;
47
48 init->request->CSELR &= ~((0xf) << ((init->channel_idx - 1) * 4));
49 init->request->CSELR |= (init->dma_chan_request << ((init->channel_idx - 1) * 4));
50
51 return true;
52}
