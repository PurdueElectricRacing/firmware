44uint8_t PHAL_flashWriteU64(uint32_t Address, uint64_t Data) {
45 uint32_t timeout = 0;
46 uint8_t ret;
47 ret = flashUnlock();
48 if (ret != FLASH_OK)
49 return ret;
50
51 /* Check the parameters */
52 if (*((uint32_t*)Address) != 0xFFFFFFFF)
53 return FLASH_ADDR_NOT_CLEARED;
54 if (*((uint32_t*)(Address + 4)) != 0xFFFFFFFF)
55 return FLASH_ADDR_NOT_CLEARED;
56
57 // Wait until not busy
58 while ((FLASH->SR & FLASH_SR_BSY) && ++timeout < PHAL_FLASH_TIMEOUT)
59 __asm__("nop");
60 if (timeout == PHAL_FLASH_TIMEOUT)
61 return FLASH_TIMEOUT;
62 timeout = 0;
63
64 // Enable flash programming
65 FLASH->CR |= FLASH_CR_PG;
66
67 /* Program first word */
68 *(__IO uint32_t*)Address = (uint32_t)Data;
69
70 /* Barrier to ensure programming is performed in 2 steps, in right order
71 (independently of compiler optimization behavior) */
72 __ISB();
73
74 /* Program second word */
75 *(__IO uint32_t*)(Address + 4U) = (uint32_t)(Data >> 32);
76 __asm__("nop");
77 __asm__("nop");
78
79 while ((FLASH->SR & FLASH_SR_BSY) && ++timeout < PHAL_FLASH_TIMEOUT)
80 __asm__("nop");
81 if (timeout == PHAL_FLASH_TIMEOUT)
82 return FLASH_TIMEOUT;
83
84 // Disable flash programming
85 FLASH->CR &= ~(FLASH_CR_PG);
86 flashLock();
87
88 // Check for proper flash write
89 if (*((uint64_t*)Address) != Data || FLASH->CR & 0x1C3FA) {
90 return FLASH_FAIL;
91 }
92
93 return FLASH_OK;
94}

96uint8_t PHAL_flashErasePage(uint16_t page) {
97 // Validate page request
98 if (page > MAX_FLASH_SECTOR) {
99 return FLASH_FAIL;
100 }
101
102 uint32_t timeout = 0;
103 uint8_t ret;
104 ret = flashUnlock();
105 if (ret != FLASH_OK)
106 return ret;
107
108 // Enable page erase
109 FLASH->CR |= FLASH_CR_PER;
110
111 // Set bank if defined
112#ifdef FLASH_CR_BKER
113 FLASH->CR &= ~FLASH_CR_BKER;
114 if (page > FLASH_BANK_PG_COUNT) {
115 page -= FLASH_BANK_PG_COUNT;
116 FLASH->CR |= FLASH_CR_BKER;
117 }
118#endif
119 FLASH->CR &= ~FLASH_CR_PNB_Msk;
120 FLASH->CR |= ((page) << FLASH_CR_PNB_Pos) & FLASH_CR_PNB_Msk;
121
122 // Execute the erase
123 FLASH->CR |= FLASH_CR_STRT;
124
125 while ((FLASH->SR & FLASH_SR_BSY) && ++timeout < PHAL_FLASH_TIMEOUT)
126 __asm__("nop");
127 if (timeout == PHAL_FLASH_TIMEOUT)
128 return FLASH_TIMEOUT;
129
130 // Disable page erase and clear the page request
131 FLASH->CR &= ~(FLASH_CR_PER | FLASH_CR_PNB_Msk);
132
133 flashLock();
134
135 return FLASH_OK;
136}
