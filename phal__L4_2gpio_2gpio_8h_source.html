222inline void PHAL_writeGPIO(GPIO_TypeDef* bank, uint8_t pin, bool value) {
223 bank->BSRR |= 1 << (pin + (16 * (!value))); // BSRR has "set" as bottom 16 bits and "reset" as top 16
