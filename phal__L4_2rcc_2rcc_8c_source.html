27uint8_t PHAL_configureClockRates(ClockRateConfig_t* config) {
28 uint8_t ret_code = 0;
29 switch (config->system_source) {
30 case SYSTEM_CLOCK_SRC_PLL:
31 if (config->pll_src == PLL_SRC_MSI)
32 ret_code |= !PHAL_configureMSIClock(config->msi_output_rate_target_hz);
33 ret_code |= (!PHAL_configurePLLVCO(config->pll_src, config->vco_output_rate_target_hz)) << 7;
34 ret_code |= (!PHAL_configurePLLSystemClock(config->system_clock_target_hz)) << 6;
35 break;
36 case SYSTEM_CLOCK_SRC_HSE: // TODO: Implement
37 return 0xFF;
38 case SYSTEM_CLOCK_SRC_HSI: // TODO: Implement
39 ret_code |= (!PHAL_configureHSISystemClock()) << 4;
40 config->system_clock_target_hz = 16000000;
41 break;
42 default:
43 return 0xFF;
44 }
45
46 ret_code |= (!PHAL_configureAHBClock(config->ahb_clock_target_hz)) << 0;
47 ret_code |= (!PHAL_configureAPB1Clock(config->apb1_clock_target_hz)) << 1;
48 ret_code |= (!PHAL_configureAPB2Clock(config->apb2_clock_target_hz)) << 2;
49 if (ret_code == 0)
50 return 0;

55bool PHAL_configurePLLVCO(PLLSrc_t pll_source, uint32_t vco_output_rate_target_hz) {
56 // Valid range for PLL output is 64Mhx <= PLL_CLK <= 344Mhz
57 vco_output_rate_target_hz = vco_output_rate_target_hz > (uint32_t)344e6 ? (uint32_t)344e6 : vco_output_rate_target_hz;
58 vco_output_rate_target_hz = vco_output_rate_target_hz < (uint32_t)64e6 ? (uint32_t)64e6 : vco_output_rate_target_hz;
59
60 RCC->CR &= ~RCC_CR_PLLON;
61 while ((RCC->CR & RCC_CR_PLLRDY))
62 ; // Wait for PLL to turn off
63
64 RCC->PLLCFGR &= ~(RCC_PLLCFGR_PLLSRC_Msk | RCC_PLLCFGR_PLLN_Msk | RCC_PLLCFGR_PLLM_Msk);
65
66 /* Determine input clock rate from the selected source */
67 uint32_t pll_input_f_hz;
68 switch (pll_source) {
69 case PLL_SRC_HSI16:
70 RCC->PLLCFGR |= RCC_PLLCFGR_PLLSRC_HSI;
71 RCC->CR |= RCC_CR_HSION;
72 while (!(RCC->CR & RCC_CR_HSIRDY))
73 ; // Wait for HSI to turn on
74 pll_input_f_hz = HSI_CLOCK_RATE_HZ;
75 break;
76
77 case PLL_SRC_MSI:
78 RCC->PLLCFGR |= RCC_PLLCFGR_PLLSRC_MSI;
79 pll_input_f_hz = MSIClockRateHz;
80 break;
81
82 case PLL_SRC_HSE:
83 if (HSE_CLOCK_RATE_HZ == HSE_CLOCK_RATE_HZ_INVALID) {
84 return false; // Make sure you set the HSE_CLOCK_RATE_HZ to what is on the PCB!
85 }
86 RCC->PLLCFGR |= RCC_PLLCFGR_PLLSRC_HSE;
87 pll_input_f_hz = HSE_CLOCK_RATE_HZ;
88 break;
89 }
90
91 /* Search for a possible PLL configuration */
92 uint8_t pll_input_divisor = 1;
93 uint8_t pll_output_multiplier = 8;
94 bool valid_rate = false;
95 for (; pll_input_divisor <= 8; pll_input_divisor++) {
96 // VCO input clock is only allowed between 4MHz and 16MHz
97 uint32_t pll_vco_in_rate = pll_input_f_hz / pll_input_divisor;
98 if (pll_vco_in_rate < 4000000 || pll_vco_in_rate > 16000000) {
99 continue;
100 }
101
102 for (; pll_output_multiplier <= 86; pll_output_multiplier++) {
103 if ((pll_input_f_hz / pll_input_divisor) * pll_output_multiplier == vco_output_rate_target_hz) {
104 valid_rate = true;
105 break;
106 }
107 }
108 if (valid_rate)
109 break;
110 }
111
112 if (!valid_rate) {
113 return false; // Unable to find a valid clock rate!
114 }
115
116 RCC->PLLCFGR |= ((pll_input_divisor - 1) << RCC_PLLCFGR_PLLM_Pos) & RCC_PLLCFGR_PLLM_Msk;
117 RCC->PLLCFGR |= ((pll_output_multiplier) << RCC_PLLCFGR_PLLN_Pos) & RCC_PLLCFGR_PLLN_Msk;
118
119 // Update global variable used to reference the PLL
120 PLLClockRateHz = (pll_input_f_hz / pll_input_divisor) * pll_output_multiplier;
121
122 RCC->PLLCFGR |= RCC_PLLCFGR_PLLPEN;
123 RCC->CR |= RCC_CR_PLLON;
124 while (!(RCC->CR & RCC_CR_PLLRDY))
125 ; // Wait for PLL to turn on
126
127 SystemCoreClockUpdate();
128 return true;
129}

131bool PHAL_configurePLLSystemClock(uint32_t system_clock_target_hz) {
132 if (system_clock_target_hz > 80000000) {
133 system_clock_target_hz = 80000000;
134 }
135
136 // Valid number for PLLR divisor are 2,4,6,8 (2 bit encoded)
137 uint8_t pll_r_divisor = PLLClockRateHz / system_clock_target_hz;
138 if (pll_r_divisor == 0 || pll_r_divisor % 2 != 0 || pll_r_divisor > 8) {
139 return false;
140 }
141
142 // Enable PLLR and set the PLLR divisor
143 RCC->PLLCFGR |= RCC_PLLCFGR_PLLREN;
144 RCC->PLLCFGR &= ~RCC_PLLCFGR_PLLR_Msk;
145 RCC->PLLCFGR |= (((pll_r_divisor / 2) - 1) << RCC_PLLCFGR_PLLR_Pos) & RCC_PLLCFGR_PLLR_Msk;
146
147 // Voltge Scaling
148 __DSB();
149 PWR->CR1 |= (PWR_CR1_VOS_0 << PWR_CR1_VOS_Pos) & PWR_CR1_VOS_Msk;
150 while (PWR->SR2 & PWR_SR2_VOSF)
151 ; // Wait for flag to clear
152 __DSB();
153
154 // Flash latency adjustment, see ST RM0395 Table 9
155 uint32_t flash_acr_temp = FLASH->ACR;
156 flash_acr_temp &= ~(FLASH_ACR_LATENCY_Msk);
157 if (system_clock_target_hz >= 80000000)
158 flash_acr_temp |= FLASH_ACR_LATENCY_4WS << FLASH_ACR_LATENCY_Pos;
159 else if (system_clock_target_hz >= 64000000)
160 flash_acr_temp |= FLASH_ACR_LATENCY_3WS << FLASH_ACR_LATENCY_Pos;
161 else if (system_clock_target_hz >= 48000000)
162 flash_acr_temp |= FLASH_ACR_LATENCY_2WS << FLASH_ACR_LATENCY_Pos;
163 else if (system_clock_target_hz >= 32000000)
164 flash_acr_temp |= FLASH_ACR_LATENCY_1WS << FLASH_ACR_LATENCY_Pos;
165 else if (system_clock_target_hz >= 16000000)
166 flash_acr_temp |= FLASH_ACR_LATENCY_0WS << FLASH_ACR_LATENCY_Pos;
167
168 FLASH->ACR = flash_acr_temp;
169
170 __DSB();
171 RCC->CFGR |= RCC_CFGR_SW_PLL;
172 while ((RCC->CFGR & RCC_CFGR_SWS_PLL != RCC_CFGR_SWS_PLL))
173 ; // Wait for PLL to be the new system clock
174 __DSB();
175
176 SystemCoreClockUpdate();
177 return true;
178}

180bool PHAL_configureHSISystemClock() {
181 // Enable and wait for HSI to enable
182 RCC->CR |= RCC_CR_HSION;
183 while (!(RCC->CR & RCC_CR_HSIRDY))
184 ;
185
186 __DSB();
187 PWR->CR1 |= (PWR_CR1_VOS_0 << PWR_CR1_VOS_Pos) & PWR_CR1_VOS_Msk;
188 while (PWR->SR2 & PWR_SR2_VOSF)
189 ; // Wait for flag to clear
190 __DSB();
191

195 flash_acr_temp |= FLASH_ACR_LATENCY_0WS << FLASH_ACR_LATENCY_Pos;
196 FLASH->ACR = flash_acr_temp;
