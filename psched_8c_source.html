104// @funcname: schedInit()
105//
106// @brief: Initializes the scheduler
107//
108// @param: freq: Frequency of MCU in Hz
109void schedInit(uint32_t freq) {
110 /*
111 Note: This system uses timer 7 and the watchdog peripheral
112 If you want/need to use this timer, it is on you to edit the configuration
113 to use a different timer.
114 DO NOT ATTEMPT TO CONFIGURE THIS TIMER IN CUBE (or on your own)!
115 The configuration for this timer is done manually, right here.
116 The watchdog will trigger a reset if all loops take longer than 10 ms to return
117 Also, functions need to return to work properly. The scheduler works on a timer interrupt.
118 If the functions called in the timer interrupt, you're going to have a stack overflow.
119
120 Frequencies are given in OS ticks (1 ms)
121 */
122
123 // Configure timer 2
124 // Targeting an interrupt every 1 ms
125 if (RCC->CFGR & RCC_CFGR_PPRE1_2)
126 freq *= 2; // RM0394 pg 188 (timer clock doubles if apb1 prescaler != 0)
127#if (defined(STM32F407xx) || defined(STM32F732xx))
128 RCC->APB1ENR |= RCC_APB1ENR_TIM7EN;
129#else
130 RCC->APB1ENR1 |= RCC_APB1ENR1_TIM7EN;
131#endif
132 TIM7->PSC = (freq / 1000000) - 1;
133 TIM7->ARR = ARR_SET;
134 TIM7->CR1 &= ~(TIM_CR1_DIR);
135 TIM7->DIER |= TIM_DIER_UIE;
136
137 // Default all values
138 memsetu((uint8_t *)&sched, 0, sizeof(sched));
139 sched.of = freq;
140}

145void schedStart() {

185// @funcname: waitMicros()
186//
187// @brief: Waits for set time in microseconds,
188// but never for more than 100μs at a time
189//
190// @param: time: time to wait in μs
191//
192// @note: Think long and hard about the use of
193// this function before you slap it everywhere
194void waitMicros(uint8_t time) {
195 if (time > 100)
196 time = 100;
197
198 // ARR is 1k, so no cast issues
199 uint16_t entry_time = TIM7->CNT;
200 int16_t exit_time = (int16_t)entry_time - time;
201
202 if (exit_time < 0) {
203 exit_time += 1000;
204 while (TIM7->CNT < entry_time)
205 ;
206 }
207
208 while (TIM7->CNT > exit_time)
209 ;
210}

226// @funcname: schedLoop()
227//
228// @brief: Main loop that'll run each task
229static void schedLoop() {
230 // Locals
231 uint8_t i;
232
233 while (sched.running == 1) {
234 // Prep iteration
235 sched.run_next = 0;
236 IWDG->KR = 0xAAAA;
237
238 sched.fg_time.tick_entry = sched.os_ticks;
239 sched.fg_time.cnt_entry = TIM7->CNT;
240
241 // Execute tasks
242 if (sched.preflight_required && (!sched.anim_complete || !sched.preflight_complete)) {
243 if (sched.os_ticks % sched.anim_time == 0) {
244 sched.anim();
245 }
246
247 sched.preflight();
248 checkPreflightEnd();
249 } else {
250 for (i = 0; i < sched.fg_count; i++) {
251 if ((sched.os_ticks - sched.ind_fg_time[i].tick_entry) > sched.task_time[i]) {
252 sched.ind_fg_time[i].tick_entry = sched.os_ticks;
253 sched.ind_fg_time[i].cnt_entry = TIM7->CNT;
254 (*sched.task_pointer[i])();
255 sched.ind_fg_time[i].tick_exit = sched.os_ticks;
256 sched.ind_fg_time[i].cnt_exit = TIM7->CNT;
257 }
258 }
259 }
260
261 sched.fg_time.tick_exit = sched.os_ticks;
262 sched.fg_time.cnt_exit = TIM7->CNT;
263
264 // Check if we missed timing requirements
265 if (sched.run_next == 1) {
266 ++sched.skips;
267 }
268
269 schedBg();
270
271 calcTime(&sched.fg_time, 1, E_FG_MISS);
272 calcTime(&sched.bg_time, 1, E_BG_MISS);
273 calcTime(sched.ind_fg_time, sched.fg_count, E_IND_FG_MISS);
274 }
275}

318static void calcTime(cpu_time_t *time, uint8_t count, int type) {
319 uint8_t error;
320 uint8_t e_cnt;
321 uint8_t i;
322
323 e_cnt = 0;
324
325 for (i = 0; i < count; i++) {
326 error = (type != E_BG_MISS) ? time[i].tick_entry != time[i].tick_exit
327 : time[i].tick_exit - time[i].tick_entry > 1;
328
329 if (error) {
330 time[i].cpu_use = 100;
331 time[i].has_missed = 1;
332 ++e_cnt;
333 } else {
334 updateTime(&time[i]);
335 }
336 }
337
338 if (e_cnt) {
339 sched.error |= 1U << type;
340 } else {
341 sched.error &= ~(1U << type);
342 }
343}
