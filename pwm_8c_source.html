77 * PWMfreq = Fclk / ((ARR + 1) Ã— (PSC + 1)).

132 tim->CR1 |= TIM_CR1_CEN; // Enable counter (turn on timer)
133 tim->CR1 |= TIM_CR1_CEN; // Enable tim
134
135 return true;
