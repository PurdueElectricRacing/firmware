29static inline void update_car_state_telemetry() {
30 if (can_data.main_hb.stale) {
31 NXT_setText(CAR_STATE, "STALE");
32 NXT_setFontColor(CAR_STATE, WHITE);
33 return;
34 }
35
36 switch (can_data.main_hb.car_state) {
37 case CAR_STATE_INIT:
38 NXT_setFontColor(CAR_STATE, WHITE);
39 NXT_setText(CAR_STATE, "INIT");
40 NXT_setBorderColor(CAR_STATE, WHITE);
41 break;
42 case CAR_STATE_IDLE:
43 NXT_setFontColor(CAR_STATE, WHITE);
44 NXT_setText(CAR_STATE, "IDLE");
45 NXT_setBorderColor(CAR_STATE, WHITE);
46 break;
47 case CAR_STATE_PRECHARGING:
48 NXT_setFontColor(CAR_STATE, YELLOW);
49 NXT_setText(CAR_STATE, "PRECHRG");
50 NXT_setBorderColor(CAR_STATE, YELLOW);
51 break;
52 case CAR_STATE_ENERGIZED:
53 NXT_setFontColor(CAR_STATE, GREEN);
54 NXT_setText(CAR_STATE, "ENERGZD");
55 NXT_setBorderColor(CAR_STATE, GREEN);
56 break;
57 case CAR_STATE_BUZZING:
58 NXT_setFontColor(CAR_STATE, YELLOW);
59 NXT_setText(CAR_STATE, "BUZZING");
60 NXT_setBorderColor(CAR_STATE, YELLOW);
61 break;
62 case CAR_STATE_READY2DRIVE:
63 NXT_setFontColor(CAR_STATE, GREEN);
64 NXT_setText(CAR_STATE, "R2D");
65 NXT_setBorderColor(CAR_STATE, GREEN);
66 break;
67 case CAR_STATE_FATAL:
68 NXT_setFontColor(CAR_STATE, RED);
69 NXT_setText(CAR_STATE, "FATAL");
70 NXT_setBorderColor(CAR_STATE, RED);
71 break;
72 default:
73 NXT_setFontColor(CAR_STATE, WHITE);
74 NXT_setText(CAR_STATE, "UNKNOWN");
75 NXT_setBorderColor(CAR_STATE, WHITE);
76 break;
77 }
