15uint8_t PHAL_configureRTC(RTC_timestamp_t* initial_time, bool force_time) {
16 // Enable the LSI always (CSR is reset unlike the BDCR register)
17 RCC->CSR |= RCC_CSR_LSION;
18 // TODO: make timeout
19 while (!(RCC->CSR & RCC_CSR_LSIRDY))
20 ;
21
22 // Check if already initialized
23 if (!force_time && RTC->ISR & RTC_ISR_INITS)
24 return true;
25
26 RCC->APB1ENR |= RCC_APB1ENR_PWREN;
27 // Once the RTCCLK clock source has been selected, the only possible way of modifying the selection is to reset the power domain
28 // After system reset, the RTC registers are protected against parasitic write access
29 // with the DBP bit of the PWR power control register (PWR_CR)
30 // The DBP bit must be set to enable RTC registers write access
31#if defined(STM32F407xx)
32 PWR->CR |= PWR_CR_DBP;
33 while ((PWR->CR & PWR_CR_DBP) == 0)
34 ;
35#elif defined(STM32F732xx)
36 PWR->CR1 |= PWR_CR1_DBP;
37 while ((PWR->CR1 & PWR_CR1_DBP) == 0)
38 ;
39#else
40#error "Please define a MCU arch"
41#endif
42
43 // The LSEON, LSEBYP, RTCSEL and RTCEN bits in the RCC Backup domain control register (RCC_BDCR)
44 // TODO: make propper timeout
45 // are in the Backup domain. As a result, after Reset, these bits are write-protected and the DBP bit
46 // has to be set before these can be modified.
47
48 // Software reset backup power domain
49 // RCC->BDCR |= RCC_BDCR_BDRST;
50
51 RCC->BDCR &= ~(RCC_BDCR_RTCSEL); // Clear RTCSEL bits
52 RCC->BDCR |= RCC_BDCR_RTCSEL_1; // select LSI
53 RCC->BDCR |= RCC_BDCR_RTCEN;
54
55 // After backup domain reset, all the RTC registers are write-protected
56 // Writing to the RTC registers is enabled by writing a key into the Write Protection register, RTC_WPR
57
58 // TODO, investigate backup domain reset conditions and handling
59 // For now, just unlock with key from page 803
60 // RTC->WPR |= (0xCA << RTC_WPR_KEY_Pos) & RTC_WPR_KEY_Msk;
61 // RTC->WPR |= (0x53 << RTC_WPR_KEY_Pos) & RTC_WPR_KEY_Msk;
62 RTC->WPR = 0xCA;
63 RTC->WPR = 0x53;
64
65 // Set INIT bit to 1 in the RTC_ISR register to enter initialization mode
66 // In this mode, the calendar counter is stopped and its value can be updated
67 RTC->ISR |= RTC_ISR_INIT;
68
69 // Poll INITF bit of in the RTC_ISR register. The initialization phase mode is entered when
70 // INITF is set to 1. It takes from 1 to 2 RTCCLK clock cycles (due to clock synchronization)
71
72 // TODO: make propper timeout
73 while (false == (RTC->ISR & RTC_ISR_INITF))
74 ;
75
76 // To generate a 1 Hz clock for the calendar counter, program first the synchronous prescaler
77 // factor in RTC_PRER register, and then program the asynchronous prescaler factor
78 // Even if only one of the two fields needs to be changed, 2 separate write accesses
79 // must be performed to the RTC_PRER register
80 RTC->PRER |= (RTC_SYNC_PRESCAL << RTC_PRER_PREDIV_S_Pos) & RTC_PRER_PREDIV_S_Msk;
81 RTC->PRER |= (RTC_ASYNC_PRESCAL << RTC_PRER_PREDIV_A_Pos) & RTC_PRER_PREDIV_A_Msk;
82
83 // Load the initial time and date values in the shadow registers (RTC_TR and RTC_DR), and
84 // configure the time format (12 or 24 hours) through the FMT bit in the RTC_CR register
85
86 RTC->CR |= (RTC_FORMAT_24_HOUR << RTC_CR_FMT_Pos) & RTC_CR_FMT_Msk;
87
88 // The TR, DR registers are write protected. The write access procedure is
89 // described in RTC register write protection on RM 0090 page 803
90 uint8_t yearTens = (initial_time->date.year_bcd >> 4) & 0x0F;
91 uint8_t yearUnits = initial_time->date.year_bcd & 0x0F;
92 uint8_t monthTens = (initial_time->date.month_bcd >> 4) & 0x0F;
93 uint8_t monthUnits = initial_time->date.month_bcd & 0x0F;
94 uint8_t dayTens = (initial_time->date.day_bcd >> 4) & 0x0F;
95 uint8_t dayUnits = initial_time->date.day_bcd & 0x0F;
96
97 uint32_t DR = 0;
98
99 DR |= (yearUnits << RTC_DR_YU_Pos) & RTC_DR_YU_Msk;
100 DR |= (yearTens << RTC_DR_YT_Pos) & RTC_DR_YT_Msk;
101 DR |= (initial_time->date.weekday << RTC_DR_WDU_Pos) & RTC_DR_WDU_Msk;
102 DR |= (monthUnits << RTC_DR_MU_Pos) & RTC_DR_MU_Msk;
103 DR |= (monthTens << RTC_DR_MT_Pos) & RTC_DR_MT_Msk;
104 DR |= (dayUnits << RTC_DR_DU_Pos) & RTC_DR_DU_Msk;
105 DR |= (dayTens << RTC_DR_DT_Pos) & RTC_DR_DT_Msk;
106
107 RTC->DR = DR;
108
109 uint8_t hoursTens = (initial_time->time.hours_bcd >> 4) & 0x0F;
110 uint8_t hoursUnits = initial_time->time.hours_bcd & 0x0F;
111 uint8_t minutesTens = (initial_time->time.minutes_bcd >> 4) & 0x0F;
112 uint8_t minutesUnits = initial_time->time.minutes_bcd & 0x0F;
113 uint8_t secondsTens = (initial_time->time.seconds_bcd >> 4) & 0x0F;
114 uint8_t secondsUnits = initial_time->time.seconds_bcd & 0x0F;
115
116 uint32_t TR = 0;
117
118 TR |= (RTC_FORMAT_24_HOUR << RTC_TR_PM_Pos) & RTC_TR_PM_Msk;
119 TR |= (hoursUnits << RTC_TR_HU_Pos) & RTC_TR_HU_Msk;
120 TR |= (hoursTens << RTC_TR_HT_Pos) & RTC_TR_HT_Msk;
121 TR |= (minutesUnits << RTC_TR_MNU_Pos) & RTC_TR_MNU_Msk;
122 TR |= (minutesTens << RTC_TR_MNT_Pos) & RTC_TR_MNT_Msk;
123 TR |= (secondsUnits << RTC_TR_SU_Pos) & RTC_TR_SU_Msk;
124 TR |= (secondsTens << RTC_TR_ST_Pos) & RTC_TR_ST_Msk;
125
126 RTC->TR = TR;
127
128 // 5. Exit the initialization mode by clearing the INIT bit. The actual calendar counter value is then automatically loaded and the counting restarts after 4 RTCCLK clock cycles.
129 // When the initialization sequence is complete, the calendar starts counting.
130 RTC->ISR &= ~(RTC_ISR_INIT);
131 while (RTC->ISR & RTC_ISR_INITF)
132 ;
133
134 // Relock registers
135#if defined(STM32F407xx)
136 PWR->CR &= ~(PWR_CR_DBP);
137#elif defined(STM32F732xx)
138 PWR->CR1 &= ~(PWR_CR1_DBP);
139#else
140#error "Please define a MCU arch"
141#endif
142
143 return true;
144}

146bool PHAL_getTimeRTC(RTC_timestamp_t* currentTimestamp) {
147 if (RTC->ISR & RTC_ISR_INITS) {
148 uint8_t seconds = (RTC->TR & RTC_TR_SU_Msk) >> RTC_TR_SU_Pos;
149 seconds += ((RTC->TR & RTC_TR_ST_Msk) >> RTC_TR_ST_Pos) * 10;
150
151 uint8_t minutes = (RTC->TR & RTC_TR_MNU_Msk) >> RTC_TR_MNU_Pos;
152 minutes += ((RTC->TR & RTC_TR_MNT_Msk) >> RTC_TR_MNT_Pos) * 10;
153
154 uint8_t hours = (RTC->TR & RTC_TR_HU_Msk) >> RTC_TR_HU_Pos;
155 hours += ((RTC->TR & RTC_TR_HT_Msk) >> RTC_TR_HT_Pos) * 10;
156
157 uint8_t day = (RTC->DR & RTC_DR_DU_Msk) >> RTC_DR_DU_Pos;
158 day += ((RTC->DR & RTC_DR_DT_Msk) >> RTC_DR_DT_Pos) * 10;
159
160 uint8_t month = (RTC->DR & RTC_DR_MU_Msk) >> RTC_DR_MU_Pos;
161 month += ((RTC->DR & RTC_DR_MT_Msk) >> RTC_DR_MT_Pos) * 10;
162
163 uint8_t year = (RTC->DR & RTC_DR_YU_Msk) >> RTC_DR_YU_Pos;
164 year += ((RTC->DR & RTC_DR_YT_Msk) >> RTC_DR_YT_Pos) * 10;
165
166 currentTimestamp->date.year_bcd = year;
167 currentTimestamp->date.month_bcd = month;
168 currentTimestamp->date.day_bcd = day;
169 currentTimestamp->time.hours_bcd = hours;
170 currentTimestamp->time.minutes_bcd = minutes;
171 currentTimestamp->time.seconds_bcd = seconds;
172 currentTimestamp->time.time_format = RTC_FORMAT_24_HOUR;
173
174 return true;
175 }
176 return false;
