38void update_SDC() {
39 static uint8_t sdc_poll_index = 0;
40 const sdc_node_t *current_node = &SDC_NODE_LUT[sdc_poll_index];
41 uint8_t mux_addr = current_node->mux_addr;
42
43 if (mux_addr != SDC_UNREADABLE) {
44 // Set mux control
45 PHAL_writeGPIO(SDC_MUX_S0_PORT, SDC_MUX_S0_PIN, (mux_addr >> 0) & 0x1);
46 PHAL_writeGPIO(SDC_MUX_S1_PORT, SDC_MUX_S1_PIN, (mux_addr >> 1) & 0x1);
47 PHAL_writeGPIO(SDC_MUX_S2_PORT, SDC_MUX_S2_PIN, (mux_addr >> 2) & 0x1);
48 PHAL_writeGPIO(SDC_MUX_S3_PORT, SDC_MUX_S3_PIN, (mux_addr >> 3) & 0x1);
49
50 // delay to allow mux signals to stabilize
51 osDelay(1);
52
53 // ! reading the input pin as 1 = closed, 0 = open
54 bool is_node_open = !PHAL_readGPIO(SDC_MUX_PORT, SDC_MUX_PIN);
55
56 // Read the signal and update the relevant fault state
57 update_fault(current_node->fault_id, is_node_open);
58 }
59
60 // update the poll id for the next cycle (0-16 = sdc 1-17)
61 sdc_poll_index++;
62 if (sdc_poll_index >= NUM_SDC_NODES) sdc_poll_index = 0;
63}
