208SD_Error SD_PowerON(void) {
209 __IO SD_Error errorstatus = SD_OK;
210 uint32_t response = 0, count = 0, validvoltage = 0;
211 uint32_t SDType = SD_STD_CAPACITY;
212 PHAL_SD_Cmd_t cmd;
213
216 cmd.idx = SD_CMD_GO_IDLE_STATE;
217 cmd.res = SD_RESP_NONE;
218 cmd.arg = 0;
219 PHAL_SDIO_SendCommand(&cmd);
220
221 errorstatus = CmdError();
222
223 if (errorstatus != SD_OK) {
225 return (errorstatus);
226 }
227
228 // Wait ~1 ms (If card reset by CMD0, not power on, takes time to reset)
229 for (uint32_t i = 0; i < 128000; ++i)
230 ;
231
238 cmd.idx = SDIO_SEND_IF_COND;
239 cmd.res = SD_RESP_SHORT;
240 cmd.arg = SD_CHECK_PATTERN;
241 PHAL_SDIO_SendCommand(&cmd);
242
243 errorstatus = CmdResp7Error();
244
245 if (errorstatus == SD_OK) {
246 CardType = SDIO_STD_CAPACITY_SD_CARD_V2_0;
247 SDType = SD_HIGH_CAPACITY;
248 } else {
250 cmd.idx = SD_CMD_APP_CMD;
251 cmd.res = SD_RESP_SHORT;
252 cmd.arg = 0;
253 PHAL_SDIO_SendCommand(&cmd);
254 errorstatus = CmdResp1Error(SD_CMD_APP_CMD);
255 }
256
258 cmd.idx = SD_CMD_APP_CMD;
259 cmd.res = SD_RESP_SHORT;
260 cmd.arg = 0;
261 PHAL_SDIO_SendCommand(&cmd);
262 errorstatus = CmdResp1Error(SD_CMD_APP_CMD);
263
267 if (errorstatus == SD_OK) {
270 while ((!validvoltage) && (count < SD_MAX_VOLT_TRIAL)) {
272 cmd.idx = SD_CMD_APP_CMD;
273 cmd.res = SD_RESP_SHORT;
274 cmd.arg = 0;
275 PHAL_SDIO_SendCommand(&cmd);
276
277 errorstatus = CmdResp1Error(SD_CMD_APP_CMD);
278
279 if (errorstatus != SD_OK) {
280 return (errorstatus);
281 }
282
283 cmd.idx = SD_CMD_SD_APP_OP_COND;
284 cmd.res = SD_RESP_SHORT;
285 cmd.arg = SD_VOLTAGE_WINDOW_SD | SDType;
286 PHAL_SDIO_SendCommand(&cmd);
287
288 errorstatus = CmdResp3Error();
289 if (errorstatus != SD_OK) {
290 return (errorstatus);
291 }
292
293 response = SDIO->RESP1;
294 validvoltage = (((response >> 31) == 1) ? 1 : 0);
295 count++;
296 }
297 if (count >= SD_MAX_VOLT_TRIAL) {
298 errorstatus = SD_INVALID_VOLTRANGE;
299 return (errorstatus);
300 }
301
302 if (response &= SD_HIGH_CAPACITY) {
303 CardType = SDIO_HIGH_CAPACITY_SD_CARD;
304 }
305
306 }
308 return (errorstatus);
