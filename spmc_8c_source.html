27int SPMC_enqueue_ISR(
28 SPMC_t *spmc,
29 timestamped_frame_t *incoming_frame
30) {
31 // calc next head and account for wraparound
32 size_t next_head = spmc->head + 1;
33 if (next_head == SPMC_NUM_FRAMES) {
34 next_head = 0;
