49SPMC_status_t SPMC_enqueue_from_ISR(SPMC_t *spmc, timestamped_frame_t *incoming_frame) {
50 // calc next head and account for wraparound
51 size_t next_head = spmc->head + 1;
52 if (next_head == SPMC_CAPACITY) {
53 next_head = 0;
54 }
55
56 // ! the frame is dropped if the buffer is full
57 if (next_head == spmc->master_tail) {
58 spmc->overflows++;
59 return SPMC_FULL;
60 }
61
62 // write the frame into the buffer
63 spmc->data[spmc->head] = *incoming_frame;
64 __DMB(); // do not switch the order of the data write and head update
65 spmc->head = next_head;
66
67 __DSB(); // halt the cpu until write is finished
68
69 return SPMC_OK;
70}

82size_t SPMC_master_peek_all(SPMC_t *spmc, timestamped_frame_t **first_item, size_t *total_unread) {
83 const size_t head = spmc->head;
84 __DMB(); // ensure data visibility of head before reading tail
85 const size_t tail = spmc->master_tail;
86
87 __DMB(); // dont read the buffer before tail/head are ready
88 *first_item = &spmc->data[tail];
89
90 // contiguous case
91 if (head >= tail) {
92 *total_unread = head - tail;
93 return *total_unread;
94 }
95
96 // wraparound case
97 *total_unread = (SPMC_CAPACITY - tail) + head;
98 return (SPMC_CAPACITY - tail);
99}
