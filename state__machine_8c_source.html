94void fsm_periodic() {
95 g_car.current_state = g_car.next_state;
96 g_car.next_state = g_car.current_state; // explicit self loop
97
98 // zero torque request by default
99 g_torque_request.front_right = 0;
100 g_torque_request.front_left = 0;
101 g_torque_request.rear_left = 0;
102 g_torque_request.rear_right = 0;
103
104 // check SDC before doing anything else
105 if (is_fatal_latched()) {
106 g_car.current_state = CARSTATE_FATAL;
107 g_car.next_state = CARSTATE_FATAL;
108 }
109
110 set_brake_light();
111
112 // update precharge status
113 bool precharge_pin = !PHAL_readGPIO(PRECHARGE_COMPLETE_PORT, PRECHARGE_COMPLETE_PIN);
114 update_fault(FAULT_ID_MAIN_MODULE_PRECHARGE_INCOMPLETE, precharge_pin);
115 // amks need a bool to point to for precharge status
116 g_car.is_precharge_complete = !is_latched(FAULT_ID_MAIN_MODULE_PRECHARGE_INCOMPLETE);
117
118 switch (g_car.current_state) {
119 case CARSTATE_INIT: {
120 // do nothing for now
121
122 if (is_init_complete()) {
123 g_car.next_state = CARSTATE_IDLE;
124 }
125 break;
126 }
127 case CARSTATE_IDLE: {
128 // do nothing for now
129
130 if (is_TSMS_closed()) {
131 g_car.next_state = CARSTATE_PRECHARGING;
132 }
133 break;
134 }
135 case CARSTATE_PRECHARGING: {
136 // do nothing for now
137
138 if (is_precharge_complete()) {
139 g_car.next_state = CARSTATE_ENERGIZED;
140 }
141 break;
142 }
143 case CARSTATE_ENERGIZED: {
144 // do nothing for now
145
146 if (is_start_button_pressed() && is_all_AMKS_running()) {
147 g_car.buzzer_start_time = OS_TICKS;
148 g_car.next_state = CARSTATE_BUZZING;
149 }
150 break;
151 }
152 case CARSTATE_BUZZING: {
153 g_car.buzzer_enable = true;
154
155 if (is_buzzing_time_elapsed()) {
156 g_car.buzzer_enable = false; // explicitly turn off the buzzer before transitioning
157 g_car.next_state = CARSTATE_READY2DRIVE;
158 }
159 break;
160 }
161 case CARSTATE_READY2DRIVE: {
162 ready2drive_periodic();
163
164 if (is_start_button_pressed()) {
165 g_car.next_state = CARSTATE_IDLE;
166 }
167 break;
168 }
169 case CARSTATE_FATAL: {
170 // nothing for now
171
172 if (!is_fatal_latched()) {
173 g_car.next_state = CARSTATE_IDLE;
174 }
175 break;
176 }
177 default: { // should never reach here
178 g_car.next_state = CARSTATE_FATAL;
179 break;
180 }
181 }
182
183 AMK_set_torque(&g_car.front_right, g_torque_request.front_right);
184 AMK_set_torque(&g_car.front_left, g_torque_request.front_left);
185 AMK_set_torque(&g_car.rear_left, g_torque_request.rear_left);
186 AMK_set_torque(&g_car.rear_right, g_torque_request.rear_right);
187
188 // flush the internal state
189 PHAL_writeGPIO(BRAKE_LIGHT_PORT, BRAKE_LIGHT_PIN, g_car.brake_light);
190 PHAL_writeGPIO(TSAL_GREEN_CTRL_PORT, TSAL_GREEN_CTRL_PIN, g_car.tsal_green_enable);
191 PHAL_writeGPIO(TSAL_RED_CTRL_PORT, TSAL_RED_CTRL_PIN, g_car.tsal_red_enable);
192 PHAL_writeGPIO(BUZZER_PORT, BUZZER_PIN, g_car.buzzer_enable);
193}
