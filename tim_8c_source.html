79bool PHAL_initPWMChannel(TIM_TypeDef* timer, TimerCCRegister_t chnl, TimerInputMode_t input_source, bool is_falling) {
80 // Input selection
81 if (chnl <= CC2) {
82 timer->CCMR1 &= ~(TIM_CCMR1_CC1S << (chnl - 1) * TIM_CCMR1_CC2S_Pos);
83 timer->CCMR1 |= (input_source & TIM_CCMR1_CC1S) << (chnl - 1) * TIM_CCMR1_CC2S_Pos;
84 } else if (chnl <= CC4) {
85 timer->CCMR2 &= ~(TIM_CCMR2_CC3S << (chnl - 3) * TIM_CCMR2_CC4S_Pos);
86 timer->CCMR2 |= (input_source & TIM_CCMR2_CC3S) << (chnl - 3) * TIM_CCMR2_CC4S_Pos;
87 } else {
88 return false;
89 }
90
91 // polarity
92 timer->CCER &= ~((TIM_CCER_CC1P | TIM_CCER_CC1NP) << (chnl - 1) * 4);
93 if (is_falling) {
94 timer->CCER |= (TIM_CCER_CC2P << (chnl - 1) * 4);
95 }
96
97 // enable channel
98 timer->CCER |= TIM_CCER_CC1E << (chnl - 1) * 4;
99
100 return true;
101}

103bool PHAL_initPWMOut(TIM_TypeDef* timer, uint16_t counter_period, uint16_t ccmr1, uint16_t prescaler) {
104 if (!PHAL_enableTIMClk(timer))
105 return false;
106
107 // auto reload preload CR1 ARPE
108 timer->CR1 |= TIM_CR1_ARPE; // buffered
109 // auto reload value ARR (period)
110 timer->ARR = counter_period - 1;
111 // prescaler PSC (prescaler)
112 timer->PSC = prescaler - 1;
113 // reload for prescaler and rep counter
114 timer->EGR = TIM_EGR_UG;
115
116 // -- channel config --
117 // disable chnl 1 bit clear CCER CC1E
118 timer->CCER &= ~(TIM_CCER_CC1E);
119 // set to output CC1s in CCMR1
120 timer->CCMR1 &= ~(TIM_CCMR1_CC1S);
121 // select OCMode OC1M in CCMR1 (pwm mode)
122 timer->CCMR1 &= ~(TIM_CCMR1_OC1M);
123 timer->CCMR1 |= TIM_CCMR1_OC1M_2 | TIM_CCMR1_OC1M_1;
124 // select output polarity CCER CC1P and CCER CC1N or something
125 timer->CCER &= ~(TIM_CCER_CC1P);
126 // set CCR1 to Pulse
127 timer->CCR1 = ccmr1;
128 // Preload enable bit CCMR1 TIM_CCMR1_OC1PE
129 timer->CCMR1 |= TIM_CCMR1_OC1PE;
130 // CCMR1 (OCFastMode) (leave disabled OC1FE)
131
132 // enable cap compare chnl CCx_ENABLE TIM_CCxChannelCmd
133 timer->CCER |= TIM_CCER_CC1E;
134 //BDTR set MOE
135 timer->BDTR |= TIM_BDTR_MOE;
136
137 return true;
138}
