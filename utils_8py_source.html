74def get_layout_hash(message):
75 """
76 Calculates a fingerprint hash for a message layout
77 """
78 layout_str = ""
79 # Add signal properties that affect layout into a single string
80 # ! does not account for message ID or period, as they do not affect the layout
81 for sig in message.signals:
82 layout_str += f"{sig.name}:{sig.c_type}:{sig.bit_shift}:{sig.length};"
83
84 payload = layout_str.encode('utf-8')
85 full_hash = hashlib.sha256(payload).hexdigest().upper()
86 return f"0x{full_hash[:16]}" # take the first 16 chars (to fit in 64 bits)
