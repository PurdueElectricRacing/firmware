207int vsnprintf(char* str, size_t size, const char* format, va_list args) {
208 INTMAX_T value;
209 unsigned char cvalue;
210 const char* strvalue;
211 INTMAX_T* intmaxptr;
212 PTRDIFF_T* ptrdiffptr;
213 SSIZE_T* sizeptr;
214 LLONG* llongptr;
215 long int* longptr;
216 int* intptr;
217 short int* shortptr;
218 signed char* charptr;
219 size_t len = 0;
220 int overflow = 0;
221 int base = 0;
222 int cflags = 0;
223 int flags = 0;
224 int width = 0;
225 int precision = -1;
226 int state = PRINT_S_DEFAULT;
227 char ch = *format++;
228
229 /*
230 * C99 says: "If `n' is zero, nothing is written, and `s' may be a null
231 * pointer." (7.19.6.5, 2) We're forgiving and allow a NULL pointer
232 * even if a size larger than zero was specified. At least NetBSD's
233 * snprintf(3) does the same, as well as other versions of this file.
234 * (Though some of these versions will write to a non-NULL buffer even
235 * if a size of zero was specified, which violates the standard.)
236 */
237 if (str == NULL && size != 0)
238 size = 0;
239
240 while (ch != '\0')
241 switch (state) {
242 case PRINT_S_DEFAULT:
243 if (ch == '%')
244 state = PRINT_S_FLAGS;
245 else
246 OUTCHAR(str, len, size, ch);
247 ch = *format++;
248 break;
249 case PRINT_S_FLAGS:
250 switch (ch) {
251 case '-':
252 flags |= PRINT_F_MINUS;
253 ch = *format++;
254 break;
255 case '+':
256 flags |= PRINT_F_PLUS;
257 ch = *format++;
258 break;
259 case ' ':
260 flags |= PRINT_F_SPACE;
261 ch = *format++;
262 break;
263 case '#':
264 flags |= PRINT_F_NUM;
265 ch = *format++;
266 break;
267 case '0':
268 flags |= PRINT_F_ZERO;
269 ch = *format++;
270 break;
271 case '\'': /* SUSv2 flag (not in C99). */
272 flags |= PRINT_F_QUOTE;
273 ch = *format++;
274 break;
275 default:
276 state = PRINT_S_WIDTH;
277 break;
278 }
279 break;
280 case PRINT_S_WIDTH:
281 if (ISDIGIT(ch)) {
282 ch = CHARTOINT(ch);
283 if (width > (INT_MAX - ch) / 10) {
284 overflow = 1;
285 goto out;
286 }
287 width = 10 * width + ch;
288 ch = *format++;
289 } else if (ch == '*') {
290 /*
291 * C99 says: "A negative field width argument is
292 * taken as a `-' flag followed by a positive
293 * field width." (7.19.6.1, 5)
294 */
295 if ((width = va_arg(args, int)) < 0) {
296 flags |= PRINT_F_MINUS;
297 width = -width;
298 }
299 ch = *format++;
300 state = PRINT_S_DOT;
301 } else
302 state = PRINT_S_DOT;
303 break;
304 case PRINT_S_DOT:
305 if (ch == '.') {
306 state = PRINT_S_PRECISION;
307 ch = *format++;
308 } else
309 state = PRINT_S_MOD;
310 break;
311 case PRINT_S_PRECISION:
312 if (precision == -1)
313 precision = 0;
314 if (ISDIGIT(ch)) {
315 ch = CHARTOINT(ch);
316 if (precision > (INT_MAX - ch) / 10) {
317 overflow = 1;
318 goto out;
319 }
320 precision = 10 * precision + ch;
321 ch = *format++;
322 } else if (ch == '*') {
323 /*
324 * C99 says: "A negative precision argument is
325 * taken as if the precision were omitted."
326 * (7.19.6.1, 5)
327 */
328 if ((precision = va_arg(args, int)) < 0)
329 precision = -1;
330 ch = *format++;
331 state = PRINT_S_MOD;
332 } else
333 state = PRINT_S_MOD;
334 break;
335 case PRINT_S_MOD:
336 switch (ch) {
337 case 'h':
338 ch = *format++;
339 if (ch == 'h') { /* It's a char. */
340 ch = *format++;
341 cflags = PRINT_C_CHAR;
342 } else
343 cflags = PRINT_C_SHORT;
344 break;
345 case 'l':
346 ch = *format++;
347 if (ch == 'l') { /* It's a long long. */
348 ch = *format++;
349 cflags = PRINT_C_LLONG;
350 } else
351 cflags = PRINT_C_LONG;
352 break;
353 case 'j':
354 cflags = PRINT_C_INTMAX;
355 ch = *format++;
356 break;
357 case 't':
358 cflags = PRINT_C_PTRDIFF;
359 ch = *format++;
360 break;
361 case 'z':
362 cflags = PRINT_C_SIZE;
363 ch = *format++;
364 break;
365 }
366 state = PRINT_S_CONV;
367 break;
368 case PRINT_S_CONV:
369 switch (ch) {
370 case 'd':
371 /* FALLTHROUGH */
372 case 'i':
373 switch (cflags) {
374 case PRINT_C_CHAR:
375 value = (signed char)va_arg(args, int);
376 break;
377 case PRINT_C_SHORT:
378 value = (short int)va_arg(args, int);
379 break;
380 case PRINT_C_LONG:
381 value = va_arg(args, long int);
382 break;
383 case PRINT_C_LLONG:
384 value = va_arg(args, LLONG);
385 break;
386 case PRINT_C_SIZE:
387 value = va_arg(args, SSIZE_T);
388 break;
389 case PRINT_C_INTMAX:
390 value = va_arg(args, INTMAX_T);
391 break;
392 case PRINT_C_PTRDIFF:
393 value = va_arg(args, PTRDIFF_T);
394 break;
395 default:
396 value = va_arg(args, int);
397 break;
398 }
399 fmtint(str, &len, size, value, 10, width, precision, flags);
400 break;
401 case 'X':
402 flags |= PRINT_F_UP;
403 /* FALLTHROUGH */
404 case 'x':
405 base = 16;
406 /* FALLTHROUGH */
407 case 'o':
408 if (base == 0)
409 base = 8;
410 /* FALLTHROUGH */
411 case 'u':
412 if (base == 0)
413 base = 10;
414 flags |= PRINT_F_UNSIGNED;
415 switch (cflags) {
416 case PRINT_C_CHAR:
417 value = (unsigned char)va_arg(args, unsigned int);
418 break;
419 case PRINT_C_SHORT:
420 value = (unsigned short int)va_arg(args, unsigned int);
421 break;
422 case PRINT_C_LONG:
423 value = va_arg(args, unsigned long int);
424 break;
425 case PRINT_C_LLONG:
426 value = va_arg(args, ULLONG);
427 break;
428 case PRINT_C_SIZE:
429 value = va_arg(args, size_t);
430 break;
431 case PRINT_C_INTMAX:
432 value = va_arg(args, UINTMAX_T);
433 break;
434 case PRINT_C_PTRDIFF:
435 value = va_arg(args, UPTRDIFF_T);
436 break;
437 default:
438 value = va_arg(args, unsigned int);
439 break;
440 }
441 fmtint(str, &len, size, value, base, width, precision, flags);
442 break;
443 case 'c':
444 cvalue = va_arg(args, int);
445 OUTCHAR(str, len, size, cvalue);
446 break;
447 case 's':
448 strvalue = va_arg(args, char*);
449 fmtstr(str, &len, size, strvalue, width, precision, flags);
450 break;
451 case 'p':
452 /*
453 * C99 says: "The value of the pointer is
454 * converted to a sequence of printing
455 * characters, in an implementation-defined
456 * manner." (C99: 7.19.6.1, 8)
457 */
458 if ((strvalue = va_arg(args, void*)) == NULL)
459 /*
460 * We use the glibc format. BSD prints
461 * "0x0", SysV "0".
462 */
463 fmtstr(str, &len, size, "(nil)", width, -1, flags);
464 else {
465 /*
466 * We use the BSD/glibc format. SysV
467 * omits the "0x" prefix (which we emit
468 * using the PRINT_F_NUM flag).
469 */
470 flags |= PRINT_F_NUM;
471 flags |= PRINT_F_UNSIGNED;
472 fmtint(str, &len, size, (UINTPTR_T)strvalue, 16, width, precision, flags);
473 }
474 break;
475 case 'n':
476 switch (cflags) {
477 case PRINT_C_CHAR:
478 charptr = va_arg(args, signed char*);
479 *charptr = len;
480 break;
481 case PRINT_C_SHORT:
482 shortptr = va_arg(args, short int*);
483 *shortptr = len;
484 break;
485 case PRINT_C_LONG:
486 longptr = va_arg(args, long int*);
487 *longptr = len;
488 break;
489 case PRINT_C_LLONG:
490 llongptr = va_arg(args, LLONG*);
491 *llongptr = len;
492 break;
493 case PRINT_C_SIZE:
494 /*
495 * C99 says that with the "z" length
496 * modifier, "a following `n' conversion
497 * specifier applies to a pointer to a
498 * signed integer type corresponding to
499 * size_t argument." (7.19.6.1, 7)
500 */
501 sizeptr = va_arg(args, SSIZE_T*);
502 *sizeptr = len;
503 break;
504 case PRINT_C_INTMAX:
505 intmaxptr = va_arg(args, INTMAX_T*);
506 *intmaxptr = len;
507 break;
508 case PRINT_C_PTRDIFF:
509 ptrdiffptr = va_arg(args, PTRDIFF_T*);
510 *ptrdiffptr = len;
511 break;
512 default:
513 intptr = va_arg(args, int*);
514 *intptr = len;
515 break;
516 }
517 break;
518 case '%': /* Print a "%" character verbatim. */
519 OUTCHAR(str, len, size, ch);
520 break;
521 default: /* Skip other characters. */
522 break;
523 }
524 ch = *format++;
525 state = PRINT_S_DEFAULT;
526 base = cflags = flags = width = 0;
527 precision = -1;
528 break;
529 }
530out:
531 if (len < size)
532 str[len] = '\0';
533 else if (size > 0)
534 str[size - 1] = '\0';
535
536 if (overflow || len >= INT_MAX) {
537 return -1;
538 }
539 return (int)len;
540}

542static void fmtstr(char* str, size_t* len, size_t size, const char* value, int width, int precision, int flags) {
543 int padlen, strln; /* Amount to pad. */
544 int noprecision = (precision == -1);
545
546 if (value == NULL) /* We're forgiving. */
547 value = "(null)";
548
549 /* If a precision was specified, don't read the string past it. */
550 for (strln = 0; value[strln] != '\0' && (noprecision || strln < precision); strln++)
551 continue;
552
553 if ((padlen = width - strln) < 0)
554 padlen = 0;
555 if (flags & PRINT_F_MINUS) /* Left justify. */
556 padlen = -padlen;
557
558 while (padlen > 0) { /* Leading spaces. */
559 OUTCHAR(str, *len, size, ' ');
560 padlen--;
561 }
562 while (*value != '\0' && (noprecision || precision-- > 0)) {
563 OUTCHAR(str, *len, size, *value);
564 value++;
565 }
566 while (padlen < 0) { /* Trailing spaces. */
567 OUTCHAR(str, *len, size, ' ');
568 padlen++;
569 }
570}

572static void fmtint(char* str, size_t* len, size_t size, INTMAX_T value, int base, int width, int precision, int flags) {
573 UINTMAX_T uvalue;
574 char iconvert[MAX_CONVERT_LENGTH];
575 char sign = 0;
576 char hexprefix = 0;
577 int spadlen = 0; /* Amount to space pad. */
578 int zpadlen = 0; /* Amount to zero pad. */
579 int pos;
580 int separators = (flags & PRINT_F_QUOTE);
581 int noprecision = (precision == -1);
582
583 if (flags & PRINT_F_UNSIGNED)
584 uvalue = value;
585 else {
586 uvalue = (value >= 0) ? value : -value;
587 if (value < 0)
588 sign = '-';
589 else if (flags & PRINT_F_PLUS) /* Do a sign. */
590 sign = '+';
591 else if (flags & PRINT_F_SPACE)
592 sign = ' ';
593 }
594
595 pos = convert(uvalue, iconvert, sizeof(iconvert), base, flags & PRINT_F_UP);
596
597 if (flags & PRINT_F_NUM && uvalue != 0) {
598 /*
599 * C99 says: "The result is converted to an `alternative form'.
600 * For `o' conversion, it increases the precision, if and only
601 * if necessary, to force the first digit of the result to be a
602 * zero (if the value and precision are both 0, a single 0 is
603 * printed). For `x' (or `X') conversion, a nonzero result has
604 * `0x' (or `0X') prefixed to it." (7.19.6.1, 6)
605 */
606 switch (base) {
607 case 8:
608 if (precision <= pos)
609 precision = pos + 1;
610 break;
611 case 16:
612 hexprefix = (flags & PRINT_F_UP) ? 'X' : 'x';
613 break;
614 }
615 }
616
617 if (separators) /* Get the number of group separators we'll print. */
618 separators = getnumsep(pos);
619
620 zpadlen = precision - pos - separators;
621 spadlen = width /* Minimum field width. */
622 - separators /* Number of separators. */
623 - MAX(precision, pos) /* Number of integer digits. */
624 - ((sign != 0) ? 1 : 0) /* Will we print a sign? */
625 - ((hexprefix != 0) ? 2 : 0); /* Will we print a prefix? */
626
627 if (zpadlen < 0)
628 zpadlen = 0;
629 if (spadlen < 0)
630 spadlen = 0;
631
632 /*
633 * C99 says: "If the `0' and `-' flags both appear, the `0' flag is
634 * ignored. For `d', `i', `o', `u', `x', and `X' conversions, if a
635 * precision is specified, the `0' flag is ignored." (7.19.6.1, 6)
636 */
637 if (flags & PRINT_F_MINUS) /* Left justify. */
638 spadlen = -spadlen;
639 else if (flags & PRINT_F_ZERO && noprecision) {
640 zpadlen += spadlen;
641 spadlen = 0;
642 }
643 while (spadlen > 0) { /* Leading spaces. */
644 OUTCHAR(str, *len, size, ' ');
645 spadlen--;
646 }
647 if (sign != 0) /* Sign. */
648 OUTCHAR(str, *len, size, sign);
649 if (hexprefix != 0) { /* A "0x" or "0X" prefix. */
650 OUTCHAR(str, *len, size, '0');
651 OUTCHAR(str, *len, size, hexprefix);
652 }
653 while (zpadlen > 0) { /* Leading zeros. */
654 OUTCHAR(str, *len, size, '0');
655 zpadlen--;
656 }
657 while (pos > 0) { /* The actual digits. */
658 pos--;
659 OUTCHAR(str, *len, size, iconvert[pos]);
660 if (separators > 0 && pos > 0 && pos % 3 == 0)
661 printsep(str, len, size);
662 }
663 while (spadlen < 0) { /* Trailing spaces. */
664 OUTCHAR(str, *len, size, ' ');
665 spadlen++;
666 }
667}

673static int getnumsep(int digits) {
674 int separators = (digits - ((digits % 3 == 0) ? 1 : 0)) / 3;
675 return separators;
676}

678static int convert(UINTMAX_T value, char* buf, size_t size, int base, int caps) {
679 const char* digits = caps ? "0123456789ABCDEF" : "0123456789abcdef";
680 size_t pos = 0;
681
682 /* We return an unterminated buffer with the digits in reverse order. */
683 do {
684 buf[pos++] = digits[value % base];
685 value /= base;
686 } while (value != 0 && pos < size);
687
688 return (int)pos;
689}
